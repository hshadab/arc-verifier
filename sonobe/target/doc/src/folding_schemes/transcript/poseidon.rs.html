15impl<F: PrimeField + Absorb> Transcript<F> for PoseidonSponge<F> {
16    fn new_with_pp_hash(config: &Self::Config, pp_hash: F) -> Self {
17        let mut sponge = Self::new(config);
18        sponge.absorb(&pp_hash);
19        sponge
20    }
21
22    // Compatible with the in-circuit `TranscriptVar::absorb_point`
23    fn absorb_point<C: CurveGroup<BaseField = F>>(&mut self, p: &C) {
24        let (x, y) = p.into_affine().xy().unwrap_or_default();
25        self.absorb(&x);
26        self.absorb(&y);
27    }
28    fn absorb_nonnative<V: AbsorbNonNative>(&mut self, v: &V) {
29        self.absorb(&v.to_native_sponge_field_elements_as_vec::<F>());
30    }
31    fn get_challenge(&mut self) -> F {
32        let c = self.squeeze_field_elements(1);
33        self.absorb(&c[0]);
34        c[0]
35    }
36    fn get_challenge_nbits(&mut self, nbits: usize) -> Vec<bool> {
37        let bits = self.squeeze_bits(nbits);
38        self.absorb(&F::from(F::BigInt::from_bits_le(&bits)));
39        bits
40    }
41    fn get_challenges(&mut self, n: usize) -> Vec<F> {
42        let c = self.squeeze_field_elements(n);
43        self.absorb(&c);
44        c
45    }
46}

48impl<F: PrimeField> TranscriptVar<F, PoseidonSponge<F>> for PoseidonSpongeVar<F> {
49    fn new_with_pp_hash(
50        config: &Self::Parameters,
51        pp_hash: &FpVar<F>,
52    ) -> Result<Self, SynthesisError> {
53        let mut sponge = Self::new(ConstraintSystemRef::None, config);
54        sponge.absorb(&pp_hash)?;
55        Ok(sponge)
56    }
57
58    fn absorb_point<C: CurveGroup<BaseField = F>, GC: CurveVar<C, F>>(
59        &mut self,
60        v: &GC,
61    ) -> Result<(), SynthesisError> {
62        let mut vec = v.to_constraint_field()?;
63        // The last element in the vector tells whether the point is infinity,
64        // but we can in fact avoid absorbing it without loss of soundness.
65        // This is because the `to_constraint_field` method internally invokes
66        // [`ProjectiveVar::to_afine`](https://github.com/arkworks-rs/r1cs-std/blob/4020fbc22625621baa8125ede87abaeac3c1ca26/src/groups/curves/short_weierstrass/mod.rs#L160-L195),
67        // which guarantees that an infinity point is represented as `(0, 0)`,
68        // but the y-coordinate of a non-infinity point is never 0 (for why, see
69        // https://crypto.stackexchange.com/a/108242 ).
70        vec.pop();
71        self.absorb(&vec)
72    }
73    fn absorb_nonnative<V: AbsorbNonNativeGadget<F>>(
74        &mut self,
75        v: &V,
76    ) -> Result<(), SynthesisError> {
77        self.absorb(&v.to_native_sponge_field_elements()?)
78    }
79    fn get_challenge(&mut self) -> Result<FpVar<F>, SynthesisError> {
80        let c = self.squeeze_field_elements(1)?;
81        self.absorb(&c[0])?;
82        Ok(c[0].clone())
83    }
84
85    /// returns the bit representation of the challenge, we use its output in-circuit for the
86    /// `GC.scalar_mul_le` method.
87    fn get_challenge_nbits(&mut self, nbits: usize) -> Result<Vec<Boolean<F>>, SynthesisError> {
88        let bits = self.squeeze_bits(nbits)?;
89        self.absorb(&Boolean::le_bits_to_fp(&bits)?)?;
90        Ok(bits)
91    }
92    fn get_challenges(&mut self, n: usize) -> Result<Vec<FpVar<F>>, SynthesisError> {
93        let c = self.squeeze_field_elements(n)?;
94        self.absorb(&c)?;
95        Ok(c)
96    }
97}

99/// This Poseidon configuration generator produces a Poseidon configuration with custom parameters
100pub fn poseidon_custom_config<F: PrimeField>(
101    full_rounds: usize,
102    partial_rounds: usize,
103    alpha: u64,
104    rate: usize,
105    capacity: usize,
106) -> PoseidonConfig<F> {
107    let (ark, mds) = find_poseidon_ark_and_mds::<F>(
108        F::MODULUS_BIT_SIZE as u64,
109        rate,
110        full_rounds as u64,
111        partial_rounds as u64,
112        0,
113    );
114
115    PoseidonConfig::new(full_rounds, partial_rounds, alpha, mds, ark, rate, capacity)
116}
117
118/// This Poseidon configuration generator agrees with Circom's Poseidon(4) in the case of BN254's scalar field
119pub fn poseidon_canonical_config<F: PrimeField>() -> PoseidonConfig<F> {
120    // 120 bit security target as in
121    // https://eprint.iacr.org/2019/458.pdf
122    // t = rate + 1
123
124    let full_rounds = 8;
125    let partial_rounds = 60;
126    let alpha = 5;
127    let rate = 4;
128
129    poseidon_custom_config(full_rounds, partial_rounds, alpha, rate, 1)
130}

252    #[test]
253    fn test_transcript_and_transcriptvar_nbits() -> Result<(), Error> {
254        let nbits = crate::constants::NOVA_N_BITS_RO;
255
256        // use 'native' transcript
257        let config = poseidon_canonical_config::<Fq>();
258        let mut tr = PoseidonSponge::<Fq>::new(&config);
259        tr.absorb(&Fq::from(42_u32));
260
261        // get challenge from native transcript
262        let c_bits = tr.get_challenge_nbits(nbits);
263
264        // use 'gadget' transcript
265        let cs = ConstraintSystem::<Fq>::new_ref();
266        let mut tr_var = PoseidonSpongeVar::<Fq>::new(cs.clone(), &config);
267        let v = FpVar::<Fq>::new_witness(cs.clone(), || Ok(Fq::from(42_u32)))?;
268        tr_var.absorb(&v)?;
269
270        // get challenge from circuit transcript
271        let c_var = tr_var.get_challenge_nbits(nbits)?;
272
273        let P = G1::generator();
274        let PVar = GVar::new_witness(cs.clone(), || Ok(P))?;
275
276        // multiply point P by the challenge in different formats, to ensure that we get the same
277        // result natively and in-circuit
278
279        // native c*P
280        let c_Fr = Fr::from_bigint(BigInteger::from_bits_le(&c_bits)).ok_or(Error::OutOfBounds)?;
281        let cP_native = P * c_Fr;
282
283        // native c*P using mul_bits_be (notice the .rev to convert the LE to BE)
284        let cP_native_bits = P.mul_bits_be(c_bits.into_iter().rev());
285
286        // in-circuit c*P using scalar_mul_le
287        let cPVar = PVar.scalar_mul_le(c_var.iter())?;
288
289        // check that they are equal
290        assert_eq!(cP_native.into_affine(), cPVar.value()?.into_affine());
291        assert_eq!(cP_native_bits.into_affine(), cPVar.value()?.into_affine());
292        Ok(())
293    }
