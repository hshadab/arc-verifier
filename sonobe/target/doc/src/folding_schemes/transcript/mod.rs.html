9/// An interface for objects that can be absorbed by a `Transcript`.
10///
11/// Matches `Absorb` in `ark-crypto-primitives`.
12pub trait AbsorbNonNative {
13    /// Converts the object into field elements that can be absorbed by a `Transcript`.
14    /// Append the list to `dest`
15    fn to_native_sponge_field_elements<F: PrimeField>(&self, dest: &mut Vec<F>);
16
17    /// Converts the object into field elements that can be absorbed by a `Transcript`.
18    /// Return the list as `Vec`
19    fn to_native_sponge_field_elements_as_vec<F: PrimeField>(&self) -> Vec<F> {
20        let mut result = Vec::new();
21        self.to_native_sponge_field_elements(&mut result);
22        result
23    }
24}

26/// An interface for objects that can be absorbed by a `TranscriptVar` whose constraint field
27/// is `F`.
28///
29/// Matches `AbsorbGadget` in `ark-crypto-primitives`.
30pub trait AbsorbNonNativeGadget<F: PrimeField> {
31    /// Converts the object into field elements that can be absorbed by a `TranscriptVar`.
32    fn to_native_sponge_field_elements(&self) -> Result<Vec<FpVar<F>>, SynthesisError>;
33}
34
35impl<T: AbsorbNonNative> AbsorbNonNative for [T] {
36    fn to_native_sponge_field_elements<F: PrimeField>(&self, dest: &mut Vec<F>) {
37        for t in self.iter() {
38            t.to_native_sponge_field_elements(dest);
39        }
40    }
41}
42
43impl<F: PrimeField, T: AbsorbNonNativeGadget<F>> AbsorbNonNativeGadget<F> for &T {
44    fn to_native_sponge_field_elements(&self) -> Result<Vec<FpVar<F>>, SynthesisError> {
45        T::to_native_sponge_field_elements(self)
46    }
47}
48
49impl<F: PrimeField, T: AbsorbNonNativeGadget<F>> AbsorbNonNativeGadget<F> for [T] {
50    fn to_native_sponge_field_elements(&self) -> Result<Vec<FpVar<F>>, SynthesisError> {
51        let mut result = Vec::new();
52        for t in self.iter() {
53            result.extend(t.to_native_sponge_field_elements()?);
54        }
55        Ok(result)
56    }
57}

59pub trait Transcript<F: PrimeField>: CryptographicSponge {
60    /// `new_with_pp_hash` creates a new transcript / sponge with the given
61    /// hash of the public parameters.
62    fn new_with_pp_hash(config: &Self::Config, pp_hash: F) -> Self;
63
64    /// `absorb_point` is for absorbing points whose `BaseField` is the field of
65    /// the sponge, i.e., the type `C` of these points should satisfy
66    /// `C::BaseField = F`.
67    ///
68    /// If the sponge field `F` is `C::ScalarField`, call `absorb_nonnative`
69    /// instead.
70    fn absorb_point<C: CurveGroup<BaseField = F>>(&mut self, v: &C);
71    /// `absorb_nonnative` is for structs that contain non-native (field or
72    /// group) elements, including:
73    ///
74    /// - A field element of type `T: PrimeField` that will be absorbed into a
75    ///   sponge that operates in another field `F != T`.
76    /// - A group element of type `C: CurveGroup` that will be absorbed into a
77    ///   sponge that operates in another field `F != C::BaseField`, e.g.,
78    ///   `F = C::ScalarField`.
79    /// - A `CommittedInstance` on the secondary curve (used for CycleFold) that
80    ///   will be absorbed into a sponge that operates in the (scalar field of
81    ///   the) primary curve.
82    ///
83    ///   Note that although a `CommittedInstance` for `AugmentedFCircuit` on
84    ///   the primary curve also contains non-native elements, we still regard
85    ///   it as native, because the sponge is on the same curve.
86    fn absorb_nonnative<V: AbsorbNonNative>(&mut self, v: &V);
87
88    fn get_challenge(&mut self) -> F;
89    /// get_challenge_nbits returns a field element of size nbits
90    fn get_challenge_nbits(&mut self, nbits: usize) -> Vec<bool>;
91    fn get_challenges(&mut self, n: usize) -> Vec<F>;
92}

94pub trait TranscriptVar<F: PrimeField, S: CryptographicSponge>:
95    CryptographicSpongeVar<F, S>
96{
97    /// `new_with_pp_hash` creates a new transcript / sponge with the given
98    /// hash of the public parameters.
99    fn new_with_pp_hash(
100        config: &Self::Parameters,
101        pp_hash: &FpVar<F>,
102    ) -> Result<Self, SynthesisError>;
103
104    /// `absorb_point` is for absorbing points whose `BaseField` is the field of
105    /// the sponge, i.e., the type `C` of these points should satisfy
106    /// `C::BaseField = F`.
107    ///
108    /// If the sponge field `F` is `C::ScalarField`, call `absorb_nonnative`
109    /// instead.
110    fn absorb_point<C: CurveGroup<BaseField = F>, GC: CurveVar<C, F>>(
111        &mut self,
112        v: &GC,
113    ) -> Result<(), SynthesisError>;
114    /// `absorb_nonnative` is for structs that contain non-native (field or
115    /// group) elements, including:
116    ///
117    /// - A field element of type `T: PrimeField` that will be absorbed into a
118    ///   sponge that operates in another field `F != T`.
119    /// - A group element of type `C: CurveGroup` that will be absorbed into a
120    ///   sponge that operates in another field `F != C::BaseField`, e.g.,
121    ///   `F = C::ScalarField`.
122    /// - A `CommittedInstance` on the secondary curve (used for CycleFold) that
123    ///   will be absorbed into a sponge that operates in the (scalar field of
124    ///   the) primary curve.
125    ///
126    ///   Note that although a `CommittedInstance` for `AugmentedFCircuit` on
127    ///   the primary curve also contains non-native elements, we still regard
128    ///   it as native, because the sponge is on the same curve.
129    fn absorb_nonnative<V: AbsorbNonNativeGadget<F>>(
130        &mut self,
131        v: &V,
132    ) -> Result<(), SynthesisError>;
133
134    fn get_challenge(&mut self) -> Result<FpVar<F>, SynthesisError>;
135    /// returns the bit representation of the challenge, we use its output in-circuit for the
136    /// `GC.scalar_mul_le` method.
137    fn get_challenge_nbits(&mut self, nbits: usize) -> Result<Vec<Boolean<F>>, SynthesisError>;
138    fn get_challenges(&mut self, n: usize) -> Result<Vec<FpVar<F>>, SynthesisError>;
139}
