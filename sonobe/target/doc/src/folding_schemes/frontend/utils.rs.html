13/// DummyCircuit is a circuit that has dummy state whose length is specified in the `state_len`
14/// parameter, without any constraints.
15#[derive(Clone, Debug)]
16pub struct DummyCircuit {
17    state_len: usize,
18}
19impl<F: PrimeField> FCircuit<F> for DummyCircuit {
20    type Params = usize;
21    type ExternalInputs = ();
22    type ExternalInputsVar = ();
23
24    fn new(state_len: Self::Params) -> Result<Self, Error> {
25        Ok(Self { state_len })
26    }
27    fn state_len(&self) -> usize {
28        self.state_len
29    }
30    fn generate_step_constraints(
31        &self,
32        cs: ConstraintSystemRef<F>,
33        _i: usize,
34        _z_i: Vec<FpVar<F>>,
35        _external_inputs: Self::ExternalInputsVar,
36    ) -> Result<Vec<FpVar<F>>, SynthesisError> {
37        Vec::new_witness(cs.clone(), || Ok(vec![Zero::zero(); self.state_len]))
38    }
39}

41/// CubicFCircuit is a struct that implements the FCircuit trait, for the R1CS example circuit
42/// from https://www.vitalik.ca/general/2016/12/10/qap.html, which checks `x^3 + x + 5 = y`.
43/// `z_i` is used as `x`, and `z_{i+1}` is used as `y`, and at the next step, `z_{i+1}` will be
44/// assigned to `z_i`, and a new `z+{i+1}` will be computted.
45#[cfg(test)]
46#[derive(Clone, Copy, Debug)]
47pub struct CubicFCircuit<F: PrimeField> {
48    _f: PhantomData<F>,
49}
50
51#[cfg(test)]
52impl<F: PrimeField> FCircuit<F> for CubicFCircuit<F> {
53    type Params = ();
54    type ExternalInputs = ();
55    type ExternalInputsVar = ();
56
57    fn new(_params: Self::Params) -> Result<Self, Error> {
58        Ok(Self { _f: PhantomData })
59    }
60    fn state_len(&self) -> usize {
61        1
62    }
63    fn generate_step_constraints(
64        &self,
65        cs: ConstraintSystemRef<F>,
66        _i: usize,
67        z_i: Vec<FpVar<F>>,
68        _external_inputs: Self::ExternalInputsVar,
69    ) -> Result<Vec<FpVar<F>>, SynthesisError> {
70        let five = FpVar::<F>::new_constant(cs.clone(), F::from(5u32))?;
71        let z_i = z_i[0].clone();
72
73        Ok(vec![&z_i * &z_i * &z_i + &z_i + &five])
74    }
75}

77/// Native implementation of `CubicFCircuit`
78#[cfg(test)]
79pub fn cubic_step_native<F: PrimeField>(z_i: Vec<F>) -> Vec<F> {
80    let z = z_i[0];
81    vec![z * z * z + z + F::from(5)]
82}

84/// CustomFCircuit is a circuit that has the number of constraints specified in the
85/// `n_constraints` parameter. Note that the generated circuit will have very sparse matrices.
86#[derive(Clone, Copy, Debug)]
87pub struct CustomFCircuit<F: PrimeField> {
88    _f: PhantomData<F>,
89    pub n_constraints: usize,
90}
91
92impl<F: PrimeField> FCircuit<F> for CustomFCircuit<F> {
93    type Params = usize;
94    type ExternalInputs = ();
95    type ExternalInputsVar = ();
96
97    fn new(params: Self::Params) -> Result<Self, Error> {
98        Ok(Self {
99            _f: PhantomData,
100            n_constraints: params,
101        })
102    }
103    fn state_len(&self) -> usize {
104        1
105    }
106    fn generate_step_constraints(
107        &self,
108        _cs: ConstraintSystemRef<F>,
109        _i: usize,
110        z_i: Vec<FpVar<F>>,
111        _external_inputs: Self::ExternalInputsVar,
112    ) -> Result<Vec<FpVar<F>>, SynthesisError> {
113        let mut z_i1 = z_i[0].clone();
114        for _ in 0..self.n_constraints - 1 {
115            z_i1 = z_i1.square()?;
116        }
117
118        Ok(vec![z_i1])
119    }
120}

122/// Native implementation of `CubicFCircuit`
123#[cfg(test)]
124pub fn custom_step_native<F: PrimeField>(z_i: Vec<F>, n_constraints: usize) -> Vec<F> {
125    let mut z_i1 = z_i[0];
126    for _ in 0..n_constraints - 1 {
127        z_i1 = z_i1.square();
128    }
129    vec![z_i1]

132/// WrapperCircuit is a circuit that wraps any circuit that implements the FCircuit trait. This
133/// is used to test the `FCircuit.generate_step_constraints` method. This is a similar wrapping
134/// than the one done in the `AugmentedFCircuit`, but without adding all the extra constraints
135/// of the AugmentedF circuit logic, in order to run lighter tests when we're not interested in
136/// the AugmentedF logic but in the wrapping of the circuits.
137pub struct WrapperCircuit<F: PrimeField, FC: FCircuit<F>> {
138    pub FC: FC, // F circuit
139    pub z_i: Option<Vec<F>>,
140    pub z_i1: Option<Vec<F>>,
141}
142
143impl<F: PrimeField, FC: FCircuit<F>> ConstraintSynthesizer<F> for WrapperCircuit<F, FC> {
144    fn generate_constraints(self, cs: ConstraintSystemRef<F>) -> Result<(), SynthesisError> {
145        let z_i =
146            Vec::<FpVar<F>>::new_witness(cs.clone(), || Ok(self.z_i.unwrap_or(vec![F::zero()])))?;
147        let z_i1 =
148            Vec::<FpVar<F>>::new_input(cs.clone(), || Ok(self.z_i1.unwrap_or(vec![F::zero()])))?;
149        let external_inputs =
150            FC::ExternalInputsVar::new_input(cs.clone(), || Ok(FC::ExternalInputs::default()))?;
151        let computed_z_i1 =
152            self.FC
153                .generate_step_constraints(cs.clone(), 0, z_i.clone(), external_inputs)?;
154
155        use ark_r1cs_std::eq::EqGadget;
156        computed_z_i1.enforce_equal(&z_i1)?;
157        Ok(())
158    }
159}
