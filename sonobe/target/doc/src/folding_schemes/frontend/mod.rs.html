9/// FCircuit defines the trait of the circuit of the F function, which is the one being folded (ie.
10/// inside the agmented F' function).
11/// The parameter z_i denotes the current state, and z_{i+1} denotes the next state after applying
12/// the step.
13/// Note that the external inputs for the specific circuit are defined at the implementation of
14/// both `FCircuit::ExternalInputs` and `FCircuit::ExternalInputsVar`, where the `Default` trait
15/// implementation for the `ExternalInputs` returns the initialized data structure (ie. if the type
16/// contains a vector, it is initialized at the expected length).
17pub trait FCircuit<F: PrimeField>: Clone + Debug {
18    type Params: Debug;
19    type ExternalInputs: Clone + Default + Debug;
20    type ExternalInputsVar: Clone + Debug + AllocVar<Self::ExternalInputs, F>;
21
22    /// returns a new FCircuit instance
23    fn new(params: Self::Params) -> Result<Self, Error>;
24
25    /// returns the number of elements in the state of the FCircuit, which corresponds to the
26    /// FCircuit inputs.
27    fn state_len(&self) -> usize;
28
29    /// generates the constraints for the step of F for the given z_i
30    fn generate_step_constraints(
31        // this method uses self, so that each FCircuit implementation (and different frontends)
32        // can hold a state if needed to store data to generate the constraints.
33        &self,
34        cs: ConstraintSystemRef<F>,
35        i: usize,
36        z_i: Vec<FpVar<F>>,
37        external_inputs: Self::ExternalInputsVar, // inputs that are not part of the state
38    ) -> Result<Vec<FpVar<F>>, SynthesisError>;
39}

41#[cfg(test)]
42pub mod tests {
43    use super::*;
44    use ark_bn254::Fr;
45    use ark_relations::gr1cs::{ConstraintSynthesizer, ConstraintSystem};

47    use utils::{custom_step_native, CubicFCircuit, CustomFCircuit, WrapperCircuit};
48
49    #[test]
50    fn test_testfcircuit() -> Result<(), Error> {
51        let cs = ConstraintSystem::<Fr>::new_ref();
52        let F_circuit = CubicFCircuit::<Fr>::new(())?;
53
54        let wrapper_circuit = WrapperCircuit::<Fr, CubicFCircuit<Fr>> {
55            FC: F_circuit,
56            z_i: Some(vec![Fr::from(3_u32)]),
57            z_i1: Some(vec![Fr::from(35_u32)]),
58        };
59        wrapper_circuit.generate_constraints(cs.clone())?;
60        assert_eq!(cs.num_constraints(), 3);
61        Ok(())
62    }
63
64    #[test]
65    fn test_customtestfcircuit() -> Result<(), Error> {
66        let cs = ConstraintSystem::<Fr>::new_ref();
67        let n_constraints = 1000;
68        let custom_circuit = CustomFCircuit::<Fr>::new(n_constraints)?;
69        let z_i = vec![Fr::from(5_u32)];
70        let wrapper_circuit = WrapperCircuit::<Fr, CustomFCircuit<Fr>> {
71            FC: custom_circuit,
72            z_i: Some(z_i.clone()),
73            z_i1: Some(custom_step_native(z_i, n_constraints)),
74        };
75        wrapper_circuit.generate_constraints(cs.clone())?;
76        assert_eq!(cs.num_constraints(), n_constraints);
77        Ok(())
78    }
