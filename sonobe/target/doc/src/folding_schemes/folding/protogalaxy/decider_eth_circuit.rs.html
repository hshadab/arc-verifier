39/// In-circuit representation of the Witness associated to the CommittedInstance.
40#[derive(Debug, Clone)]
41pub struct WitnessVar<F: PrimeField> {
42    pub W: Vec<FpVar<F>>,
43    pub rW: FpVar<F>,
44}
45
46impl<F: PrimeField> AllocVar<Witness<F>, F> for WitnessVar<F> {
47    fn new_variable<T: Borrow<Witness<F>>>(
48        cs: impl Into<Namespace<F>>,
49        f: impl FnOnce() -> Result<T, SynthesisError>,
50        mode: AllocationMode,
51    ) -> Result<Self, SynthesisError> {
52        f().and_then(|val| {
53            let cs = cs.into();
54
55            let W = Vec::new_variable(cs.clone(), || Ok(val.borrow().w.to_vec()), mode)?;
56            let rW = FpVar::new_variable(cs.clone(), || Ok(val.borrow().r_w), mode)?;
57
58            Ok(Self { W, rW })
59        })
60    }
61}
62
63impl<F: PrimeField> WitnessVarOps<F> for WitnessVar<F> {
64    fn get_openings(&self) -> Vec<(&[FpVar<F>], FpVar<F>)> {
65        vec![(&self.W, self.rW.clone())]
66    }
67}

69pub type DeciderEthCircuit<C1, C2> = GenericOnchainDeciderCircuit<
70    C1,
71    C2,
72    CommittedInstance<C1, RUNNING>,
73    CommittedInstance<C1, INCOMING>,
74    Witness<CF1<C1>>,
75    R1CS<CF1<C1>>,
76    R1CSMatricesVar<CF1<C1>, FpVar<CF1<C1>>>,
77    DeciderProtoGalaxyGadget,
78>;
79
80/// returns an instance of the DeciderEthCircuit from the given ProtoGalaxy struct
81impl<
82        C1: Curve,
83        C2: Curve,
84        FC: FCircuit<C1::ScalarField>,
85        CS1: CommitmentScheme<C1, false>,
86        // enforce that the CS2 is Pedersen commitment scheme, since we're at Ethereum's EVM decider
87        CS2: CommitmentScheme<C2, false, ProverParams = PedersenParams<C2>>,
88    > TryFrom<ProtoGalaxy<C1, C2, FC, CS1, CS2>> for DeciderEthCircuit<C1, C2>
89{
90    type Error = Error;
91
92    fn try_from(protogalaxy: ProtoGalaxy<C1, C2, FC, CS1, CS2>) -> Result<Self, Error> {
93        let mut transcript =
94            PoseidonSponge::new_with_pp_hash(&protogalaxy.poseidon_config, protogalaxy.pp_hash);
95
96        let (U_i1, W_i1, proof, aux) = Folding::prove(
97            &mut transcript,
98            &protogalaxy.r1cs,
99            &protogalaxy.U_i,
100            &protogalaxy.W_i,
101            &[protogalaxy.u_i.clone()],
102            &[protogalaxy.w_i.clone()],
103        )?;
104
105        // compute the KZG challenges used as inputs in the circuit
106        let kzg_challenges = KZGChallengesGadget::get_challenges_native(&mut transcript, &U_i1);
107
108        // get KZG evals
109        let kzg_evaluations = W_i1
110            .get_openings()
111            .iter()
112            .zip(&kzg_challenges)
113            .map(|((v, _), &c)| EvalGadget::evaluate_native(v, c))
114            .collect::<Result<Vec<_>, _>>()?;
115
116        Ok(Self {
117            _avar: PhantomData,
118            arith: protogalaxy.r1cs,
119            cf_arith: protogalaxy.cf_r1cs,
120            cf_pedersen_params: protogalaxy.cf_cs_params,
121            poseidon_config: protogalaxy.poseidon_config,
122            pp_hash: protogalaxy.pp_hash,
123            i: protogalaxy.i,
124            z_0: protogalaxy.z_0,
125            z_i: protogalaxy.z_i,
126            U_i: protogalaxy.U_i,
127            W_i: protogalaxy.W_i,
128            u_i: protogalaxy.u_i,
129            w_i: protogalaxy.w_i,
130            U_i1,
131            W_i1,
132            proof,
133            randomness: aux.L_X_evals,
134            cf_U_i: protogalaxy.cf_U_i,
135            cf_W_i: protogalaxy.cf_W_i,
136            kzg_challenges,
137            kzg_evaluations,
138        })
139    }
140}

142pub struct DeciderProtoGalaxyGadget;
143
144impl<C: Curve>
145    DeciderEnabledNIFS<
146        C,
147        CommittedInstance<C, RUNNING>,
148        CommittedInstance<C, INCOMING>,
149        Witness<CF1<C>>,
150        R1CS<CF1<C>>,
151    > for DeciderProtoGalaxyGadget
152{
153    type Proof = ProtoGalaxyProof<CF1<C>>;
154    type ProofDummyCfg = (usize, usize, usize);
155    type Randomness = Vec<CF1<C>>;
156    type RandomnessDummyCfg = usize;
157
158    fn fold_field_elements_gadget(
159        _arith: &R1CS<CF1<C>>,
160        transcript: &mut PoseidonSpongeVar<CF1<C>>,
161        U: CommittedInstanceVar<C, RUNNING>,
162        _U_vec: Vec<FpVar<CF1<C>>>,
163        u: CommittedInstanceVar<C, INCOMING>,
164        proof: Self::Proof,
165        randomness: Self::Randomness,
166    ) -> Result<CommittedInstanceVar<C, RUNNING>, SynthesisError> {
167        let cs = U.e.cs();
168        let F_coeffs = Vec::new_witness(cs.clone(), || Ok(&proof.F_coeffs[..]))?;
169        let K_coeffs = Vec::new_witness(cs.clone(), || Ok(&proof.K_coeffs[..]))?;
170        let randomness = Vec::new_input(cs.clone(), || Ok(randomness))?;
171
172        let (U_next, L_X_evals) =
173            FoldingGadget::fold_committed_instance(transcript, &U, &[u], F_coeffs, K_coeffs)?;
174        L_X_evals.enforce_equal(&randomness)?;
175
176        Ok(U_next)
177    }
178
179    fn fold_group_elements_native(
180        U_commitments: &[C],
181        u_commitments: &[C],
182        _: Option<Self::Proof>,
183        L_X_evals: Self::Randomness,
184    ) -> Result<Vec<C>, Error> {
185        let U_phi = U_commitments[0];
186        let u_phi = u_commitments[0];
187        Ok(vec![U_phi * L_X_evals[0] + u_phi * L_X_evals[1]])
188    }
189}

192pub mod tests {
193    use ark_bn254::{Fr, G1Projective as Projective};
194    use ark_grumpkin::Projective as Projective2;

204    #[test]
205    fn test_decider_circuit() -> Result<(), Error> {
206        let mut rng = ark_std::test_rng();
207        let poseidon_config = poseidon_canonical_config::<Fr>();
208
209        let F_circuit = CubicFCircuit::<Fr>::new(())?;
210        let z_0 = vec![Fr::from(3_u32)];
211
212        type PG = ProtoGalaxy<
213            Projective,
214            Projective2,
215            CubicFCircuit<Fr>,
216            Pedersen<Projective>,
217            Pedersen<Projective2>,
218        >;
219        let pg_params = PG::preprocess(&mut rng, &(poseidon_config, F_circuit))?;
220
221        // generate a Nova instance and do a step of it
222        let mut protogalaxy = PG::init(&pg_params, F_circuit, z_0.clone())?;
223        protogalaxy.prove_step(&mut rng, (), None)?;
224
225        let ivc_proof = protogalaxy.ivc_proof();
226        PG::verify(pg_params.1, ivc_proof)?;
227
228        // load the DeciderEthCircuit from the generated Nova instance
229        let decider_circuit = DeciderEthCircuit::<Projective, Projective2>::try_from(protogalaxy)?;
230
231        let cs = ConstraintSystem::<Fr>::new_ref();
232
233        // generate the constraints and check that are satisfied by the inputs
234        decider_circuit.generate_constraints(cs.clone())?;
235        assert!(cs.is_satisfied()?);
236        dbg!(cs.num_constraints());
237        Ok(())
238    }
