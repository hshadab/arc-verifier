28// Implements the trait for absorbing ProtoGalaxy's CommittedInstance.
29impl<C: Curve, const TYPE: bool> Absorb for CommittedInstance<C, TYPE> {
30    fn to_sponge_bytes(&self, dest: &mut Vec<u8>) {
31        C::ScalarField::batch_to_sponge_bytes(&self.to_sponge_field_elements_as_vec(), dest);
32    }
33
34    fn to_sponge_field_elements<F: PrimeField>(&self, dest: &mut Vec<F>) {
35        self.phi.to_native_sponge_field_elements(dest);
36        self.betas.to_sponge_field_elements(dest);
37        self.e.to_sponge_field_elements(dest);
38        self.x.to_sponge_field_elements(dest);
39    }
40}

42// Implements the trait for absorbing ProtoGalaxy's CommittedInstanceVar in-circuit.
43impl<C: Curve, const TYPE: bool> AbsorbGadget<C::ScalarField> for CommittedInstanceVar<C, TYPE> {
44    fn to_sponge_bytes(&self) -> Result<Vec<UInt8<C::ScalarField>>, SynthesisError> {
45        FpVar::batch_to_sponge_bytes(&self.to_sponge_field_elements()?)
46    }
47
48    fn to_sponge_field_elements(&self) -> Result<Vec<FpVar<C::ScalarField>>, SynthesisError> {
49        Ok([
50            self.phi.to_native_sponge_field_elements()?,
51            self.betas.to_sponge_field_elements()?,
52            self.e.to_sponge_field_elements()?,
53            self.x.to_sponge_field_elements()?,
54        ]
55        .concat())
56    }
57}

59/// Implements [`ArithRelation`] for R1CS, where the witness is of type
60/// [`Witness`], and the committed instance is of type [`CommittedInstance`].
61///
62/// Due to the error term `CommittedInstance.e`, R1CS here is considered as a
63/// relaxed R1CS.
64///
65/// See `nova/traits.rs` for the rationale behind the design.
66impl<C: Curve, const TYPE: bool> ArithRelation<Witness<CF1<C>>, CommittedInstance<C, TYPE>>
67    for R1CS<CF1<C>>
68{
69    type Evaluation = Vec<CF1<C>>;
70
71    fn eval_relation(
72        &self,
73        w: &Witness<CF1<C>>,
74        u: &CommittedInstance<C, TYPE>,
75    ) -> Result<Self::Evaluation, Error> {
76        self.eval_at_z(&[&[C::ScalarField::one()][..], &u.x, &w.w].concat())
77    }
78
79    fn check_evaluation(
80        _w: &Witness<C::ScalarField>,
81        u: &CommittedInstance<C, TYPE>,
82        e: Vec<C::ScalarField>,
83    ) -> Result<(), Error> {
84        let ok = if TYPE == RUNNING {
85            if u.betas.len() != log2(e.len()) as usize {
86                return Err(Error::NotSameLength(
87                    "instance.betas.len()".to_string(),
88                    u.betas.len(),
89                    "log2(e.len())".to_string(),
90                    log2(e.len()) as usize,
91                ));
92            }
93
94            u.e == cfg_into_iter!(e)
95                .enumerate()
96                .map(|(i, e_i)| pow_i(i, &u.betas) * e_i)
97                .sum::<CF1<C>>()
98        } else {
99            is_zero_vec(&e)
100        };
101        ok.then_some(()).ok_or(Error::NotSatisfied)
102    }
103}

105/// Unlike its native counterpart, we only need to support running instances in
106/// circuit, as the decider circuit only checks running instance satisfiability.
107impl<C: Curve> ArithRelationGadget<WitnessVar<CF1<C>>, CommittedInstanceVar<C, RUNNING>>
108    for R1CSMatricesVar<CF1<C>, FpVar<CF1<C>>>
109{
110    type Evaluation = (Vec<FpVar<CF1<C>>>, Vec<FpVar<CF1<C>>>);
111
112    fn eval_relation(
113        &self,
114        w: &WitnessVar<CF1<C>>,
115        u: &CommittedInstanceVar<C, RUNNING>,
116    ) -> Result<Self::Evaluation, SynthesisError> {
117        self.eval_at_z(&[&[FpVar::one()][..], &u.x, &w.W].concat())
118    }
119
120    fn enforce_evaluation(
121        _w: &WitnessVar<C::ScalarField>,
122        u: &CommittedInstanceVar<C, RUNNING>,
123        (AzBz, uCz): Self::Evaluation,
124    ) -> Result<(), SynthesisError> {
125        let mut e = vec![];
126        for (i, (l, r)) in AzBz.iter().zip(uCz).enumerate() {
127            e.push(pow_i_var(i, &u.betas) * (l - r));
128        }
129        // Call `sum` on a vector instead of computing the sum in the above loop
130        // to avoid stack overflow (the cause of this is similar to issue #80
131        // https://github.com/privacy-scaling-explorations/sonobe/issues/80)
132        e.iter().sum::<FpVar<_>>().enforce_equal(&u.e)
133    }
134}

137pub mod tests {
138    use super::*;
139    use ark_bn254::{Fr, G1Projective as Projective};

145    /// test that checks the native CommittedInstance.to_sponge_{bytes,field_elements}
146    /// vs the R1CS constraints version
147    #[test]
148    pub fn test_committed_instance_to_sponge_preimage() -> Result<(), Error> {
149        let mut rng = ark_std::test_rng();
150
151        let t = rng.gen::<u8>() as usize;
152        let io_len = rng.gen::<u8>() as usize;
153
154        let ci = CommittedInstance::<Projective, true> {
155            phi: Projective::rand(&mut rng),
156            betas: (0..t).map(|_| Fr::rand(&mut rng)).collect(),
157            e: Fr::rand(&mut rng),
158            x: (0..io_len).map(|_| Fr::rand(&mut rng)).collect(),
159        };
160
161        let bytes = ci.to_sponge_bytes_as_vec();
162        let field_elements = ci.to_sponge_field_elements_as_vec();
163
164        let cs = ConstraintSystem::<Fr>::new_ref();
165
166        let ciVar =
167            CommittedInstanceVar::<Projective, true>::new_witness(cs.clone(), || Ok(ci.clone()))?;
168        let bytes_var = ciVar.to_sponge_bytes()?;
169        let field_elements_var = ciVar.to_sponge_field_elements()?;
170
171        assert!(cs.is_satisfied()?);
172
173        // check that the natively computed and in-circuit computed hashes match
174        assert_eq!(bytes_var.value()?, bytes);
175        assert_eq!(field_elements_var.value()?, field_elements);
176        Ok(())
177    }
