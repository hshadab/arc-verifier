73impl<C: Curve> CycleFoldConfig<C> for ProtoGalaxyCycleFoldConfig<C> {
74    const RANDOMNESS_BIT_LENGTH: usize = C::ScalarField::MODULUS_BIT_SIZE as usize;
75    const N_UNIQUE_RANDOMNESSES: usize = 2;
76    const N_INPUT_POINTS: usize = 2;
77
78    fn alloc_points(&self, cs: ConstraintSystemRef<CF2<C>>) -> Result<Vec<C::Var>, SynthesisError> {
79        let points = Vec::new_witness(cs.clone(), || Ok(self.points.clone()))?;
80        for point in &points {
81            Self::mark_point_as_public(point)?;
82        }
83        Ok(points)
84    }
85
86    fn alloc_randomnesses(
87        &self,
88        cs: ConstraintSystemRef<CF2<C>>,
89    ) -> Result<Vec<Vec<Boolean<CF2<C>>>>, SynthesisError> {
90        let rs = vec![self.rs[0]]
91            .into_iter()
92            .chain(self.rs.windows(2).map(|r| r[1] / r[0]))
93            .map(|r| {
94                let mut bits = r.into_bigint().to_bits_le();
95                bits.resize(CF1::<C>::MODULUS_BIT_SIZE as usize, false);
96                Vec::new_witness(cs.clone(), || Ok(bits))
97            })
98            .collect::<Result<Vec<_>, _>>()?;
99        Self::mark_randomness_as_public(&rs.concat())?;
100        Ok(rs)
101    }
102}

104/// The committed instance of ProtoGalaxy.
105///
106/// We use `TYPE` to distinguish between incoming and running instances, as
107/// they have slightly different structures (e.g., length of `betas`) and
108/// behaviors (e.g., in satisfiability checks).
109#[derive(Clone, Debug, PartialEq, Eq, CanonicalSerialize, CanonicalDeserialize)]
110pub struct CommittedInstance<C: Curve, const TYPE: bool> {
111    phi: C,
112    betas: Vec<C::ScalarField>,
113    e: C::ScalarField,
114    x: Vec<C::ScalarField>,
115}

117impl<C: Curve, const TYPE: bool> Dummy<(usize, usize)> for CommittedInstance<C, TYPE> {
118    fn dummy((io_len, t): (usize, usize)) -> Self {
119        if TYPE == INCOMING {
120            assert_eq!(t, 0);
121        }
122        Self {
123            phi: C::zero(),
124            betas: vec![Zero::zero(); t],
125            e: Zero::zero(),
126            x: vec![Zero::zero(); io_len],
127        }
128    }
129}

131impl<C: Curve, const TYPE: bool> Dummy<&R1CS<CF1<C>>> for CommittedInstance<C, TYPE> {
132    fn dummy(r1cs: &R1CS<CF1<C>>) -> Self {
133        let t = if TYPE == RUNNING {
134            log2(r1cs.n_constraints()) as usize
135        } else {
136            0
137        };
138        Self::dummy((r1cs.n_public_inputs(), t))
139    }
140}

176impl<C: Curve, const TYPE: bool> AllocVar<CommittedInstance<C, TYPE>, C::ScalarField>
177    for CommittedInstanceVar<C, TYPE>
178{
179    fn new_variable<T: Borrow<CommittedInstance<C, TYPE>>>(
180        cs: impl Into<Namespace<C::ScalarField>>,
181        f: impl FnOnce() -> Result<T, SynthesisError>,
182        mode: AllocationMode,
183    ) -> Result<Self, SynthesisError> {
184        f().and_then(|u| {
185            let cs = cs.into();
186
187            let u = u.borrow();
188
189            Ok(Self {
190                phi: NonNativeAffineVar::new_variable(cs.clone(), || Ok(u.phi), mode)?,
191                betas: Vec::new_variable(cs.clone(), || Ok(u.betas.clone()), mode)?,
192                e: if TYPE == RUNNING {
193                    FpVar::new_variable(cs.clone(), || Ok(u.e), mode)?
194                } else {
195                    FpVar::zero()
196                },
197                x: Vec::new_variable(cs.clone(), || Ok(u.x.clone()), mode)?,
198            })
199        })
200    }

203impl<C: Curve, const TYPE: bool> GR1CSVar<C::ScalarField> for CommittedInstanceVar<C, TYPE> {
204    type Value = CommittedInstance<C, TYPE>;

206    fn cs(&self) -> ConstraintSystemRef<C::ScalarField> {
207        self.phi
208            .cs()
209            .or(self.betas.cs())
210            .or(self.e.cs())
211            .or(self.x.cs())
212    }

214    fn value(&self) -> Result<Self::Value, SynthesisError> {
215        Ok(CommittedInstance {
216            phi: self.phi.value()?,
217            betas: self
218                .betas
219                .iter()
220                .map(|v| v.value())
221                .collect::<Result<_, _>>()?,
222            e: self.e.value()?,
223            x: self.x.iter().map(|v| v.value()).collect::<Result<_, _>>()?,
224        })
225    }

228impl<C: Curve, const TYPE: bool> CommittedInstanceVarOps<C> for CommittedInstanceVar<C, TYPE> {
229    type PointVar = NonNativeAffineVar<C>;

231    fn get_commitments(&self) -> Vec<Self::PointVar> {
232        vec![self.phi.clone()]
233    }

236        &self.x

239    fn enforce_incoming(&self) -> Result<(), SynthesisError> {
240        // We don't need to check if `self` is an incoming instance in-circuit,
241        // because incoming instances and running instances already have
242        // different types of `e` (constant vs witness) when we allocate them

250        self.betas.enforce_equal(&other.betas)?;
251        self.e.enforce_equal(&other.e)?;
252        self.x.enforce_equal(&other.x)
253    }

557impl<C1, C2, FC, CS1, CS2> ProtoGalaxy<C1, C2, FC, CS1, CS2>
558where
559    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
560    C2: Curve,
561    FC: FCircuit<C1::ScalarField>,
562    CS1: CommitmentScheme<C1>,
563    CS2: CommitmentScheme<C2>,
564{
565    /// This method computes the parameter `t` in ProtoGalaxy for folding `F'`,
566    /// the augmented circuit of `F`
567    fn compute_t(
568        poseidon_config: &PoseidonConfig<CF1<C1>>,
569        F: &FC,
570        d: usize,
571        k: usize,
572    ) -> Result<usize, Error> {
573        // In ProtoGalaxy, prover and verifier are parameterized by `t = log(n)`
574        // where `n` is the number of constraints in the circuit (known as the
575        // mapping `f` in the paper).
576        // For IVC, `f` is the augmented circuit `F'`, which not only includes
577        // the original computation `F`, but also the in-circuit verifier of
578        // ProtoGalaxy.
579        // Therefore, `t` depends on the size of `F'`, but the size of `F'` in
580        // turn depends on `t`.
581        // To address this circular dependency, we first find `t_lower_bound`,
582        // the lower bound of `t`. Then we incrementally increase `t` and build
583        // the circuit `F'` with `t` as ProtoGalaxy's parameter, until `t` is
584        // the smallest integer that equals the logarithm of the number of
585        // constraints.
586
587        // For `t_lower_bound`, we configure `F'` with `t = 1` and compute log2
588        // of the size of `F'`.
589        let state_len = F.state_len();
590
591        // `F'` includes `F` and `ProtoGalaxy.V`, where `F` might be costly.
592        // Observing that the cost of `F` is constant with respect to `t`, we
593        // separately compute `step_constraints`, the size of `F`.
594        // Later, we only need to re-run the rest of `F'` with updated `t` to
595        // get the size of `F'`.
596        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
597        cs.set_mode(SynthesisMode::Setup);
598        F.generate_step_constraints(
599            cs.clone(),
600            0,
601            Vec::new_witness(cs.clone(), || Ok(vec![Zero::zero(); state_len]))?,
602            FC::ExternalInputsVar::new_witness(cs.clone(), || Ok(FC::ExternalInputs::default()))?,
603        )?;
604        let step_constraints = cs.num_constraints();
605
606        // Create a dummy circuit with the same state length and external inputs
607        // length as `F`, which replaces `F` in the augmented circuit `F'`.
608        let dummy_circuit: DummyCircuit = FCircuit::<C1::ScalarField>::new(state_len)?;
609
610        // Compute `augmentation_constraints`, the size of `F'` without `F`.
611        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
612        cs.set_mode(SynthesisMode::Setup);
613        AugmentedFCircuit::<C1, C2, DummyCircuit>::empty(
614            poseidon_config,
615            dummy_circuit.clone(),
616            1,
617            d,
618            k,
619        )
620        .generate_constraints(cs.clone())?;
621        let augmentation_constraints = cs.num_constraints();
622
623        // The sum of `step_constraints` and `augmentation_constraints` is the
624        // size of `F'` with `t = 1`, and hence the actual `t` should have lower
625        // bound `log2(step_constraints + augmentation_constraints)`.
626        let t_lower_bound = log2(step_constraints + augmentation_constraints) as usize;
627        // Optimization: we in fact only need to try two values of `t`.
628        // This is because increasing `t` will only slightly affect the size of
629        // `F'` (more specifically, the size of `F'` will never be doubled).
630        // Thus, `t_lower_bound` (the log2 size of `F'` with `t = 1`) is very
631        // close to the actual `t` (either `t` or `t - 1`).
632        let t_upper_bound = t_lower_bound + 1;
633
634        for t in t_lower_bound..=t_upper_bound {
635            let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
636            cs.set_mode(SynthesisMode::Setup);
637            AugmentedFCircuit::<C1, C2, DummyCircuit>::empty(
638                poseidon_config,
639                dummy_circuit.clone(),
640                t,
641                d,
642                k,
643            )
644            .generate_constraints(cs.clone())?;
645            let augmentation_constraints = cs.num_constraints();
646            if t == log2(step_constraints + augmentation_constraints) as usize {
647                return Ok(t);
648            }
649        }
650        unreachable!()
651    }
652}

655where
656    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
657    C2: Curve,
658    FC: FCircuit<C1::ScalarField>,
659    CS1: CommitmentScheme<C1>,
660    CS2: CommitmentScheme<C2>,

662    type PreprocessorParam = (PoseidonConfig<CF1<C1>>, FC);
663    type ProverParam = ProverParams<C1, C2, CS1, CS2>;
664    type VerifierParam = VerifierParams<C1, C2, CS1, CS2>;
665    type RunningInstance = (CommittedInstance<C1, true>, Witness<C1::ScalarField>);
666    type IncomingInstance = (CommittedInstance<C1, false>, Witness<C1::ScalarField>);
667    type MultiCommittedInstanceWithWitness =
668        (CommittedInstance<C1, false>, Witness<C1::ScalarField>);
669    type CFInstance = (CycleFoldCommittedInstance<C2>, CycleFoldWitness<C2>);
670    type IVCProof = IVCProof<C1, C2>;

672    fn pp_deserialize_with_mode<R: std::io::prelude::Read>(
673        reader: R,
674        compress: ark_serialize::Compress,
675        validate: ark_serialize::Validate,
676        _fc_params: FC::Params, // FCircuit params
677    ) -> Result<Self::ProverParam, Error> {
678        Ok(Self::ProverParam::deserialize_with_mode(
679            reader, compress, validate,
680        )?)

682
683    fn vp_deserialize_with_mode<R: std::io::prelude::Read>(
684        mut reader: R,
685        compress: ark_serialize::Compress,
686        validate: ark_serialize::Validate,
687        fc_params: FC::Params,
688    ) -> Result<Self::VerifierParam, Error> {
689        let poseidon_config = poseidon_canonical_config::<C1::ScalarField>();
690
691        // generate the r1cs & cf_r1cs needed for the VerifierParams. In this way we avoid needing
692        // to serialize them, saving significant space in the VerifierParams serialized size.
693
694        let f_circuit = FC::new(fc_params)?;
695        let k = 1;
696        let d = R1CS::<CF1<C1>>::empty().degree();
697        let t = Self::compute_t(&poseidon_config, &f_circuit, d, k)?;
698
699        // main circuit R1CS:
700        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
701        cs.set_mode(SynthesisMode::Setup);
702        let augmented_F_circuit =
703            AugmentedFCircuit::<C1, C2, FC>::empty(&poseidon_config, f_circuit.clone(), t, d, k);
704        augmented_F_circuit.generate_constraints(cs.clone())?;
705        cs.finalize();
706        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
707        let r1cs = extract_r1cs::<C1::ScalarField>(&cs)?;
708
709        // CycleFold circuit R1CS
710        let cs2 = ConstraintSystem::<C1::BaseField>::new_ref();
711        cs2.set_mode(SynthesisMode::Setup);
712        let cf_circuit = CycleFoldCircuit::<_, ProtoGalaxyCycleFoldConfig<C1>>::default();
713        cf_circuit.generate_constraints(cs2.clone())?;
714        cs2.finalize();
715        let cs2 = cs2.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
716        let cf_r1cs = extract_r1cs::<C1::BaseField>(&cs2)?;
717
718        let cs_vp = CS1::VerifierParams::deserialize_with_mode(&mut reader, compress, validate)?;
719        let cf_cs_vp = CS2::VerifierParams::deserialize_with_mode(&mut reader, compress, validate)?;
720
721        Ok(Self::VerifierParam {
722            poseidon_config,
723            r1cs,
724            cf_r1cs,
725            cs_vp,
726            cf_cs_vp,
727        })
728    }

730    fn preprocess(
731        mut rng: impl RngCore,
732        (poseidon_config, F): &Self::PreprocessorParam,
733    ) -> Result<(Self::ProverParam, Self::VerifierParam), Error> {
734        // We fix `k`, the number of incoming instances, to 1, because
735        // multi-instances folding is not supported yet.
736        // TODO: Support multi-instances folding and make `k` a constant generic parameter (as in
737        // HyperNova). Tracking issue:
738        // https://github.com/privacy-scaling-explorations/sonobe/issues/82
739        let k = 1;
740        let d = R1CS::<CF1<C1>>::empty().degree();
741        let t = Self::compute_t(poseidon_config, F, d, k)?;
742
743        // prepare the circuit to obtain its R1CS
744        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
745        cs.set_mode(SynthesisMode::Setup);
746        let cs2 = ConstraintSystem::<C1::BaseField>::new_ref();
747        cs2.set_mode(SynthesisMode::Setup);
748
749        let augmented_F_circuit =
750            AugmentedFCircuit::<C1, C2, FC>::empty(poseidon_config, F.clone(), t, d, k);
751        let cf_circuit = CycleFoldCircuit::<_, ProtoGalaxyCycleFoldConfig<C1>>::default();
752
753        augmented_F_circuit.generate_constraints(cs.clone())?;
754        cs.finalize();
755        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
756        let r1cs = extract_r1cs::<C1::ScalarField>(&cs)?;
757
758        cf_circuit.generate_constraints(cs2.clone())?;
759        cs2.finalize();
760        let cs2 = cs2.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
761        let cf_r1cs = extract_r1cs::<C1::BaseField>(&cs2)?;
762
763        // `CS1` is for committing to ProtoGalaxy's witness vector `w`, so we
764        // set `len` to the number of witnesses in `r1cs`.
765        let (cs_pp, cs_vp) = CS1::setup(&mut rng, r1cs.n_witnesses())?;
766        // `CS2` is for committing to CycleFold's witness vector `w` and error
767        // term `e`, where the length of `e` is the number of constraints, so we
768        // set `len` to the maximum of `e` and `w`'s lengths.
769        let (cf_cs_pp, cf_cs_vp) = CS2::setup(
770            &mut rng,
771            max(cf_r1cs.n_constraints(), cf_r1cs.n_witnesses()),
772        )?;
773
774        Ok((
775            Self::ProverParam {
776                poseidon_config: poseidon_config.clone(),
777                cs_params: cs_pp,
778                cf_cs_params: cf_cs_pp,
779            },
780            Self::VerifierParam {
781                poseidon_config: poseidon_config.clone(),
782                r1cs,
783                cf_r1cs,
784                cs_vp,
785                cf_cs_vp,
786            },
787        ))
788    }

790    /// Initializes the ProtoGalaxy+CycleFold's IVC for the given parameters and
791    /// initial state `z_0`.
792    fn init(
793        (pp, vp): &(Self::ProverParam, Self::VerifierParam),
794        F: FC,
795        z_0: Vec<C1::ScalarField>,
796    ) -> Result<Self, Error> {
797        // compute the public params hash
798        let pp_hash = vp.pp_hash()?;

803        let (cf_W_dummy, cf_U_dummy) = vp.cf_r1cs.dummy_witness_instance();
804
805        // W_dummy=W_0 is a 'dummy witness', all zeroes, but with the size corresponding to the
806        // R1CS that we're working with.
807        Ok(Self {
808            r1cs: vp.r1cs.clone(),
809            cf_r1cs: vp.cf_r1cs.clone(),
810            poseidon_config: pp.poseidon_config.clone(),
811            cs_params: pp.cs_params.clone(),
812            cf_cs_params: pp.cf_cs_params.clone(),
813            F,
814            pp_hash,
815            i: C1::ScalarField::zero(),
816            z_0: z_0.clone(),
817            z_i: z_0,
818            w_i: w_dummy,
819            u_i: u_dummy,
820            W_i: W_dummy,
821            U_i: U_dummy,
822            // cyclefold running instance
823            cf_W_i: cf_W_dummy,
824            cf_U_i: cf_U_dummy,
825        })
826    }

828    /// Implements IVC.P of ProtoGalaxy+CycleFold
829    fn prove_step(
830        &mut self,
831        mut rng: impl RngCore,
832        external_inputs: FC::ExternalInputs,
833        _other_instances: Option<Self::MultiCommittedInstanceWithWitness>,
834    ) -> Result<(), Error> {
835        // Multi-instances folding is not supported yet.
836        if _other_instances.is_some() {
837            return Err(Error::NoMultiInstances);
838        }
839        // We fix `k`, the number of incoming instances, to 1, because
840        // multi-instances folding is not supported yet.
841        // TODO: Support multi-instances folding and make `k` a constant generic parameter (as in
842        // HyperNova). Tracking issue:
843        // https://github.com/privacy-scaling-explorations/sonobe/issues/82
844        let k = 1;
845        let d = self.r1cs.degree();
846
847        // `sponge` is for digest computation.
848        let sponge = PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(
849            &self.poseidon_config,
850            self.pp_hash,
851        );
852        // `transcript` is for challenge generation.
853        let mut transcript_prover = sponge.clone();
854
855        let mut augmented_F_circuit: AugmentedFCircuit<C1, C2, FC>;
856
857        if self.z_i.len() != self.F.state_len() {
858            return Err(Error::NotSameLength(
859                "z_i.len()".to_string(),
860                self.z_i.len(),
861                "F.state_len()".to_string(),
862                self.F.state_len(),
863            ));
864        }
865
866        let i_bn: BigUint = self.i.into();
867        let i_usize: usize = i_bn.try_into().map_err(|_| Error::MaxStep)?;
868
869        if self.i.is_zero() {
870            augmented_F_circuit = AugmentedFCircuit::empty(
871                &self.poseidon_config,
872                self.F.clone(),
873                self.U_i.betas.len(),
874                d,
875                k,
876            );
877            augmented_F_circuit.pp_hash = self.pp_hash;
878            augmented_F_circuit.z_0.clone_from(&self.z_0);
879            augmented_F_circuit.z_i.clone_from(&self.z_i);
880            augmented_F_circuit
881                .external_inputs
882                .clone_from(&external_inputs);
883
884        // There is no need to update `self.U_i` etc. as they are unchanged.
885        } else {
886            // Primary part:
887            // Compute `U_{i+1}` by folding `u_i` into `U_i`.
888            let (U_i1, W_i1, proof, aux) = Folding::prove(
889                &mut transcript_prover,
890                &self.r1cs,
891                &self.U_i,
892                &self.W_i,
893                &[self.u_i.clone()],
894                &[self.w_i.clone()],
895            )?;
896
897            // CycleFold part:
898            // Create cyclefold circuit for enforcing:
899            // U_i.phi * L_evals[0] + u_i.phi * L_evals[1] = U_i1.phi
900            let (cf_w_i, cf_u_i) = ProtoGalaxyCycleFoldConfig {
901                rs: aux.L_X_evals,
902                points: vec![self.U_i.phi, self.u_i.phi],
903            }
904            .build_circuit()
905            .generate_incoming_instance_witness::<_, CS2, false>(&self.cf_cs_params, &mut rng)?;
906
907            // fold cf_U_i + cf_u_i -> folded running instance cf_U_i1
908            let (cf_W_i1, cf_U_i1, cf_cmTs) =
909                CycleFoldAugmentationGadget::fold_native::<_, CS2, false>(
910                    &mut transcript_prover,
911                    &self.cf_r1cs,
912                    &self.cf_cs_params,
913                    self.cf_W_i.clone(),
914                    self.cf_U_i.clone(),
915                    vec![cf_w_i],
916                    vec![cf_u_i.clone()],
917                )?;
918
919            augmented_F_circuit = AugmentedFCircuit {
920                poseidon_config: self.poseidon_config.clone(),
921                pp_hash: self.pp_hash,
922                i: self.i,
923                i_usize,
924                z_0: self.z_0.clone(),
925                z_i: self.z_i.clone(),
926                external_inputs: external_inputs.clone(),
927                u_i_phi: self.u_i.phi,
928                U_i: self.U_i.clone(),
929                U_i1_phi: U_i1.phi,
930                F_coeffs: proof.F_coeffs.clone(),
931                K_coeffs: proof.K_coeffs.clone(),
932                F: self.F.clone(),
933                // cyclefold values
934                cf_u_i_cmW: cf_u_i.cmW,
935                cf_U_i: self.cf_U_i.clone(),
936                cf_cmT: cf_cmTs[0],
937            };
938
939            #[cfg(test)]
940            {
941                let mut transcript_verifier = sponge.clone();
942                assert_eq!(
943                    Folding::verify(
944                        &mut transcript_verifier,
945                        &self.U_i,
946                        &[self.u_i.clone()],
947                        proof
948                    )?,
949                    U_i1
950                );
951            }
952
953            self.W_i = W_i1;
954            self.U_i = U_i1;
955            self.cf_W_i = cf_W_i1;
956            self.cf_U_i = cf_U_i1;
957        }
958
959        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
960
961        let z_i1 = augmented_F_circuit
962            .compute_next_state(cs.clone())?
963            .value()?;
964
965        #[cfg(test)]
966        assert!(cs.is_satisfied()?);
967
968        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
969        let (w_i1, x_i1) = extract_w_x::<C1::ScalarField>(&cs);
970
971        #[cfg(test)]
972        if x_i1.len() != 2 {
973            return Err(Error::NotExpectedLength(x_i1.len(), 2));
974        }
975
976        // set values for next iteration
977        self.i += C1::ScalarField::one();
978        self.z_i = z_i1;
979        self.w_i = Witness::new(w_i1);
980        self.u_i = self.w_i.commit::<CS1, C1>(&self.cs_params, x_i1)?;
981
982        #[cfg(test)]
983        {
984            self.u_i.check_incoming()?;
985            self.r1cs.check_relation(&self.w_i, &self.u_i)?;
986            self.r1cs.check_relation(&self.W_i, &self.U_i)?;
987        }
988
989        Ok(())
990    }

1050    /// Implements IVC.V of ProtoGalaxy+CycleFold
1051    fn verify(vp: Self::VerifierParam, ivc_proof: Self::IVCProof) -> Result<(), Error> {
1052        let Self::IVCProof {
1053            i: num_steps,
1054            z_0,
1055            z_i,
1056            W_i,
1057            U_i,
1058            w_i,
1059            u_i,
1060            cf_W_i,
1061            cf_U_i,
1062        } = ivc_proof;
1063
1064        let sponge = PoseidonSponge::new_with_pp_hash(&vp.poseidon_config, vp.pp_hash()?);
1065
1066        if u_i.x.len() != 2 || U_i.x.len() != 2 {
1067            return Err(Error::IVCVerificationFail);
1068        }
1069
1070        // check that u_i's output points to the running instance
1071        // u_i.X[0] == H(i, z_0, z_i, U_i)
1072        let expected_u_i_x = U_i.hash(&sponge, num_steps, &z_0, &z_i);
1073        if expected_u_i_x != u_i.x[0] {
1074            return Err(Error::IVCVerificationFail);
1075        }
1076        // u_i.X[1] == H(cf_U_i)
1077        let expected_cf_u_i_x = cf_U_i.hash_cyclefold(&sponge);
1078        if expected_cf_u_i_x != u_i.x[1] {
1079            return Err(Error::IVCVerificationFail);
1080        }
1081
1082        // check R1CS satisfiability, which is equivalent to checking if `u_i`
1083        // is an incoming instance and if `w_i` and `u_i` satisfy RelaxedR1CS
1084        u_i.check_incoming()?;
1085        vp.r1cs.check_relation(&w_i, &u_i)?;
1086        // check RelaxedR1CS satisfiability
1087        vp.r1cs.check_relation(&W_i, &U_i)?;
1088
1089        // check CycleFold RelaxedR1CS satisfiability
1090        vp.cf_r1cs.check_relation(&cf_W_i, &cf_U_i)?;
1091
1092        Ok(())
1093    }

1159        let poseidon_config = poseidon_canonical_config::<Fr>();
1160        for state_len in [1, 10, 100] {
1161            let dummy_circuit: DummyCircuit = FCircuit::<Fr>::new(state_len)?;
1162
1163            let costs: Vec<usize> = (1..32)
1164                .into_par_iter()
1165                .map(|t| {
1166                    let cs = ConstraintSystem::<Fr>::new_ref();
1167                    AugmentedFCircuit::<Projective, Projective2, DummyCircuit>::empty(
1168                        &poseidon_config,
1169                        dummy_circuit.clone(),
1170                        t,
1171                        d,
1172                        k,
1173                    )
1174                    .generate_constraints(cs.clone())?;
1175                    Ok(cs.num_constraints())
1176                })
1177                .collect::<Result<Vec<usize>, Error>>()?;
1178
1179            for t_lower_bound in log2(costs[0]) as usize..32 {
1180                let num_constraints = (1 << t_lower_bound) - costs[0] + costs[t_lower_bound - 1];
1181                let t = log2(num_constraints) as usize;
1182                assert!(t == t_lower_bound || t == t_lower_bound + 1);
1183            }
1184        }
1185        Ok(())
1186    }
