63/// Configuration for Nova's CycleFold circuit
64pub struct NovaCycleFoldConfig<C: Curve> {
65    r: Vec<bool>,
66    points: Vec<C>,
67}
68
69impl<C: Curve> Default for NovaCycleFoldConfig<C> {
70    fn default() -> Self {
71        Self {
72            r: vec![false; NOVA_N_BITS_RO],
73            points: vec![C::zero(); 2],
74        }
75    }
76}
77
78impl<C: Curve> CycleFoldConfig<C> for NovaCycleFoldConfig<C> {
79    const RANDOMNESS_BIT_LENGTH: usize = NOVA_N_BITS_RO;
80    // Number of points to be folded in the CycleFold circuit, in Nova's case, this is a fixed
81    // amount:
82    // 2 points to be folded.
83    const N_INPUT_POINTS: usize = 2;
84    const N_UNIQUE_RANDOMNESSES: usize = 1;
85
86    fn alloc_points(&self, cs: ConstraintSystemRef<CF2<C>>) -> Result<Vec<C::Var>, SynthesisError> {
87        let points = Vec::new_witness(cs.clone(), || Ok(self.points.clone()))?;
88        for point in &points {
89            Self::mark_point_as_public(point)?;
90        }
91        Ok(points)
92    }
93
94    fn alloc_randomnesses(
95        &self,
96        cs: ConstraintSystemRef<CF2<C>>,
97    ) -> Result<Vec<Vec<Boolean<CF2<C>>>>, SynthesisError> {
98        let one = &CF1::<C>::one().into_bigint().to_bits_le()[..NOVA_N_BITS_RO];
99        let one_var = Vec::new_constant(cs.clone(), one)?;
100        let r_var = Vec::new_witness(cs.clone(), || Ok(self.r.clone()))?;
101        Self::mark_randomness_as_public(&r_var)?;
102        Ok(vec![one_var, r_var])
103    }
104}

170#[derive(Debug, Clone, Eq, PartialEq, CanonicalSerialize, CanonicalDeserialize)]
171pub struct Witness<C: Curve> {
172    pub E: Vec<C::ScalarField>,
173    pub rE: C::ScalarField,
174    pub W: Vec<C::ScalarField>,
175    pub rW: C::ScalarField,
176}
177
178impl<C: Curve> Witness<C> {
179    pub fn new<const H: bool>(w: Vec<C::ScalarField>, e_len: usize, mut rng: impl RngCore) -> Self {
180        let (rW, rE) = if H {
181            (
182                C::ScalarField::rand(&mut rng),
183                C::ScalarField::rand(&mut rng),
184            )
185        } else {
186            (C::ScalarField::zero(), C::ScalarField::zero())
187        };
188
189        Self {
190            E: vec![C::ScalarField::zero(); e_len],
191            rE,
192            W: w,
193            rW,
194        }
195    }
196
197    pub fn commit<CS: CommitmentScheme<C, HC>, const HC: bool>(
198        &self,
199        params: &CS::ProverParams,
200        x: Vec<C::ScalarField>,
201    ) -> Result<CommittedInstance<C>, Error> {
202        let mut cmE = C::zero();
203        if !is_zero_vec::<C::ScalarField>(&self.E) {
204            cmE = CS::commit(params, &self.E, &self.rE)?;
205        }
206        let cmW = CS::commit(params, &self.W, &self.rW)?;
207        Ok(CommittedInstance {
208            cmE,
209            u: C::ScalarField::one(),
210            cmW,
211            x,
212        })
213    }
214}

490impl<C1, C2, FC, CS1, CS2, const H: bool> FoldingScheme<C1, C2, FC>
491    for Nova<C1, C2, FC, CS1, CS2, H>
492where
493    C1: Curve,
494    C2: Curve,
495    FC: FCircuit<C1::ScalarField>,
496    CS1: CommitmentScheme<C1, H>,
497    CS2: CommitmentScheme<C2, H>,
498    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
499{
500    type PreprocessorParam = PreprocessorParam<C1, C2, FC, CS1, CS2, H>;
501    type ProverParam = ProverParams<C1, C2, CS1, CS2, H>;
502    type VerifierParam = VerifierParams<C1, C2, CS1, CS2, H>;
503    type RunningInstance = (CommittedInstance<C1>, Witness<C1>);
504    type IncomingInstance = (CommittedInstance<C1>, Witness<C1>);
505    type MultiCommittedInstanceWithWitness = ();
506    type CFInstance = (CycleFoldCommittedInstance<C2>, CycleFoldWitness<C2>);
507    type IVCProof = IVCProof<C1, C2>;
508
509    fn pp_deserialize_with_mode<R: std::io::prelude::Read>(
510        reader: R,
511        compress: ark_serialize::Compress,
512        validate: ark_serialize::Validate,
513        _fc_params: FC::Params, // FCircuit params
514    ) -> Result<Self::ProverParam, Error> {
515        Ok(Self::ProverParam::deserialize_with_mode(
516            reader, compress, validate,
517        )?)
518    }
519    fn vp_deserialize_with_mode<R: std::io::prelude::Read>(
520        mut reader: R,
521        compress: ark_serialize::Compress,
522        validate: ark_serialize::Validate,
523        fc_params: FC::Params,
524    ) -> Result<Self::VerifierParam, Error> {
525        let poseidon_config = poseidon_canonical_config::<C1::ScalarField>();
526
527        // generate the r1cs & cf_r1cs needed for the VerifierParams. In this way we avoid needing
528        // to serialize them, saving significant space in the VerifierParams serialized size.
529
530        // main circuit R1CS:
531        let f_circuit = FC::new(fc_params)?;
532        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
533        cs.set_mode(SynthesisMode::Setup);
534        let augmented_F_circuit =
535            AugmentedFCircuit::<C1, C2, FC>::empty(&poseidon_config, f_circuit.clone());
536        augmented_F_circuit.generate_constraints(cs.clone())?;
537        cs.finalize();
538        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
539        let r1cs = extract_r1cs::<C1::ScalarField>(&cs)?;
540
541        // CycleFold circuit R1CS
542        let cs2 = ConstraintSystem::<C1::BaseField>::new_ref();
543        cs2.set_mode(SynthesisMode::Setup);
544        let cf_circuit = CycleFoldCircuit::<_, NovaCycleFoldConfig<C1>>::default();
545        cf_circuit.generate_constraints(cs2.clone())?;
546        cs2.finalize();
547        let cs2 = cs2.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
548        let cf_r1cs = extract_r1cs::<C1::BaseField>(&cs2)?;
549
550        let cs_vp = CS1::VerifierParams::deserialize_with_mode(&mut reader, compress, validate)?;
551        let cf_cs_vp = CS2::VerifierParams::deserialize_with_mode(&mut reader, compress, validate)?;
552
553        Ok(Self::VerifierParam {
554            poseidon_config,
555            r1cs,
556            cf_r1cs,
557            cs_vp,
558            cf_cs_vp,
559        })
560    }
561
562    fn preprocess(
563        mut rng: impl RngCore,
564        prep_param: &Self::PreprocessorParam,
565    ) -> Result<(Self::ProverParam, Self::VerifierParam), Error> {
566        let (r1cs, cf_r1cs) =
567            get_r1cs::<C1, C2, FC>(&prep_param.poseidon_config, prep_param.F.clone())?;
568
569        // if cs params exist, use them, if not, generate new ones
570        let (cs_pp, cs_vp) = match (&prep_param.cs_pp, &prep_param.cs_vp) {
571            (Some(cs_pp), Some(cs_vp)) => (cs_pp.clone(), cs_vp.clone()),
572            _ => CS1::setup(
573                &mut rng,
574                // `CS1` is for committing to Nova's witness vector `w` and
575                // error term `e`, where the length of `e` is the number of
576                // constraints, so we set `len` to the maximum of `e` and `w`'s
577                // lengths.
578                max(r1cs.n_constraints(), r1cs.n_witnesses()),
579            )?,
580        };
581        let (cf_cs_pp, cf_cs_vp) = match (&prep_param.cf_cs_pp, &prep_param.cf_cs_vp) {
582            (Some(cf_cs_pp), Some(cf_cs_vp)) => (cf_cs_pp.clone(), cf_cs_vp.clone()),
583            _ => CS2::setup(
584                &mut rng,
585                // `CS2` is for committing to CycleFold's witness vector `w` and
586                // error term `e`, where the length of `e` is the number of
587                // constraints, so we set `len` to the maximum of `e` and `w`'s
588                // lengths.
589                max(cf_r1cs.n_constraints(), cf_r1cs.n_witnesses()),
590            )?,
591        };
592
593        let prover_params = ProverParams::<C1, C2, CS1, CS2, H> {
594            poseidon_config: prep_param.poseidon_config.clone(),
595            cs_pp: cs_pp.clone(),
596            cf_cs_pp: cf_cs_pp.clone(),
597        };
598        let verifier_params = VerifierParams::<C1, C2, CS1, CS2, H> {
599            poseidon_config: prep_param.poseidon_config.clone(),
600            r1cs,
601            cf_r1cs,
602            cs_vp,
603            cf_cs_vp,
604        };
605
606        Ok((prover_params, verifier_params))
607    }

609    /// Initializes the Nova+CycleFold's IVC for the given parameters and initial state `z_0`.
610    fn init(
611        params: &(Self::ProverParam, Self::VerifierParam),
612        F: FC,
613        z_0: Vec<C1::ScalarField>,
614    ) -> Result<Self, Error> {
615        let (pp, vp) = params;
616
617        // prepare the circuit to obtain its R1CS
618        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
619        cs.set_mode(SynthesisMode::Setup);
620        let cs2 = ConstraintSystem::<C1::BaseField>::new_ref();
621        cs2.set_mode(SynthesisMode::Setup);
622
623        let augmented_F_circuit =
624            AugmentedFCircuit::<C1, C2, FC>::empty(&pp.poseidon_config, F.clone());
625        let cf_circuit = CycleFoldCircuit::<_, NovaCycleFoldConfig<C1>>::default();
626
627        augmented_F_circuit.generate_constraints(cs.clone())?;
628        cs.finalize();
629        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
630        let r1cs = extract_r1cs::<C1::ScalarField>(&cs)?;
631
632        cf_circuit.generate_constraints(cs2.clone())?;
633        cs2.finalize();
634        let cs2 = cs2.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
635        let cf_r1cs = extract_r1cs::<C1::BaseField>(&cs2)?;
636
637        // compute the public params hash
638        let pp_hash = vp.pp_hash()?;
639
640        // setup the dummy instances
641        let (W_dummy, U_dummy) = r1cs.dummy_witness_instance();
642        let (w_dummy, u_dummy) = r1cs.dummy_witness_instance();
643        let (cf_W_dummy, cf_U_dummy) = cf_r1cs.dummy_witness_instance();
644
645        // W_dummy=W_0 is a 'dummy witness', all zeroes, but with the size corresponding to the
646        // R1CS that we're working with.
647        Ok(Self {
648            r1cs,
649            cf_r1cs,
650            poseidon_config: pp.poseidon_config.clone(),
651            cs_pp: pp.cs_pp.clone(),
652            cf_cs_pp: pp.cf_cs_pp.clone(),
653            F,
654            pp_hash,
655            i: C1::ScalarField::zero(),
656            z_0: z_0.clone(),
657            z_i: z_0,
658            w_i: w_dummy,
659            u_i: u_dummy,
660            W_i: W_dummy,
661            U_i: U_dummy,
662            // cyclefold running instance
663            cf_W_i: cf_W_dummy,
664            cf_U_i: cf_U_dummy,
665        })
666    }

668    /// Implements IVC.P of Nova+CycleFold
669    fn prove_step(
670        &mut self,
671        mut rng: impl RngCore,
672        external_inputs: FC::ExternalInputs,
673        // Nova does not support multi-instances folding (by design)
674        _other_instances: Option<Self::MultiCommittedInstanceWithWitness>,
675    ) -> Result<(), Error> {
676        // ensure that commitments are blinding if user has specified so.
677        if H && self.i >= C1::ScalarField::one() {
678            let blinding_commitments = if self.i == C1::ScalarField::one() {
679                // blinding values of the running instances are zero at the first iteration
680                vec![self.w_i.rW, self.w_i.rE]
681            } else {
682                vec![self.w_i.rW, self.w_i.rE, self.W_i.rW, self.W_i.rE]
683            };
684            if blinding_commitments.contains(&C1::ScalarField::zero()) {
685                return Err(Error::IncorrectBlinding(
686                    H,
687                    format!("{blinding_commitments:?}"),
688                ));
689            }
690        }
691        // `sponge` is for digest computation.
692        let sponge = PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(
693            &self.poseidon_config,
694            self.pp_hash,
695        );
696        // `transcript` is for challenge generation.
697        let mut transcript = sponge.clone();
698
699        let augmented_F_circuit: AugmentedFCircuit<C1, C2, FC>;
700
701        // Nova does not support (by design) multi-instances folding
702        if _other_instances.is_some() {
703            return Err(Error::NoMultiInstances);
704        }
705
706        if self.z_i.len() != self.F.state_len() {
707            return Err(Error::NotSameLength(
708                "z_i.len()".to_string(),
709                self.z_i.len(),
710                "F.state_len()".to_string(),
711                self.F.state_len(),
712            ));
713        }
714
715        if self.i > C1::ScalarField::from_le_bytes_mod_order(&usize::MAX.to_le_bytes()) {
716            return Err(Error::MaxStep);
717        }
718
719        let i_usize;
720
721        #[cfg(target_pointer_width = "64")]
722        {
723            let mut i_bytes: [u8; 8] = [0; 8];
724            i_bytes.copy_from_slice(&self.i.into_bigint().to_bytes_le()[..8]);
725            i_usize = usize::from_le_bytes(i_bytes);
726        }
727
728        #[cfg(target_pointer_width = "32")]
729        {
730            let mut i_bytes: [u8; 4] = [0; 4];
731            i_bytes.copy_from_slice(&self.i.into_bigint().to_bytes_le()[..4]);
732            i_usize = usize::from_le_bytes(i_bytes);
733        }
734
735        // fold Nova instances
736        let (W_i1, U_i1, cmT, r_bits): (Witness<C1>, CommittedInstance<C1>, C1, Vec<bool>) =
737            NIFS::<C1, CS1, PoseidonSponge<C1::ScalarField>, H>::prove(
738                &self.cs_pp,
739                &self.r1cs,
740                &mut transcript,
741                &self.W_i,
742                &self.U_i,
743                &self.w_i,
744                &self.u_i,
745            )?;
746
747        if self.i == C1::ScalarField::zero() {
748            // base case
749            augmented_F_circuit = AugmentedFCircuit::<C1, C2, FC> {
750                poseidon_config: self.poseidon_config.clone(),
751                pp_hash: Some(self.pp_hash),
752                i: Some(C1::ScalarField::zero()), // = i=0
753                i_usize: Some(0),
754                z_0: Some(self.z_0.clone()), // = z_i
755                z_i: Some(self.z_i.clone()),
756                external_inputs: Some(external_inputs.clone()),
757                u_i_cmW: Some(self.u_i.cmW), // = dummy
758                U_i: Some(self.U_i.clone()), // = dummy
759                U_i1_cmE: Some(U_i1.cmE),
760                U_i1_cmW: Some(U_i1.cmW),
761                cmT: Some(cmT),
762                F: self.F.clone(),
763                cf1_u_i_cmW: None,
764                cf2_u_i_cmW: None,
765                cf_U_i: None,
766                cf1_cmT: None,
767                cf2_cmT: None,
768            };
769
770            #[cfg(test)]
771            {
772                let r_Fr = C1::ScalarField::from_bigint(BigInteger::from_bits_le(&r_bits))
773                    .ok_or(Error::OutOfBounds)?;
774                let expected =
775                    NIFS::<C1, CS1, PoseidonSponge<C1::ScalarField>, H>::fold_committed_instances(
776                        r_Fr, &self.U_i, &self.u_i, &cmT,
777                    );
778                assert_eq!(U_i1, expected);
779            }
780        } else {
781            // CycleFold part:
782            let (cfW_w_i, cfW_u_i) = NovaCycleFoldConfig {
783                r: r_bits.clone(),
784                points: vec![self.U_i.clone().cmW, self.u_i.clone().cmW],
785            }
786            .build_circuit()
787            .generate_incoming_instance_witness::<_, CS2, H>(&self.cf_cs_pp, &mut rng)?;
788            let (cfE_w_i, cfE_u_i) = NovaCycleFoldConfig {
789                r: r_bits.clone(),
790                points: vec![self.U_i.clone().cmE, cmT],
791            }
792            .build_circuit()
793            .generate_incoming_instance_witness::<_, CS2, H>(&self.cf_cs_pp, &mut rng)?;
794
795            let (cf_W_i1, cf_U_i1, cf_cmTs) = CycleFoldAugmentationGadget::fold_native::<_, CS2, H>(
796                &mut transcript,
797                &self.cf_r1cs,
798                &self.cf_cs_pp,
799                self.cf_W_i.clone(),
800                self.cf_U_i.clone(),
801                vec![cfW_w_i, cfE_w_i],
802                vec![cfW_u_i.clone(), cfE_u_i.clone()],
803            )?;
804
805            augmented_F_circuit = AugmentedFCircuit::<C1, C2, FC> {
806                poseidon_config: self.poseidon_config.clone(),
807                pp_hash: Some(self.pp_hash),
808                i: Some(self.i),
809                i_usize: Some(i_usize),
810                z_0: Some(self.z_0.clone()),
811                z_i: Some(self.z_i.clone()),
812                external_inputs: Some(external_inputs.clone()),
813                u_i_cmW: Some(self.u_i.cmW),
814                U_i: Some(self.U_i.clone()),
815                U_i1_cmE: Some(U_i1.cmE),
816                U_i1_cmW: Some(U_i1.cmW),
817                cmT: Some(cmT),
818                F: self.F.clone(),
819                // cyclefold values
820                cf1_u_i_cmW: Some(cfW_u_i.cmW),
821                cf2_u_i_cmW: Some(cfE_u_i.cmW),
822                cf_U_i: Some(self.cf_U_i.clone()),
823                cf1_cmT: Some(cf_cmTs[0]),
824                cf2_cmT: Some(cf_cmTs[1]),
825            };
826
827            self.cf_W_i = cf_W_i1;
828            self.cf_U_i = cf_U_i1;
829        }
830
831        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
832
833        let z_i1 = augmented_F_circuit
834            .compute_next_state(cs.clone())?
835            .value()?;
836
837        #[cfg(test)]
838        assert!(cs.is_satisfied()?);
839
840        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
841        let (w_i1, x_i1) = extract_w_x::<C1::ScalarField>(&cs);
842
843        #[cfg(test)]
844        if x_i1.len() != 2 {
845            return Err(Error::NotExpectedLength(x_i1.len(), 2));
846        }
847
848        // set values for next iteration
849        self.i += C1::ScalarField::one();
850        self.z_i = z_i1;
851        self.w_i = Witness::<C1>::new::<H>(w_i1, self.r1cs.n_constraints(), &mut rng);
852        self.u_i = self.w_i.commit::<CS1, H>(&self.cs_pp, x_i1)?;
853        self.W_i = W_i1;
854        self.U_i = U_i1;
855
856        #[cfg(test)]
857        {
858            self.u_i.check_incoming()?;
859            self.r1cs.check_relation(&self.w_i, &self.u_i)?;
860            self.r1cs.check_relation(&self.W_i, &self.U_i)?;
861        }
862
863        Ok(())
864    }

941    /// Implements IVC.V of Nov.clone()a+CycleFold. Notice that this method does not include the
942    /// commitments verification, which is done in the Decider.
943    fn verify(vp: Self::VerifierParam, ivc_proof: Self::IVCProof) -> Result<(), Error> {
944        let Self::IVCProof {
945            i: num_steps,
946            z_0,
947            z_i,
948            W_i,
949            U_i,
950            w_i,
951            u_i,
952            cf_W_i,
953            cf_U_i,
954        } = ivc_proof;
955
956        let sponge =
957            PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(&vp.poseidon_config, vp.pp_hash()?);
958
959        if num_steps == C1::ScalarField::zero() {
960            if z_0 != z_i {
961                return Err(Error::IVCVerificationFail);
962            }
963            return Ok(());
964        }
965
966        if u_i.x.len() != 2 || U_i.x.len() != 2 {
967            return Err(Error::IVCVerificationFail);
968        }
969
970        // check that u_i's output points to the running instance
971        // u_i.X[0] == H(i, z_0, z_i, U_i)
972        let expected_u_i_x = U_i.hash(&sponge, num_steps, &z_0, &z_i);
973        if expected_u_i_x != u_i.x[0] {
974            return Err(Error::IVCVerificationFail);
975        }
976        // u_i.X[1] == H(cf_U_i)
977        let expected_cf_u_i_x = cf_U_i.hash_cyclefold(&sponge);
978        if expected_cf_u_i_x != u_i.x[1] {
979            return Err(Error::IVCVerificationFail);
980        }
981
982        // check R1CS satisfiability, which is equivalent to checking if `u_i`
983        // is an incoming instance and if `w_i` and `u_i` satisfy RelaxedR1CS
984        u_i.check_incoming()?;
985        vp.r1cs.check_relation(&w_i, &u_i)?;
986        // check RelaxedR1CS satisfiability
987        vp.r1cs.check_relation(&W_i, &U_i)?;
988
989        // check CycleFold RelaxedR1CS satisfiability
990        vp.cf_r1cs.check_relation(&cf_W_i, &cf_U_i)?;
991
992        Ok(())
993    }
994}

996/// helper method to get the r1cs from the ConstraintSynthesizer
997pub fn get_r1cs_from_cs<F: PrimeField>(
998    circuit: impl ConstraintSynthesizer<F>,
999) -> Result<R1CS<F>, Error> {
1000    let cs = ConstraintSystem::<F>::new_ref();
1001    cs.set_mode(SynthesisMode::Setup);
1002    circuit.generate_constraints(cs.clone())?;
1003    cs.finalize();
1004    let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
1005    let r1cs = extract_r1cs::<F>(&cs)?;
1006    Ok(r1cs)
1007}

1009/// helper method to get the R1CS for both the AugmentedFCircuit and the CycleFold circuit
1010#[allow(clippy::type_complexity)]
1011pub fn get_r1cs<C1, C2, FC>(
1012    poseidon_config: &PoseidonConfig<C1::ScalarField>,
1013    F_circuit: FC,
1014) -> Result<(R1CS<C1::ScalarField>, R1CS<C2::ScalarField>), Error>
1015where
1016    C1: Curve,
1017    C2: Curve,
1018    FC: FCircuit<C1::ScalarField>,
1019    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
1020{
1021    let augmented_F_circuit = AugmentedFCircuit::<C1, C2, FC>::empty(poseidon_config, F_circuit);
1022    let cf_circuit = CycleFoldCircuit::<_, NovaCycleFoldConfig<C1>>::default();
1023    let r1cs = get_r1cs_from_cs::<C1::ScalarField>(augmented_F_circuit)?;
1024    let cf_r1cs = get_r1cs_from_cs::<C2::ScalarField>(cf_circuit)?;
1025    Ok((r1cs, cf_r1cs))
1026}
