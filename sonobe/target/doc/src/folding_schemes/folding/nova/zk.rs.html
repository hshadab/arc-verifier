1//! Implements Nova's zero-knowledge layer, as described in https://eprint.iacr.org/2023/573.pdf.
2//!
3//! Remark: this zk layer implementation only covers a subset of the use cases:
4//!
5//! We identify 3 interesting places to use the nova zk-layer: one before all the folding pipeline
6//! (Use-case-1), one at the end of the folding pipeline right before the final Decider SNARK
7//! proof (Use-case-2), and a third one for cases where compressed SNARK proofs are not needed, and
8//! just IVC proofs (bigger than SNARK proofs) suffice (Use-case-3):
9//!
10//! * Use-case-1: at the beginning of the folding pipeline, right when the user has their original
11//!   instance prior to be folded into the running instance, the user can fold it with the
12//!   random-satisfying-instance to then have a blinded instance that can be sent to a server that
13//!   will fold it with the running instance.
14//!     
15//!     --> In this one, the user could externalize all the IVC folding and also the Decider final
16//!   proof generation to a server.
17//!
18//! * Use-case-2: at the end of all the IVC folding steps (after n iterations of nova.prove_step),
19//!   to 'blind' the IVC proof so then it can be sent to a server that will generate the final
20//!   decider SNARK proof.
21//!     
22//!     --> In this one, the user could offload the Decider final proof generation to a server.
23//!   
24//! * Use-case-3: the user does not care about the Decider (final compressed SNARK proof), and
25//!   wants to generate a zk-proof of the IVC state to an IVC verifier (without any SNARK proof
26//!   involved). In this use-case, the zk is only added at the last IVCProof. Note that this proof
27//!   will be much bigger and expensive to verify than a Decider SNARK proof.
28//!
29//! The current implementation covers the Use-case-3.
30//! Use-case-1 can be achieved directly by a simpler version of the zk IVC scheme skipping steps
31//! and implemented directly at the app level by folding the original instance with a randomized
32//! instance (steps 2,3,4 from section D.4 of the [HyperNova](https://eprint.iacr.org/2023/573.pdf)
33//! paper).
34//! And the Use-case-2 would require a modified version of the Decider circuits.

51pub struct RandomizedIVCProof<C1: Curve, C2: Curve> {
52    pub U_i: CommittedInstance<C1>,
53    pub u_i: CommittedInstance<C1>,
54    pub U_r: CommittedInstance<C1>,
55    pub pi: C1,       // proof = cmT
56    pub pi_prime: C1, // proof' = cmT'
57    pub W_i_prime: Witness<C1>,
58    pub cf_U_i: CommittedInstance<C2>,
59    pub cf_W_i: Witness<C2>,
60}
61
62impl<C1: Curve, C2: Curve> RandomizedIVCProof<C1, C2> {
63    /// Compute a zero-knowledge proof of a Nova IVC proof
64    /// It implements the prover of appendix D.4.in https://eprint.iacr.org/2023/573.pdf
65    /// For further details on why folding is hiding, see lemma 9
66    pub fn new<
67        FC: FCircuit<C1::ScalarField>,
68        CS1: CommitmentScheme<C1, true>,
69        CS2: CommitmentScheme<C2, true>,
70    >(
71        nova: &Nova<C1, C2, FC, CS1, CS2, true>,
72        mut rng: impl RngCore,
73    ) -> Result<RandomizedIVCProof<C1, C2>, Error> {
74        let mut transcript = PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(
75            &nova.poseidon_config,
76            nova.pp_hash,
77        );
78
79        // I. Compute proof for 'regular' instances
80        // 1. Fold the instance-witness pairs (U_i, W_i) with (u_i, w_i)
81        let (W_f, U_f, cmT, _) = NIFS::<C1, CS1, PoseidonSponge<C1::ScalarField>, true>::prove(
82            &nova.cs_pp,
83            &nova.r1cs,
84            &mut transcript,
85            &nova.w_i,
86            &nova.u_i,
87            &nova.W_i,
88            &nova.U_i,
89        )?;
90
91        // 2. Sample a satisfying relaxed R1CS instance-witness pair (W_r, U_r)
92        let (W_r, U_r) = nova
93            .r1cs
94            .sample_witness_instance::<CS1>(&nova.cs_pp, &mut rng)?;
95
96        // 3. Fold the instance-witness pair (U_f, W_f) with (U_r, W_r)
97        let (W_i_prime, _, cmT_i_prime, _) =
98            NIFS::<C1, CS1, PoseidonSponge<C1::ScalarField>, true>::prove(
99                &nova.cs_pp,
100                &nova.r1cs,
101                &mut transcript,
102                &W_f,
103                &U_f,
104                &W_r,
105                &U_r,
106            )?;
107
108        Ok(RandomizedIVCProof {
109            U_i: nova.U_i.clone(),
110            u_i: nova.u_i.clone(),
111            U_r,
112            pi: cmT,
113            pi_prime: cmT_i_prime,
114            W_i_prime,
115            cf_U_i: nova.cf_U_i.clone(),
116            cf_W_i: nova.cf_W_i.clone(),
117        })
118    }
119
120    /// Verify a zero-knowledge proof of a Nova IVC proof
121    /// It implements the verifier of appendix D.4. in https://eprint.iacr.org/2023/573.pdf
122    #[allow(clippy::too_many_arguments)]
123    pub fn verify<CS1: CommitmentScheme<C1, true>, CS2: CommitmentScheme<C2, true>>(
124        r1cs: &R1CS<C1::ScalarField>,
125        cf_r1cs: &R1CS<C2::ScalarField>,
126        pp_hash: C1::ScalarField,
127        poseidon_config: &PoseidonConfig<C1::ScalarField>,
128        i: C1::ScalarField,
129        z_0: Vec<C1::ScalarField>,
130        z_i: Vec<C1::ScalarField>,
131        proof: &RandomizedIVCProof<C1, C2>,
132    ) -> Result<(), Error>
133    where
134        C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
135    {
136        // Handles case where i=0
137        if i == C1::ScalarField::zero() {
138            if z_0 == z_i {
139                return Ok(());
140            } else {
141                return Err(Error::zkIVCVerificationFail);
142            }
143        }
144
145        // 1. Check that u_i.x is correct - including the cyclefold running instance
146        // a. Check length
147        if proof.u_i.x.len() != 2 {
148            return Err(Error::IVCVerificationFail);
149        }
150
151        // b. Check computed hashes are correct
152        let sponge = PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(poseidon_config, pp_hash);
153        let mut transcript = sponge.clone();
154        let expected_u_i_x = proof.U_i.hash(&sponge, i, &z_0, &z_i);
155        if expected_u_i_x != proof.u_i.x[0] {
156            return Err(Error::zkIVCVerificationFail);
157        }
158
159        let expected_cf_u_i_x = proof.cf_U_i.hash_cyclefold(&sponge);
160        if expected_cf_u_i_x != proof.u_i.x[1] {
161            return Err(Error::IVCVerificationFail);
162        }
163
164        // 2. Check that u_i values are correct
165        if !proof.u_i.cmE.is_zero() || proof.u_i.u != C1::ScalarField::one() {
166            return Err(Error::zkIVCVerificationFail);
167        }
168
169        // 3. Obtain the U_f folded instance
170        let (U_f, _) = NIFS::<C1, CS1, PoseidonSponge<C1::ScalarField>, true>::verify(
171            &mut transcript,
172            &proof.u_i,
173            &proof.U_i,
174            &proof.pi,
175        )?;
176
177        // 4. Obtain the U^{\prime}_i folded instance
178        let (U_i_prime, _) = NIFS::<C1, CS1, PoseidonSponge<C1::ScalarField>, true>::verify(
179            &mut transcript,
180            &U_f,
181            &proof.U_r,
182            &proof.pi_prime,
183        )?;
184
185        // 5. Check that W^{\prime}_i is a satisfying witness
186        r1cs.check_relation(&proof.W_i_prime, &U_i_prime)?;
187
188        // 6. Check that the cyclefold instance-witness pair satisfies the cyclefold relaxed r1cs
189        cf_r1cs.check_relation(&proof.cf_W_i, &proof.cf_U_i)?;
190
191        Ok(())
192    }
193}

206    // Tests zk proof generation and verification for a valid nova IVC proof
207    #[test]
208    fn test_zk_nova_ivc() -> Result<(), Error> {
209        let mut rng = OsRng;
210        let poseidon_config = poseidon_canonical_config::<Fr>();
211        let F_circuit = CubicFCircuit::<Fr>::new(())?;
212        let (_, nova) = test_ivc_opt::<
213            Pedersen<Projective, true>,
214            Pedersen<Projective2, true>,
215            true,
216        >(poseidon_config.clone(), F_circuit, 3)?;
217
218        let proof = RandomizedIVCProof::new(&nova, &mut rng)?;
219        let verify =
220            RandomizedIVCProof::verify::<Pedersen<Projective, true>, Pedersen<Projective2, true>>(
221                &nova.r1cs,
222                &nova.cf_r1cs,
223                nova.pp_hash,
224                &nova.poseidon_config,
225                nova.i,
226                nova.z_0,
227                nova.z_i,
228                &proof,
229            );
230        assert!(verify.is_ok());
231        Ok(())
232    }
