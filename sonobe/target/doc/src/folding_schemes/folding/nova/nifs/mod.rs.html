29/// Defines the NIFS (Non-Interactive Folding Scheme) trait, initially defined in
30/// [Nova](https://eprint.iacr.org/2021/370.pdf), and it's variants
31/// [Ova](https://hackmd.io/V4838nnlRKal9ZiTHiGYzw) and
32/// [Mova](https://eprint.iacr.org/2024/1220.pdf).
33/// `H` specifies whether the NIFS will use a blinding factor.
34pub trait NIFSTrait<
35    C: Curve,
36    CS: CommitmentScheme<C, H>,
37    T: Transcript<C::ScalarField>,
38    const H: bool = false,
39>
40{
41    type CommittedInstance: Debug + Clone + Absorb; // + CommittedInstanceOps<C>;
42    type Witness: Debug + Clone;
43    type ProverAux: Debug + Clone; // Prover's aux params. eg. in Nova is T
44    type Proof: Debug + Clone; // proof. eg. in Nova is cmT
45
46    fn new_witness(w: Vec<C::ScalarField>, e_len: usize, rng: impl RngCore) -> Self::Witness;
47
48    fn new_instance(
49        rng: impl RngCore,
50        params: &CS::ProverParams,
51        w: &Self::Witness,
52        x: Vec<C::ScalarField>,
53        aux: Vec<C::ScalarField>, // t_or_e in Ova, empty for Nova
54    ) -> Result<Self::CommittedInstance, Error>;
55
56    fn fold_witness(
57        r: C::ScalarField,
58        W: &Self::Witness, // running witness
59        w: &Self::Witness, // incoming witness
60        aux: &Self::ProverAux,
61    ) -> Result<Self::Witness, Error>;
62
63    /// NIFS.P. Returns a tuple containing the folded Witness, the folded CommittedInstance, and
64    /// the used challenge `r` as a vector of bits, so that it can be reused in other methods.
65    #[allow(clippy::type_complexity)]
66    #[allow(clippy::too_many_arguments)]
67    fn prove(
68        cs_prover_params: &CS::ProverParams,
69        r1cs: &R1CS<C::ScalarField>,
70        transcript: &mut T,
71        W_i: &Self::Witness,           // running witness
72        U_i: &Self::CommittedInstance, // running committed instance
73        w_i: &Self::Witness,           // incoming witness
74        u_i: &Self::CommittedInstance, // incoming committed instance
75    ) -> Result<
76        (
77            Self::Witness,
78            Self::CommittedInstance,
79            Self::Proof,
80            Vec<bool>,
81        ),
82        Error,
83    >;
84
85    /// NIFS.V. Returns the folded CommittedInstance and the used challenge `r` as a vector of
86    /// bits, so that it can be reused in other methods.
87    fn verify(
88        transcript: &mut T,
89        U_i: &Self::CommittedInstance,
90        u_i: &Self::CommittedInstance,
91        proof: &Self::Proof,
92    ) -> Result<(Self::CommittedInstance, Vec<bool>), Error>;
93}

95/// Defines the NIFS (Non-Interactive Folding Scheme) Gadget trait, which specifies the in-circuit
96/// logic of the NIFS.Verify defined in [Nova](https://eprint.iacr.org/2021/370.pdf) and it's
97/// variants [Ova](https://hackmd.io/V4838nnlRKal9ZiTHiGYzw) and
98/// [Mova](https://eprint.iacr.org/2024/1220.pdf).
99pub trait NIFSGadgetTrait<C: Curve, S: CryptographicSponge, T: TranscriptVar<CF1<C>, S>> {
100    type CommittedInstance: Debug + Clone + Absorb + CommittedInstanceOps<C>;
101    type CommittedInstanceVar: Debug
102        + Clone
103        + AbsorbGadget<C::ScalarField>
104        + AllocVar<Self::CommittedInstance, CF1<C>>
105        + CommittedInstanceVarOps<C>;
106    type Proof: Debug + Clone;
107    type ProofVar: Debug + Clone + AllocVar<Self::Proof, CF1<C>>;
108
109    /// Implements the constraints for NIFS.V for u and x, since cm(E) and cm(W) are delegated to
110    /// the CycleFold circuit.
111    #[allow(clippy::type_complexity)]
112    fn verify(
113        transcript: &mut T,
114        U_i: Self::CommittedInstanceVar,
115        // U_i_vec is passed to reuse the already computed U_i_vec from previous methods
116        U_i_vec: Vec<FpVar<CF1<C>>>,
117        u_i: Self::CommittedInstanceVar,
118        proof: Option<Self::ProofVar>,
119    ) -> Result<(Self::CommittedInstanceVar, Vec<Boolean<CF1<C>>>), SynthesisError>;
120}

147    /// Test method used to test the different implementations of the NIFSTrait (ie. Nova, Mova,
148    /// Ova). Runs a loop using the NIFS trait, and returns the last Witness and CommittedInstance
149    /// so that their relation can be checked.
150    pub(crate) fn test_nifs_opt<
151        N: NIFSTrait<Projective, Pedersen<Projective>, PoseidonSponge<Fr>>,
152    >() -> Result<(N::Witness, N::CommittedInstance), Error> {
153        let r1cs: R1CS<Fr> = get_test_r1cs();
154
155        let mut rng = ark_std::test_rng();
156        let (pedersen_params, _) =
157            Pedersen::<Projective>::setup(&mut rng, max(r1cs.n_constraints(), r1cs.n_witnesses()))?;
158
159        let poseidon_config = poseidon_canonical_config::<Fr>();
160        let pp_hash = Fr::rand(&mut rng);
161        let mut transcript_p = PoseidonSponge::<Fr>::new_with_pp_hash(&poseidon_config, pp_hash);
162        let mut transcript_v = transcript_p.clone();
163
164        // prepare the running instance
165        let z = get_test_z(3);
166        let (w, x) = r1cs.split_z(&z);
167        let mut W_i = N::new_witness(w.clone(), r1cs.n_constraints(), test_rng());
168        let mut U_i = N::new_instance(&mut rng, &pedersen_params, &W_i, x, vec![])?;
169
170        let num_iters = 10;
171        for i in 0..num_iters {
172            // prepare the incoming instance
173            let incoming_instance_z = get_test_z(i + 4);
174            let (w, x) = r1cs.split_z(&incoming_instance_z);
175            let w_i = N::new_witness(w.clone(), r1cs.n_constraints(), test_rng());
176            let u_i = N::new_instance(&mut rng, &pedersen_params, &w_i, x, vec![])?;
177
178            // NIFS.P
179            let (folded_witness, _, proof, _) = N::prove(
180                &pedersen_params,
181                &r1cs,
182                &mut transcript_p,
183                &W_i,
184                &U_i,
185                &w_i,
186                &u_i,
187            )?;
188
189            // NIFS.V
190            let (folded_committed_instance, _) = N::verify(&mut transcript_v, &U_i, &u_i, &proof)?;
191
192            // set running_instance for next loop iteration
193            W_i = folded_witness;
194            U_i = folded_committed_instance;
195        }
196
197        Ok((W_i, U_i))
198    }

200    /// Test method used to test the different implementations of the NIFSGadgetTrait (ie. Nova,
201    /// Mova, Ova). It returns the last Witness and CommittedInstance so that it can be checked at
202    /// the parent test that their values match.
203    pub(crate) fn test_nifs_gadget_opt<N, NG>(
204        ci: Vec<NG::CommittedInstance>,
205        proof: NG::Proof,
206    ) -> Result<(NG::CommittedInstance, NG::CommittedInstanceVar), Error>
207    where
208        N: NIFSTrait<Projective, Pedersen<Projective>, PoseidonSponge<Fr>>,
209        NG: NIFSGadgetTrait<
210            Projective,
211            PoseidonSponge<Fr>,
212            PoseidonSpongeVar<Fr>,
213            CommittedInstance = N::CommittedInstance, // constrain that N::CI==NG::CI
214            Proof = N::Proof,                         // constrain that N::Proof==NG::Proof
215        >,
216    {
217        let mut rng = ark_std::test_rng();
218
219        let (U_i, u_i) = (ci[0].clone(), ci[1].clone());
220        let pp_hash = Fr::rand(&mut rng);
221        let poseidon_config = poseidon_canonical_config::<Fr>();
222        let mut transcript = PoseidonSponge::<Fr>::new_with_pp_hash(&poseidon_config, pp_hash);
223        let (ci3, _) = N::verify(&mut transcript, &U_i, &u_i, &proof)?;
224
225        let cs = ConstraintSystem::<Fr>::new_ref();
226
227        let pp_hashVar = FpVar::<Fr>::new_witness(cs.clone(), || Ok(pp_hash))?;
228        let mut transcriptVar =
229            PoseidonSpongeVar::<Fr>::new_with_pp_hash(&poseidon_config, &pp_hashVar)?;
230        let ci1Var = NG::CommittedInstanceVar::new_witness(cs.clone(), || Ok(U_i.clone()))?;
231        let ci2Var = NG::CommittedInstanceVar::new_witness(cs.clone(), || Ok(u_i.clone()))?;
232        let proofVar = NG::ProofVar::new_witness(cs.clone(), || Ok(proof))?;
233
234        let ci1Var_vec = ci1Var.to_sponge_field_elements()?;
235        let (out, _) = NG::verify(
236            &mut transcriptVar,
237            ci1Var.clone(),
238            ci1Var_vec,
239            ci2Var.clone(),
240            Some(proofVar.clone()),
241        )?;
242        assert!(cs.is_satisfied()?);
243
244        // return the NIFS.V and the NIFSGadget.V obtained values, so that they are checked at the
245        // parent test
246        Ok((ci3, out))
247    }

249    /// test that checks the native CommittedInstance.to_sponge_{bytes,field_elements}
250    /// vs the R1CS constraints version
251    pub(crate) fn test_committed_instance_to_sponge_preimage_opt<N, NG>(
252        ci: N::CommittedInstance,
253    ) -> Result<(), Error>
254    where
255        N: NIFSTrait<Projective, Pedersen<Projective>, PoseidonSponge<Fr>>,
256        NG: NIFSGadgetTrait<
257            Projective,
258            PoseidonSponge<Fr>,
259            PoseidonSpongeVar<Fr>,
260            CommittedInstance = N::CommittedInstance, // constrain that N::CI==NG::CI
261        >,
262    {
263        let bytes = ci.to_sponge_bytes_as_vec();
264        let field_elements = ci.to_sponge_field_elements_as_vec();
265
266        let cs = ConstraintSystem::<Fr>::new_ref();
267
268        let ciVar = NG::CommittedInstanceVar::new_witness(cs.clone(), || Ok(ci.clone()))?;
269        let bytes_var = ciVar.to_sponge_bytes()?;
270        let field_elements_var = ciVar.to_sponge_field_elements()?;
271
272        assert!(cs.is_satisfied()?);
273
274        // check that the natively computed and in-circuit computed hashes match
275        assert_eq!(bytes_var.value()?, bytes);
276        assert_eq!(field_elements_var.value()?, field_elements);
277        Ok(())
278    }

280    pub(crate) fn test_committed_instance_hash_opt<N, NG>(
281        ci: NG::CommittedInstance,
282    ) -> Result<(), Error>
283    where
284        N: NIFSTrait<Projective, Pedersen<Projective>, PoseidonSponge<Fr>>,
285        NG: NIFSGadgetTrait<
286            Projective,
287            PoseidonSponge<Fr>,
288            PoseidonSpongeVar<Fr>,
289            CommittedInstance = N::CommittedInstance, // constrain that N::CI==NG::CI
290        >,
291        N::CommittedInstance: CommittedInstanceOps<Projective>,
292    {
293        let poseidon_config = poseidon_canonical_config::<Fr>();
294        let pp_hash = Fr::from(42u32); // only for test
295        let sponge = PoseidonSponge::<Fr>::new_with_pp_hash(&poseidon_config, pp_hash);
296
297        let i = Fr::from(3_u32);
298        let z_0 = vec![Fr::from(3_u32)];
299        let z_i = vec![Fr::from(3_u32)];
300
301        // compute the CommittedInstance hash natively
302        let h = ci.hash(&sponge, i, &z_0, &z_i);
303
304        let cs = ConstraintSystem::<Fr>::new_ref();
305
306        let pp_hashVar = FpVar::<Fr>::new_witness(cs.clone(), || Ok(pp_hash))?;
307        let iVar = FpVar::<Fr>::new_witness(cs.clone(), || Ok(i))?;
308        let z_0Var = Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(z_0.clone()))?;
309        let z_iVar = Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(z_i.clone()))?;
310        let ciVar = NG::CommittedInstanceVar::new_witness(cs.clone(), || Ok(ci.clone()))?;
311
312        let sponge = PoseidonSpongeVar::<Fr>::new_with_pp_hash(&poseidon_config, &pp_hashVar)?;
313
314        // compute the CommittedInstance hash in-circuit
315        let (hVar, _) = ciVar.hash(&sponge, &iVar, &z_0Var, &z_iVar)?;
316        assert!(cs.is_satisfied()?);
317
318        // check that the natively computed and in-circuit computed hashes match
319        assert_eq!(hVar.value()?, h);
320        Ok(())
321    }
