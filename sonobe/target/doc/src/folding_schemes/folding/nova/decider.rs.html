27pub struct Proof<C1, C2, CS1, CS2, S1, S2>
28where
29    C1: Curve,
30    C2: Curve,
31    CS1: CommitmentScheme<C1>,
32    CS2: CommitmentScheme<C2>,
33    S1: SNARK<C1::ScalarField>,
34    S2: SNARK<C2::ScalarField>,
35{
36    c1_snark_proof: S1::Proof,
37    c2_snark_proof: S2::Proof,
38    cs1_proofs: [CS1::Proof; 2],
39    cs2_proofs: [CS2::Proof; 2],
40    // cmT and r are values for the last fold, U_{i+1}=NIFS.V(r, U_i, u_i, cmT), and they are
41    // checked in-circuit
42    cmT: C1,
43    r: C1::ScalarField,
44    // cyclefold committed instance
45    cf_U_final: CycleFoldCommittedInstance<C2>,
46    // the CS challenges are provided by the prover, but in-circuit they are checked to match the
47    // in-circuit computed computed ones.
48    cs1_challenges: [C1::ScalarField; 2],
49    cs2_challenges: [C2::ScalarField; 2],
50}

83#[derive(Clone, Debug)]
84pub struct Decider<C1, C2, FC, CS1, CS2, S1, S2, FS> {
85    _c1: PhantomData<C1>,
86    _c2: PhantomData<C2>,
87    _fc: PhantomData<FC>,
88    _cs1: PhantomData<CS1>,
89    _cs2: PhantomData<CS2>,
90    _s1: PhantomData<S1>,
91    _s2: PhantomData<S2>,
92    _fs: PhantomData<FS>,

95impl<C1, C2, FC, CS1, CS2, S1, S2, FS> DeciderTrait<C1, C2, FC, FS>
96    for Decider<C1, C2, FC, CS1, CS2, S1, S2, FS>
97where
98    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
99    C2: Curve,
100    FC: FCircuit<C1::ScalarField>,
101    CS1: CommitmentScheme<
102        C1,
103        ProverChallenge = C1::ScalarField,
104        Challenge = C1::ScalarField,
105        Proof = crate::commitment::kzg::Proof<C1>,
106    >,
107    CS2: CommitmentScheme<
108        C2,
109        ProverChallenge = C2::ScalarField,
110        Challenge = C2::ScalarField,
111        Proof = crate::commitment::kzg::Proof<C2>,
112    >,
113    S1: SNARK<C1::ScalarField>,
114    S2: SNARK<C2::ScalarField>,
115    FS: FoldingScheme<C1, C2, FC>,
116    // constrain FS into Nova, since this is a Decider specifically for Nova
117    Nova<C1, C2, FC, CS1, CS2, false>: From<FS>,
118    crate::folding::nova::ProverParams<C1, C2, CS1, CS2, false>:
119        From<<FS as FoldingScheme<C1, C2, FC>>::ProverParam>,
120    crate::folding::nova::VerifierParams<C1, C2, CS1, CS2, false>:
121        From<<FS as FoldingScheme<C1, C2, FC>>::VerifierParam>,
122{
123    type PreprocessorParam = ((FS::ProverParam, FS::VerifierParam), usize);
124    type ProverParam =
125        ProverParam<CS1::ProverParams, S1::ProvingKey, CS2::ProverParams, S2::ProvingKey>;
126    type Proof = Proof<C1, C2, CS1, CS2, S1, S2>;
127    type VerifierParam = VerifierParam<
128        C1,
129        CS1::VerifierParams,
130        S1::VerifyingKey,
131        CS2::VerifierParams,
132        S2::VerifyingKey,
133    >;
134    type PublicInput = Vec<C1::ScalarField>;
135    type CommittedInstance = Vec<C1>;
136
137    fn preprocess(
138        mut rng: impl RngCore + CryptoRng,
139        ((pp, vp), state_len): Self::PreprocessorParam,
140    ) -> Result<(Self::ProverParam, Self::VerifierParam), Error> {
141        // get the FoldingScheme prover & verifier params from Nova
142        let nova_pp: <Nova<C1, C2, FC, CS1, CS2, false> as FoldingScheme<C1, C2, FC>>::ProverParam =
143            pp.into();
144        let nova_vp: <Nova<C1, C2, FC, CS1, CS2, false> as FoldingScheme<
145            C1,
146            C2,
147            FC,
148        >>::VerifierParam = vp.into();
149        let pp_hash = nova_vp.pp_hash()?;
150
151        let poseidon_config1 = nova_vp.poseidon_config;
152        // Create a poseidon config on `C2`'s scalar field for `circuit2`, with
153        // the same parameters (`full_rounds` etc.) as `circuit1` to ensure the
154        // security level is the same.
155        // Note: `ark` and `mds` will be different because they depend on the
156        // field, but they will not affect the security level.
157        let poseidon_config2 = poseidon_custom_config(
158            poseidon_config1.full_rounds,
159            poseidon_config1.partial_rounds,
160            poseidon_config1.alpha,
161            poseidon_config1.rate,
162            poseidon_config1.capacity,
163        );
164
165        let circuit1 = DeciderCircuit1::<C1, C2>::dummy((
166            nova_vp.r1cs,
167            &nova_vp.cf_r1cs,
168            poseidon_config1,
169            (),
170            (),
171            state_len,
172            2, // Nova's running CommittedInstance contains 2 commitments
173        ));
174        let circuit2 = DeciderCircuit2::<C2>::dummy((
175            nova_vp.cf_r1cs,
176            poseidon_config2,
177            2, // Nova's running CommittedInstance contains 2 commitments
178        ));
179
180        // get the Groth16 specific setup for the circuits
181        let (c1_g16_pk, c1_g16_vk) = S1::circuit_specific_setup(circuit1, &mut rng)
182            .map_err(|e| Error::SNARKSetupFail(e.to_string()))?;
183        let (c2_g16_pk, c2_g16_vk) = S2::circuit_specific_setup(circuit2, &mut rng)
184            .map_err(|e| Error::SNARKSetupFail(e.to_string()))?;
185
186        let pp = Self::ProverParam {
187            c1_snark_pp: c1_g16_pk,
188            c1_cs_pp: nova_pp.cs_pp,
189            c2_snark_pp: c2_g16_pk,
190            c2_cs_pp: nova_pp.cf_cs_pp,
191        };
192        let vp = Self::VerifierParam {
193            pp_hash,
194            c1_snark_vp: c1_g16_vk,
195            c1_cs_vp: nova_vp.cs_vp,
196            c2_snark_vp: c2_g16_vk,
197            c2_cs_vp: nova_vp.cf_cs_vp,
198        };
199        Ok((pp, vp))
200    }
201
202    fn prove(
203        mut rng: impl RngCore + CryptoRng,
204        pp: Self::ProverParam,
205        fs: FS,
206    ) -> Result<Self::Proof, Error> {
207        let circuit1 = DeciderCircuit1::<C1, C2>::try_from(Nova::from(fs.clone()))?;
208        let circuit2 = DeciderCircuit2::<C2>::try_from(Nova::from(fs))?;
209
210        let cmT = circuit1.proof;
211        let r = circuit1.randomness;
212        let cf_U_final = circuit1.cf_U_i.clone();
213
214        let c1_kzg_challenges = circuit1.kzg_challenges.clone();
215        let c1_kzg_proofs = circuit1
216            .W_i1
217            .get_openings()
218            .iter()
219            .zip(&c1_kzg_challenges)
220            .map(|((v, _), &c)| {
221                CS1::prove_with_challenge(&pp.c1_cs_pp, c, v, &C1::ScalarField::zero(), None)
222            })
223            .collect::<Result<Vec<_>, _>>()?;
224        let c2_kzg_challenges = circuit2.kzg_challenges.clone();
225        let c2_kzg_proofs = circuit2
226            .cf_W_i
227            .get_openings()
228            .iter()
229            .zip(&c2_kzg_challenges)
230            .map(|((v, _), &c)| {
231                CS2::prove_with_challenge(&pp.c2_cs_pp, c, v, &C2::ScalarField::zero(), None)
232            })
233            .collect::<Result<Vec<_>, _>>()?;
234
235        let c1_snark_proof = S1::prove(&pp.c1_snark_pp, circuit1, &mut rng)
236            .map_err(|e| Error::Other(e.to_string()))?;
237        let c2_snark_proof = S2::prove(&pp.c2_snark_pp, circuit2, &mut rng)
238            .map_err(|e| Error::Other(e.to_string()))?;
239
240        Ok(Self::Proof {
241            c1_snark_proof,
242            c2_snark_proof,
243            cs1_proofs: c1_kzg_proofs
244                .try_into()
245                .map_err(|e: Vec<_>| Error::NotExpectedLength(e.len(), 2))?,
246            cs2_proofs: c2_kzg_proofs
247                .try_into()
248                .map_err(|e: Vec<_>| Error::NotExpectedLength(e.len(), 2))?,
249            cmT,
250            r,
251            cf_U_final,
252            cs1_challenges: c1_kzg_challenges
253                .try_into()
254                .map_err(|e: Vec<_>| Error::NotExpectedLength(e.len(), 2))?,
255            cs2_challenges: c2_kzg_challenges
256                .try_into()
257                .map_err(|e: Vec<_>| Error::NotExpectedLength(e.len(), 2))?,
258        })
259    }
260
261    fn verify(
262        vp: Self::VerifierParam,
263        i: C1::ScalarField,
264        z_0: Vec<C1::ScalarField>,
265        z_i: Vec<C1::ScalarField>,
266        // we don't use the instances at the verifier level, since we check them in-circuit
267        running_commitments: &Self::CommittedInstance,
268        incoming_commitments: &Self::CommittedInstance,
269        proof: &Self::Proof,
270    ) -> Result<bool, Error> {
271        if i <= C1::ScalarField::one() {
272            return Err(Error::NotEnoughSteps);
273        }
274
275        // 6.2. Fold the commitments
276        let U_final_commitments = DeciderNovaGadget::fold_group_elements_native(
277            running_commitments,
278            incoming_commitments,
279            Some(proof.cmT),
280            proof.r,
281        )?;
282        let cf_U = proof.cf_U_final.clone();
283
284        // snark proof 1
285        let c1_public_input = [
286            &[vp.pp_hash, i][..],
287            &z_0,
288            &z_i,
289            &U_final_commitments.inputize_nonnative(),
290            &cf_U.inputize_nonnative(),
291            &proof.cs1_challenges,
292            &proof.cs1_proofs.iter().map(|p| p.eval).collect::<Vec<_>>(),
293            &proof.cmT.inputize_nonnative(),
294        ]
295        .concat();
296
297        let c1_snark_v = S1::verify(&vp.c1_snark_vp, &c1_public_input, &proof.c1_snark_proof)
298            .map_err(|e| Error::Other(e.to_string()))?;
299        if !c1_snark_v {
300            return Err(Error::SNARKVerificationFail);
301        }
302
303        // snark proof 2
304        // migrate pp_hash from C1::Fr to C1::Fq
305        let pp_hash_Fq =
306            C2::ScalarField::from_le_bytes_mod_order(&vp.pp_hash.into_bigint().to_bytes_le());
307        let c2_public_input: Vec<C2::ScalarField> = [
308            &[pp_hash_Fq][..],
309            &cf_U.inputize(),
310            &proof.cs2_challenges,
311            &proof.cs2_proofs.iter().map(|p| p.eval).collect::<Vec<_>>(),
312        ]
313        .concat();
314
315        let c2_snark_v = S2::verify(&vp.c2_snark_vp, &c2_public_input, &proof.c2_snark_proof)
316            .map_err(|e| Error::Other(e.to_string()))?;
317        if !c2_snark_v {
318            return Err(Error::SNARKVerificationFail);
319        }
320
321        // 7.3. check C1 commitments (main instance commitments)
322        for ((cm, &c), pi) in U_final_commitments
323            .iter()
324            .zip(&proof.cs1_challenges)
325            .zip(&proof.cs1_proofs)
326        {
327            CS1::verify_with_challenge(&vp.c1_cs_vp, c, cm, pi)?;
328        }
329
330        // 4.3. check C2 commitments (CycleFold instance commitments)
331        for ((cm, &c), pi) in cf_U
332            .get_commitments()
333            .iter()
334            .zip(&proof.cs2_challenges)
335            .zip(&proof.cs2_proofs)
336        {
337            CS2::verify_with_challenge(&vp.c2_cs_vp, c, cm, pi)?;
338        }
339
340        Ok(true)
341    }

360    #[test]
361    fn test_decider() -> Result<(), Error> {
362        // use Nova as FoldingScheme
363        type N = Nova<
364            Projective,
365            Projective2,
366            CubicFCircuit<Fr>,
367            KZG<'static, MNT4>,
368            KZG<'static, MNT6>,
369            false,
370        >;
371        type D = Decider<
372            Projective,
373            Projective2,
374            CubicFCircuit<Fr>,
375            KZG<'static, MNT4>,
376            KZG<'static, MNT6>,
377            Groth16<MNT4>,
378            Groth16<MNT6>,
379            N, // here we define the FoldingScheme to use
380        >;
381
382        let mut rng = ark_std::test_rng();
383        let poseidon_config = poseidon_canonical_config::<Fr>();
384
385        let F_circuit = CubicFCircuit::<Fr>::new(())?;
386        let z_0 = vec![Fr::from(3_u32)];
387
388        let start = Instant::now();
389        let prep_param = PreprocessorParam::new(poseidon_config, F_circuit);
390        let nova_params = N::preprocess(&mut rng, &prep_param)?;
391        println!("Nova preprocess, {:?}", start.elapsed());
392
393        let start = Instant::now();
394        let mut nova = N::init(&nova_params, F_circuit, z_0.clone())?;
395        println!("Nova initialized, {:?}", start.elapsed());
396        let start = Instant::now();
397        nova.prove_step(&mut rng, (), None)?;
398        println!("prove_step, {:?}", start.elapsed());
399        nova.prove_step(&mut rng, (), None)?; // do a 2nd step
400
401        let mut rng = rand::rngs::OsRng;
402
403        // prepare the Decider prover & verifier params
404        let start = Instant::now();
405        let (decider_pp, decider_vp) =
406            D::preprocess(&mut rng, (nova_params, F_circuit.state_len()))?;
407        println!("Decider preprocess, {:?}", start.elapsed());
408
409        // decider proof generation
410        let start = Instant::now();
411        let proof = D::prove(rng, decider_pp, nova.clone())?;
412        println!("Decider prove, {:?}", start.elapsed());
413
414        // decider proof verification
415        let start = Instant::now();
416        let verified = D::verify(
417            decider_vp,
418            nova.i,
419            nova.z_0,
420            nova.z_i,
421            &nova.U_i.get_commitments(),
422            &nova.u_i.get_commitments(),
423            &proof,
424        )?;
425        assert!(verified);
426        println!("Decider verify, {:?}", start.elapsed());
427        Ok(())
428    }
