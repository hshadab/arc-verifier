34/// `AugmentedFCircuit` enhances the original step function `F`, so that it can
35/// be used in recursive arguments such as IVC.
36///
37/// The method for converting `F` to `AugmentedFCircuit` (`F'`) is defined in
38/// [Nova](https://eprint.iacr.org/2021/370.pdf), where `AugmentedFCircuit` not
39/// only invokes `F`, but also adds additional constraints for verifying the
40/// correct folding of primary instances (i.e., Nova's `CommittedInstance`s over
41/// `C1`).
42///
43/// Furthermore, to reduce circuit size over `C2`, we implement the constraints
44/// defined in [CycleFold](https://eprint.iacr.org/2023/1192.pdf). These extra
45/// constraints verify the correct folding of CycleFold instances.
46#[derive(Debug, Clone)]
47pub struct AugmentedFCircuit<C1: Curve, C2: Curve, FC: FCircuit<CF1<C1>>> {
48    pub(super) poseidon_config: PoseidonConfig<CF1<C1>>,
49    pub(super) pp_hash: Option<CF1<C1>>,
50    pub(super) i: Option<CF1<C1>>,
51    pub(super) i_usize: Option<usize>,
52    pub(super) z_0: Option<Vec<C1::ScalarField>>,
53    pub(super) z_i: Option<Vec<C1::ScalarField>>,
54    pub(super) external_inputs: Option<FC::ExternalInputs>,
55    pub(super) u_i_cmW: Option<C1>,
56    pub(super) U_i: Option<CommittedInstance<C1>>,
57    pub(super) U_i1_cmE: Option<C1>,
58    pub(super) U_i1_cmW: Option<C1>,
59    pub(super) cmT: Option<C1>,
60    pub(super) F: FC, // F circuit
61
62    // cyclefold verifier on C1
63    // Here 'cf1, cf2' are for each of the CycleFold circuits, corresponding to the fold of cmW and
64    // cmE respectively
65    pub(super) cf1_u_i_cmW: Option<C2>, // input
66    pub(super) cf2_u_i_cmW: Option<C2>, // input
67    pub(super) cf_U_i: Option<CycleFoldCommittedInstance<C2>>, // input
68    pub(super) cf1_cmT: Option<C2>,
69    pub(super) cf2_cmT: Option<C2>,
70}

98impl<C1, C2, FC> AugmentedFCircuit<C1, C2, FC>
99where
100    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
101    C2: Curve,
102    FC: FCircuit<CF1<C1>>,
103{
104    pub fn compute_next_state(
105        self,
106        cs: ConstraintSystemRef<CF1<C1>>,
107    ) -> Result<Vec<FpVar<CF1<C1>>>, SynthesisError> {
108        let pp_hash = FpVar::<CF1<C1>>::new_witness(cs.clone(), || {
109            Ok(self.pp_hash.unwrap_or_else(CF1::<C1>::zero))
110        })?;
111        let i = FpVar::<CF1<C1>>::new_witness(cs.clone(), || {
112            Ok(self.i.unwrap_or_else(CF1::<C1>::zero))
113        })?;
114        let z_0 = Vec::<FpVar<CF1<C1>>>::new_witness(cs.clone(), || {
115            Ok(self
116                .z_0
117                .unwrap_or(vec![CF1::<C1>::zero(); self.F.state_len()]))
118        })?;
119        let z_i = Vec::<FpVar<CF1<C1>>>::new_witness(cs.clone(), || {
120            Ok(self
121                .z_i
122                .unwrap_or(vec![CF1::<C1>::zero(); self.F.state_len()]))
123        })?;
124        let external_inputs = FC::ExternalInputsVar::new_witness(cs.clone(), || {
125            Ok(self.external_inputs.unwrap_or_default())
126        })?;
127
128        let u_dummy = CommittedInstance::dummy(2);
129        let U_i = CommittedInstanceVar::<C1>::new_witness(cs.clone(), || {
130            Ok(self.U_i.unwrap_or(u_dummy.clone()))
131        })?;
132        let U_i1_cmE = NonNativeAffineVar::new_witness(cs.clone(), || {
133            Ok(self.U_i1_cmE.unwrap_or_else(C1::zero))
134        })?;
135        let U_i1_cmW = NonNativeAffineVar::new_witness(cs.clone(), || {
136            Ok(self.U_i1_cmW.unwrap_or_else(C1::zero))
137        })?;
138
139        let cmT =
140            NonNativeAffineVar::new_witness(cs.clone(), || Ok(self.cmT.unwrap_or_else(C1::zero)))?;
141
142        let cf_u_dummy = CycleFoldCommittedInstance::dummy(NovaCycleFoldConfig::<C1>::IO_LEN);
143        let cf_U_i = CycleFoldCommittedInstanceVar::<C2>::new_witness(cs.clone(), || {
144            Ok(self.cf_U_i.unwrap_or(cf_u_dummy.clone()))
145        })?;
146        let cf1_cmT =
147            C2::Var::new_witness(cs.clone(), || Ok(self.cf1_cmT.unwrap_or_else(C2::zero)))?;
148        let cf2_cmT =
149            C2::Var::new_witness(cs.clone(), || Ok(self.cf2_cmT.unwrap_or_else(C2::zero)))?;
150
151        // `sponge` is for digest computation.
152        let sponge = PoseidonSpongeVar::<C1::ScalarField>::new_with_pp_hash(
153            &self.poseidon_config,
154            &pp_hash,
155        )?;
156        // `transcript` is for challenge generation.
157        let mut transcript = sponge.clone();
158
159        let is_basecase = i.is_zero()?;
160
161        // Primary Part
162        // P.1. Compute u_i.x
163        // u_i.x[0] = H(i, z_0, z_i, U_i)
164        let (u_i_x, U_i_vec) = U_i.clone().hash(&sponge, &i, &z_0, &z_i)?;
165        // u_i.x[1] = H(cf_U_i)
166        let (cf_u_i_x, _) = cf_U_i.clone().hash(&sponge)?;
167
168        // P.2. Construct u_i
169        let u_i = CommittedInstanceVar {
170            // u_i.cmE = cm(0)
171            cmE: NonNativeAffineVar::new_constant(cs.clone(), C1::zero())?,
172            // u_i.u = 1
173            u: FpVar::one(),
174            // u_i.cmW is provided by the prover as witness
175            cmW: NonNativeAffineVar::new_witness(cs.clone(), || {
176                Ok(self.u_i_cmW.unwrap_or(C1::zero()))
177            })?,
178            // u_i.x is computed in step 1
179            x: vec![u_i_x, cf_u_i_x],
180        };
181
182        // P.3. nifs.verify, obtains U_{i+1} by folding u_i & U_i.
183        // Notice that NIFSGadget::verify does not fold cmE & cmW.
184        // We set `U_i1.cmE` and `U_i1.cmW` to unconstrained witnesses `U_i1_cmE` and `U_i1_cmW`
185        // respectively.
186        // The correctness of them will be checked on the other curve.
187        let (mut U_i1, r_bits) = NIFSGadget::<
188            C1,
189            PoseidonSponge<C1::ScalarField>,
190            PoseidonSpongeVar<C1::ScalarField>,
191        >::verify(
192            &mut transcript,
193            U_i.clone(),
194            U_i_vec,
195            u_i.clone(),
196            Some(cmT.clone()),
197        )?;
198        U_i1.cmE = U_i1_cmE;
199        U_i1.cmW = U_i1_cmW;
200
201        // P.4.a compute and check the first output of F'
202
203        // get z_{i+1} from the F circuit
204        let i_usize = self.i_usize.unwrap_or(0);
205        let z_i1 = self
206            .F
207            .generate_step_constraints(cs.clone(), i_usize, z_i, external_inputs)?;
208
209        // Base case: u_{i+1}.x[0] == H((i+1, z_0, z_{i+1}, U_{\bot})
210        // Non-base case: u_{i+1}.x[0] == H((i+1, z_0, z_{i+1}, U_{i+1})
211        let (u_i1_x, _) =
212            U_i1.clone()
213                .hash(&sponge, &(i + FpVar::<CF1<C1>>::one()), &z_0, &z_i1)?;
214        let (u_i1_x_base, _) = CommittedInstanceVar::new_constant(cs.clone(), u_dummy)?.hash(
215            &sponge,
216            &FpVar::<CF1<C1>>::one(),
217            &z_0,
218            &z_i1,
219        )?;
220        let x = is_basecase.select(&u_i1_x_base, &u_i1_x)?;
221        // This line "converts" `x` from a witness to a public input.
222        // Instead of directly modifying the constraint system, we explicitly
223        // allocate a public input and enforce that its value is indeed `x`.
224        // While comparing `x` with itself seems redundant, this is necessary
225        // because:
226        // - `.value()` allows an honest prover to extract public inputs without
227        //   computing them outside the circuit.
228        // - `.enforce_equal()` prevents a malicious prover from claiming wrong
229        //   public inputs that are not the honest `x` computed in-circuit.
230        FpVar::new_input(cs.clone(), || x.value())?.enforce_equal(&x)?;
231
232        // CycleFold part
233        // C.1. Compute cf1_u_i.x and cf2_u_i.x
234        // C.2. Construct `cf1_u_i` and `cf2_u_i`
235        let cf1_u_i = CycleFoldCommittedInstanceVar::new_incoming_from_components(
236            // `cf1_u_i.cmW` is provided by the prover as witness.
237            C2::Var::new_witness(cs.clone(), || Ok(self.cf1_u_i_cmW.unwrap_or(C2::zero())))?,
238            // To construct `cf1_u_i.x`, we need to provide the randomness
239            // `r_bits` and the `cmW` component in committed instances `U_i`,
240            // `u_i`, and `U_{i+1}`.
241            &r_bits,
242            vec![U_i.cmW, u_i.cmW, U_i1.cmW],
243        )?;
244        let cf2_u_i = CycleFoldCommittedInstanceVar::new_incoming_from_components(
245            // `cf2_u_i.cmW` is provided by the prover as witness.
246            C2::Var::new_witness(cs.clone(), || Ok(self.cf2_u_i_cmW.unwrap_or(C2::zero())))?,
247            // To construct `cf2_u_i.x`, we need to provide the randomness
248            // `r_bits`, the `cmE` component in running instances `U_i` and
249            // `U_{i+1}`, and the cross term commitment `cmT`.
250            &r_bits,
251            vec![U_i.cmE, cmT, U_i1.cmE],
252        )?;
253
254        // C.3. nifs.verify, obtains cf_U_{i+1} by folding cf1_u_i and cf2_u_i into cf_U.
255        let cf_U_i1 = CycleFoldAugmentationGadget::fold_gadget(
256            &mut transcript,
257            cf_U_i,
258            vec![cf1_u_i, cf2_u_i],
259            vec![cf1_cmT, cf2_cmT],
260        )?;
261
262        // Back to Primary Part
263        // P.4.b compute and check the second output of F'
264        // Base case: u_{i+1}.x[1] == H(cf_U_{\bot})
265        // Non-base case: u_{i+1}.x[1] == H(cf_U_{i+1})
266        let (cf_u_i1_x, _) = cf_U_i1.clone().hash(&sponge)?;
267        let (cf_u_i1_x_base, _) =
268            CycleFoldCommittedInstanceVar::<C2>::new_constant(cs.clone(), cf_u_dummy)?
269                .hash(&sponge)?;
270        let cf_x = is_basecase.select(&cf_u_i1_x_base, &cf_u_i1_x)?;
271        // This line "converts" `cf_x` from a witness to a public input.
272        // Instead of directly modifying the constraint system, we explicitly
273        // allocate a public input and enforce that its value is indeed `cf_x`.
274        // While comparing `cf_x` with itself seems redundant, this is necessary
275        // because:
276        // - `.value()` allows an honest prover to extract public inputs without
277        //   computing them outside the circuit.
278        // - `.enforce_equal()` prevents a malicious prover from claiming wrong
279        //   public inputs that are not the honest `cf_x` computed in-circuit.
280        FpVar::new_input(cs.clone(), || cf_x.value())?.enforce_equal(&cf_x)?;
281
282        Ok(z_i1)
283    }
284}

286impl<C1, C2, FC> ConstraintSynthesizer<CF1<C1>> for AugmentedFCircuit<C1, C2, FC>
287where
288    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
289    C2: Curve,
290    FC: FCircuit<CF1<C1>>,
291{
292    fn generate_constraints(self, cs: ConstraintSystemRef<CF1<C1>>) -> Result<(), SynthesisError> {
293        self.compute_next_state(cs).map(|_| ())
294    }

312    // checks that the gadget and native implementations of the challenge computation match
313    #[test]
314    fn test_challenge_gadget() -> Result<(), Error> {
315        let mut rng = ark_std::test_rng();
316        let poseidon_config = poseidon_canonical_config::<Fr>();
317        let pp_hash = Fr::from(42u32); // only for testing
318        let mut transcript = PoseidonSponge::<Fr>::new_with_pp_hash(&poseidon_config, pp_hash);
319
320        let u_i = CommittedInstance::<Projective> {
321            cmE: Projective::rand(&mut rng),
322            u: Fr::rand(&mut rng),
323            cmW: Projective::rand(&mut rng),
324            x: vec![Fr::rand(&mut rng); 1],
325        };
326        let U_i = CommittedInstance::<Projective> {
327            cmE: Projective::rand(&mut rng),
328            u: Fr::rand(&mut rng),
329            cmW: Projective::rand(&mut rng),
330            x: vec![Fr::rand(&mut rng); 1],
331        };
332        let cmT = Projective::rand(&mut rng);
333
334        // compute the challenge natively
335        let r_bits =
336            ChallengeGadget::<Projective, CommittedInstance<Projective>>::get_challenge_native(
337                &mut transcript,
338                &U_i,
339                &u_i,
340                Some(&cmT),
341            );
342        let r = Fr::from_bigint(BigInteger::from_bits_le(&r_bits)).ok_or(Error::OutOfBounds)?;
343
344        let cs = ConstraintSystem::<Fr>::new_ref();
345        let pp_hashVar = FpVar::<Fr>::new_witness(cs.clone(), || Ok(pp_hash))?;
346        let u_iVar =
347            CommittedInstanceVar::<Projective>::new_witness(cs.clone(), || Ok(u_i.clone()))?;
348        let U_iVar =
349            CommittedInstanceVar::<Projective>::new_witness(cs.clone(), || Ok(U_i.clone()))?;
350        let cmTVar = NonNativeAffineVar::<Projective>::new_witness(cs.clone(), || Ok(cmT))?;
351        let mut transcriptVar =
352            PoseidonSpongeVar::<Fr>::new_with_pp_hash(&poseidon_config, &pp_hashVar)?;
353
354        // compute the challenge in-circuit
355        let r_bitsVar =
356            ChallengeGadget::<Projective, CommittedInstance<Projective>>::get_challenge_gadget(
357                &mut transcriptVar,
358                U_iVar.to_sponge_field_elements()?,
359                u_iVar,
360                Some(cmTVar),
361            )?;
362        assert!(cs.is_satisfied()?);
363
364        // check that the natively computed and in-circuit computed hashes match
365        let rVar = Boolean::le_bits_to_fp(&r_bitsVar)?;
366        assert_eq!(rVar.value()?, r);
367        assert_eq!(r_bitsVar.value()?, r_bits);
368        Ok(())
369    }
