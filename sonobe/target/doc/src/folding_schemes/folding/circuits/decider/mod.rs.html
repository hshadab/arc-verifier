22/// Gadget that computes the KZG challenges.
23/// It also offers the rust native implementation compatible with the gadget.
24pub struct KZGChallengesGadget {}
25
26impl KZGChallengesGadget {
27    pub fn get_challenges_native<C: Curve, T: Transcript<CF1<C>>, U: CommittedInstanceOps<C>>(
28        transcript: &mut T,
29        U_i: &U,
30    ) -> Vec<CF1<C>> {
31        let mut challenges = vec![];
32        for cm in U_i.get_commitments() {
33            transcript.absorb_nonnative(&cm);
34            challenges.push(transcript.get_challenge());
35        }
36        challenges
37    }
38
39    pub fn get_challenges_gadget<
40        C: Curve,
41        S: CryptographicSponge,
42        T: TranscriptVar<CF1<C>, S>,
43        U: CommittedInstanceVarOps<C>,
44    >(
45        transcript: &mut T,
46        U_i: &U,
47    ) -> Result<Vec<FpVar<CF1<C>>>, SynthesisError> {
48        let mut challenges = vec![];
49        for cm in U_i.get_commitments() {
50            transcript.absorb_nonnative(&cm)?;
51            challenges.push(transcript.get_challenge()?);
52        }
53        Ok(challenges)
54    }
55}

57/// Gadget that interpolates the polynomial from the given vector and returns
58/// its evaluation at the given point.
59/// It also offers the rust native implementation compatible with the gadget.
60pub struct EvalGadget {}
61
62impl EvalGadget {
63    pub fn evaluate_native<F: PrimeField>(v: &[F], point: F) -> Result<F, Error> {
64        let mut v = v.to_vec();
65        v.resize(v.len().next_power_of_two(), F::zero());
66
67        Ok(poly_from_vec(v)?.evaluate(&point))
68    }
69
70    pub fn evaluate_gadget<F: PrimeField>(
71        v: &[FpVar<F>],
72        point: &FpVar<F>,
73    ) -> Result<FpVar<F>, SynthesisError> {
74        let mut v = v.to_vec();
75        v.resize(v.len().next_power_of_two(), FpVar::zero());
76        let n = v.len() as u64;
77        let gen = F::get_root_of_unity(n).ok_or(SynthesisError::PolynomialDegreeTooLarge)?;
78        // `unwrap` below is safe because `Radix2DomainVar::new` only fails if
79        // `offset.enforce_not_equal(&FpVar::zero())` returns an error.
80        // But in our case, `offset` is `FpVar::one()`, i.e., both operands of
81        // `enforce_not_equal` are constants.
82        // Consequently, `FpVar`'s implementation of `enforce_not_equal` will
83        // always return `Ok(())`.
84        let domain = Radix2DomainVar::new(gen, log2(v.len()) as u64, FpVar::one()).unwrap();
85
86        let evaluations_var = EvaluationsVar::from_vec_and_domain(v, domain, true);
87        evaluations_var.interpolate_and_evaluate(point)
88    }
89}

91/// This is a temporary workaround for step 6 (running NIFS.V for group elements
92/// in circuit) in an NIFS-agnostic way, because different folding schemes have
93/// different interfaces of folding verification now.
94///
95/// In the future, we may introduce a better solution that uses a trait for all
96/// folding schemes that specifies their native and in-circuit behaviors.
97pub trait DeciderEnabledNIFS<
98    C: Curve,
99    RU: CommittedInstanceOps<C>, // Running instance
100    IU: CommittedInstanceOps<C>, // Incoming instance
101    W: WitnessOps<CF1<C>>,
102    A: ArithRelation<W, RU>,
103>
104{
105    type ProofDummyCfg;
106    type Proof: Dummy<Self::ProofDummyCfg>;
107    type RandomnessDummyCfg;
108    type Randomness: Dummy<Self::RandomnessDummyCfg>;
109
110    /// Fold the field elements in `U` and `u` inside the circuit.
111    ///
112    /// `U_vec` is `U` expressed as a vector of `FpVar`s, which can be reused
113    /// before or after calling this function to save constraints.
114    #[allow(clippy::too_many_arguments)]
115    fn fold_field_elements_gadget(
116        arith: &A,
117        transcript: &mut PoseidonSpongeVar<CF1<C>>,
118        U: RU::Var,
119        U_vec: Vec<FpVar<CF1<C>>>,
120        u: IU::Var,
121        proof: Self::Proof,
122        randomness: Self::Randomness,
123    ) -> Result<RU::Var, SynthesisError>;
124
125    /// Fold the group elements (i.e., commitments) in `U` and `u` outside the
126    /// circuit.
127    fn fold_group_elements_native(
128        U_commitments: &[C],
129        u_commitments: &[C],
130        proof: Option<Self::Proof>,
131        randomness: Self::Randomness,
132    ) -> Result<Vec<C>, Error>;
133}

149    // checks that the gadget and native implementations of the challenge computation match
150    #[test]
151    fn test_kzg_challenge_gadget() -> Result<(), Error> {
152        let mut rng = ark_std::test_rng();
153        let poseidon_config = poseidon_canonical_config::<Fr>();
154        let mut transcript = PoseidonSponge::<Fr>::new(&poseidon_config);
155
156        let U_i = CommittedInstance::<Projective> {
157            cmE: Projective::rand(&mut rng),
158            u: Fr::rand(&mut rng),
159            cmW: Projective::rand(&mut rng),
160            x: vec![Fr::rand(&mut rng); 1],
161        };
162
163        // compute the challenge natively
164        let challenges = KZGChallengesGadget::get_challenges_native(&mut transcript, &U_i);
165
166        let cs = ConstraintSystem::<Fr>::new_ref();
167        let U_iVar =
168            CommittedInstanceVar::<Projective>::new_witness(cs.clone(), || Ok(U_i.clone()))?;
169        let mut transcript_var = PoseidonSpongeVar::<Fr>::new(cs.clone(), &poseidon_config);
170
171        let challenges_var =
172            KZGChallengesGadget::get_challenges_gadget(&mut transcript_var, &U_iVar)?;
173        assert!(cs.is_satisfied()?);
174
175        // check that the natively computed and in-circuit computed hashes match
176        assert_eq!(challenges_var.value()?, challenges);
177        Ok(())
178    }

180    #[test]
181    fn test_polynomial_interpolation() -> Result<(), Error> {
182        let mut rng = ark_std::test_rng();
183        let n = 12;
184        let l = 1 << n;
185
186        let v: Vec<Fr> = std::iter::repeat_with(|| Fr::rand(&mut rng))
187            .take(l)
188            .collect();
189        let challenge = Fr::rand(&mut rng);
190
191        use ark_poly::Polynomial;
192        let polynomial = poly_from_vec(v.to_vec())?;
193        let eval = polynomial.evaluate(&challenge);
194
195        let cs = ConstraintSystem::<Fr>::new_ref();
196        let vVar = Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(v))?;
197        let challengeVar = FpVar::<Fr>::new_witness(cs.clone(), || Ok(challenge))?;
198
199        let evalVar = EvalGadget::evaluate_gadget(&vVar, &challengeVar)?;
200
201        assert_eq!(evalVar.value()?, eval);
202        assert!(cs.is_satisfied()?);
203        Ok(())
204    }
