31/// A generic circuit tailored for the onchain (Ethereum's EVM) verification of
32/// IVC proofs, where we support IVC built upon any folding scheme.
33///
34/// Specifically, `GenericDeciderEthCircuit` implements the in-circuit version
35/// of the IVC verification algorithm, which essentially checks the following:
36/// - `R_arith(W_i, U_i)`:
37///   The running instance `U_i` and witness `W_i` satisfy `arith`,
38///   and the commitments in `U_i` open to the values in `W_i`.
39/// - `R_arith(w_i, u_i)`:
40///   The incoming instance `u_i` and witness `w_i` satisfy `arith`,
41///   and the commitments in `u_i` open to the values in `w_i`.
42/// - `R_cf_arith(cf_W_i, cf_U_i)`:
43///   The CycleFold instance `cf_U_i` and witness `cf_W_i` satisfy `cf_arith`,
44///   and the commitments in `cf_U_i` open to the values in `cf_W_i`.
45/// - `u_i` contains the correct hash of the initial and final states.
46///
47/// To reduce the number of relation checks, the prover, before invoking the
48/// circuit, further folds `U_i, u_i` into `U_{i+1}`, and `W_i, w_i` into
49/// `W_{i+1}`.
50/// Now, the circuit only needs to perform two relation checks, i.e.,
51/// `R_arith(W_{i+1}, U_{i+1})` and `R_cf_arith(cf_W_i, cf_U_i)`, plus a few
52/// constraints for enforcing the correct hash in `u_i` and the correct folding
53/// from `U_i, u_i` to `U_{i+1}`.
54///
55/// We further reduce the circuit size by avoiding the non-native commitment
56/// checks involved in `R_arith(W_{i+1}, U_{i+1})`.
57/// Now, we now only check the satisfiability of the constraint system `arith`
58/// with the witness `W_{i+1}` and instance `U_{i+1}` in the circuit, but the
59/// actual commitment checks are done with the help of KZG.
60///
61/// For more details, see [https://privacy-scaling-explorations.github.io/sonobe-docs/design/nova-decider-onchain.html].
62pub struct GenericOnchainDeciderCircuit<
63    C1: Curve,
64    C2: Curve,
65    RU: CommittedInstanceOps<C1>,               // Running instance
66    IU: CommittedInstanceOps<C1>,               // Incoming instance
67    W: WitnessOps<CF1<C1>>,                     // Witness
68    A: ArithRelation<W, RU>,                    // Constraint system
69    AVar: ArithRelationGadget<W::Var, RU::Var>, // In-circuit representation of `A`
70    D: DeciderEnabledNIFS<C1, RU, IU, W, A>,
71> {
72    pub _avar: PhantomData<AVar>,
73    /// Constraint system of the Augmented Function circuit
74    pub arith: A,
75    /// R1CS of the CycleFold circuit
76    pub cf_arith: R1CS<CF1<C2>>,
77    /// CycleFold PedersenParams over C2
78    pub cf_pedersen_params: PedersenParams<C2>,
79    pub poseidon_config: PoseidonConfig<CF1<C1>>,
80    /// public params hash
81    pub pp_hash: CF1<C1>,
82    pub i: CF1<C1>,
83    /// initial state
84    pub z_0: Vec<CF1<C1>>,
85    /// current i-th state
86    pub z_i: Vec<CF1<C1>>,
87    /// Folding scheme instances
88    pub U_i: RU,
89    pub W_i: W,
90    pub u_i: IU,
91    pub w_i: W,
92    pub U_i1: RU,
93    pub W_i1: W,
94
95    /// Helper for folding verification
96    pub proof: D::Proof,
97    pub randomness: D::Randomness,
98
99    /// CycleFold running instance
100    pub cf_U_i: CycleFoldCommittedInstance<C2>,
101    pub cf_W_i: CycleFoldWitness<C2>,
102
103    /// KZG challenges
104    pub kzg_challenges: Vec<CF1<C1>>,
105    pub kzg_evaluations: Vec<CF1<C1>>,
106}

176impl<
177        C1: Curve,
178        C2: Curve<ScalarField = CF2<C1>, BaseField = CF1<C1>>,
179        RU: CommittedInstanceOps<C1>,
180        IU: CommittedInstanceOps<C1>,
181        W: WitnessOps<CF1<C1>>,
182        A: ArithRelation<W, RU>,
183        AVar: ArithRelationGadget<W::Var, RU::Var> + AllocVar<A, CF1<C1>>,
184        D: DeciderEnabledNIFS<C1, RU, IU, W, A>,
185    > ConstraintSynthesizer<CF1<C1>> for GenericOnchainDeciderCircuit<C1, C2, RU, IU, W, A, AVar, D>
186where
187    RU::Var: AbsorbGadget<CF1<C1>> + CommittedInstanceVarOps<C1, PointVar = NonNativeAffineVar<C1>>,
188{
189    fn generate_constraints(self, cs: ConstraintSystemRef<CF1<C1>>) -> Result<(), SynthesisError> {
190        let arith = AVar::new_witness(cs.clone(), || Ok(&self.arith))?;
191
192        let pp_hash = FpVar::new_input(cs.clone(), || Ok(self.pp_hash))?;
193        let i = FpVar::new_input(cs.clone(), || Ok(self.i))?;
194        let z_0 = Vec::new_input(cs.clone(), || Ok(self.z_0))?;
195        let z_i = Vec::new_input(cs.clone(), || Ok(self.z_i))?;
196
197        let u_i = IU::Var::new_witness(cs.clone(), || Ok(self.u_i))?;
198        let U_i = RU::Var::new_witness(cs.clone(), || Ok(self.U_i))?;
199        // here (U_i1, W_i1) = NIFS.P( (U_i,W_i), (u_i,w_i))
200        let U_i1_commitments = Vec::<NonNativeAffineVar<C1>>::new_input(cs.clone(), || {
201            Ok(self.U_i1.get_commitments())
202        })?;
203        let U_i1 = RU::Var::new_witness(cs.clone(), || Ok(self.U_i1))?;
204        let W_i1 = W::Var::new_witness(cs.clone(), || Ok(self.W_i1))?;
205        U_i1.get_commitments().enforce_equal(&U_i1_commitments)?;
206
207        let cf_U_i =
208            CycleFoldCommittedInstanceVar::<C2>::new_witness(cs.clone(), || Ok(self.cf_U_i))?;
209
210        // allocate the inputs for the check 7.1 and 7.2
211        let kzg_challenges = Vec::new_input(cs.clone(), || Ok(self.kzg_challenges))?;
212        let kzg_evaluations = Vec::new_input(cs.clone(), || Ok(self.kzg_evaluations))?;
213
214        // `sponge` is for digest computation.
215        let sponge = PoseidonSpongeVar::new_with_pp_hash(&self.poseidon_config, &pp_hash)?;
216        // `transcript` is for challenge generation.
217        let mut transcript = sponge.clone();
218
219        // NOTE: we use the same enumeration as in
220        // https://privacy-scaling-explorations.github.io/sonobe-docs/design/nova-decider-onchain.html
221        // in order to make it easier to reason about.
222
223        // 1. enforce `U_{i+1}` and `W_{i+1}` satisfy `arith`
224        arith.enforce_relation(&W_i1, &U_i1)?;
225
226        // 2. enforce `u_i` is an incoming instance
227        u_i.enforce_incoming()?;
228
229        // 3. u_i.x[0] == H(i, z_0, z_i, U_i), u_i.x[1] == H(cf_U_i)
230        let (u_i_x, U_i_vec) = U_i.hash(&sponge, &i, &z_0, &z_i)?;
231        let (cf_u_i_x, _) = cf_U_i.hash(&sponge)?;
232        u_i.get_public_inputs().enforce_equal(&[u_i_x, cf_u_i_x])?;
233
234        #[cfg(feature = "light-test")]
235        log::warn!("[WARNING]: Running with the 'light-test' feature, skipping the big part of the DeciderEthCircuit.\n           Only for testing purposes.");
236
237        // The following two checks (and their respective allocations) are disabled for normal
238        // tests since they take several millions of constraints and would take several minutes
239        // (and RAM) to run the test. It is active by default, and not active only when
240        // 'light-test' feature is used.
241        #[cfg(not(feature = "light-test"))]
242        {
243            // imports here instead of at the top of the file, so we avoid having multiple
244            // `#[cfg(not(test))]`
245            use crate::{
246                arith::r1cs::circuits::R1CSMatricesVar,
247                commitment::pedersen::PedersenGadget,
248                folding::circuits::{
249                    cyclefold::CycleFoldWitnessVar, nonnative::uint::NonNativeUintVar,
250                },
251            };
252            use ark_r1cs_std::{convert::ToBitsGadget, groups::CurveVar};
253            let cf_W_i = CycleFoldWitnessVar::<C2>::new_witness(cs.clone(), || Ok(self.cf_W_i))?;
254            // 4. check Pedersen commitments of cf_U_i.{cmE, cmW}
255            let H = C2::Var::constant(self.cf_pedersen_params.h);
256            let G = self
257                .cf_pedersen_params
258                .generators
259                .iter()
260                .map(|&g| C2::Var::constant(g.into()))
261                .collect::<Vec<_>>();
262            let cf_W_i_E_bits = cf_W_i
263                .E
264                .iter()
265                .map(|E_i| E_i.to_bits_le())
266                .collect::<Result<Vec<_>, _>>()?;
267            let cf_W_i_W_bits = cf_W_i
268                .W
269                .iter()
270                .map(|W_i| W_i.to_bits_le())
271                .collect::<Result<Vec<_>, _>>()?;
272            PedersenGadget::<C2>::commit(&H, &G, &cf_W_i_E_bits, &cf_W_i.rE.to_bits_le()?)?
273                .enforce_equal(&cf_U_i.cmE)?;
274            PedersenGadget::<C2>::commit(&H, &G, &cf_W_i_W_bits, &cf_W_i.rW.to_bits_le()?)?
275                .enforce_equal(&cf_U_i.cmW)?;
276
277            let cf_r1cs = R1CSMatricesVar::<CF1<C2>, NonNativeUintVar<CF2<C2>>>::new_constant(
278                ConstraintSystemRef::None,
279                self.cf_arith,
280            )?;
281
282            // 5. enforce `cf_U_i` and `cf_W_i` satisfy `cf_r1cs`
283            cf_r1cs.enforce_relation(&cf_W_i, &cf_U_i)?;
284        }
285
286        // 6.1. partially enforce `NIFS.V(U_i, u_i) = U_{i+1}`.
287        D::fold_field_elements_gadget(
288            &self.arith,
289            &mut transcript,
290            U_i,
291            U_i_vec,
292            u_i,
293            self.proof,
294            self.randomness,
295        )?
296        .enforce_partial_equal(&U_i1)?;
297
298        // 7.1. compute and check KZG challenges
299        KZGChallengesGadget::get_challenges_gadget(&mut transcript, &U_i1)?
300            .enforce_equal(&kzg_challenges)?;
301
302        // 7.2. check the claimed evaluations
303        for (((v, _r), c), e) in W_i1
304            .get_openings()
305            .iter()
306            .zip(&kzg_challenges)
307            .zip(&kzg_evaluations)
308        {
309            // The randomness `_r` is currently not used.
310            EvalGadget::evaluate_gadget(v, c)?.enforce_equal(e)?;
311        }
312
313        Ok(())
314    }
