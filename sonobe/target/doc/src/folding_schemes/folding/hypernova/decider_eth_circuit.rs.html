37impl<C: Curve> ArithRelationGadget<WitnessVar<CF1<C>>, LCCCSVar<C>> for CCSMatricesVar<CF1<C>> {
38    type Evaluation = Vec<FpVar<CF1<C>>>;
39
40    fn eval_relation(
41        &self,
42        w: &WitnessVar<CF1<C>>,
43        u: &LCCCSVar<C>,
44    ) -> Result<Self::Evaluation, SynthesisError> {
45        let z = [&[u.u.clone()][..], &u.x, &w.w].concat();
46
47        self.M
48            .iter()
49            .map(|M_j| {
50                let s = log2(M_j.n_rows) as usize;
51                let Mz = M_j.mul_vector(&z)?;
52                Ok(eval_mle(s, Mz, u.r_x.clone()))
53            })
54            .collect()
55    }
56
57    fn enforce_evaluation(
58        _w: &WitnessVar<CF1<C>>,
59        u: &LCCCSVar<C>,
60        v: Self::Evaluation,
61    ) -> Result<(), SynthesisError> {
62        v.enforce_equal(&u.v)
63    }
64}

66/// In-circuit representation of the Witness associated to the CommittedInstance.
67#[derive(Debug, Clone)]
68pub struct WitnessVar<F: PrimeField> {
69    pub w: Vec<FpVar<F>>,
70    pub r_w: FpVar<F>,
71}
72
73impl<F: PrimeField> AllocVar<Witness<F>, F> for WitnessVar<F> {
74    fn new_variable<T: Borrow<Witness<F>>>(
75        cs: impl Into<Namespace<F>>,
76        f: impl FnOnce() -> Result<T, SynthesisError>,
77        mode: AllocationMode,
78    ) -> Result<Self, SynthesisError> {
79        f().and_then(|val| {
80            let cs = cs.into();
81
82            let w: Vec<FpVar<F>> =
83                Vec::new_variable(cs.clone(), || Ok(val.borrow().w.clone()), mode)?;
84            let r_w = FpVar::<F>::new_variable(cs.clone(), || Ok(val.borrow().r_w), mode)?;
85
86            Ok(Self { w, r_w })
87        })
88    }
89}

91impl<F: PrimeField> WitnessVarOps<F> for WitnessVar<F> {
92    fn get_openings(&self) -> Vec<(&[FpVar<F>], FpVar<F>)> {
93        vec![(&self.w, self.r_w.clone())]
94    }
95}

97pub type DeciderEthCircuit<C1, C2> = GenericOnchainDeciderCircuit<
98    C1,
99    C2,
100    LCCCS<C1>,
101    CCCS<C1>,
102    Witness<CF1<C1>>,
103    CCS<CF1<C1>>,
104    CCSMatricesVar<CF1<C1>>,
105    DeciderHyperNovaGadget,
106>;
107
108impl<
109        C1: Curve,
110        C2: Curve,
111        FC: FCircuit<C1::ScalarField>,
112        CS1: CommitmentScheme<C1, H>,
113        // enforce that the CS2 is Pedersen commitment scheme, since we're at Ethereum's EVM decider
114        CS2: CommitmentScheme<C2, H, ProverParams = PedersenParams<C2>>,
115        const MU: usize,
116        const NU: usize,
117        const H: bool,
118    > TryFrom<HyperNova<C1, C2, FC, CS1, CS2, MU, NU, H>> for DeciderEthCircuit<C1, C2>
119{
120    type Error = Error;
121
122    fn try_from(hn: HyperNova<C1, C2, FC, CS1, CS2, MU, NU, H>) -> Result<Self, Error> {
123        // compute the U_{i+1}, W_{i+1}, by folding the last running & incoming instances
124        let mut transcript = PoseidonSponge::new_with_pp_hash(&hn.poseidon_config, hn.pp_hash);
125        let (nimfs_proof, U_i1, W_i1, rho) = NIMFS::<C1, PoseidonSponge<C1::ScalarField>>::prove(
126            &mut transcript,
127            &hn.ccs,
128            &[hn.U_i.clone()],
129            &[hn.u_i.clone()],
130            &[hn.W_i.clone()],
131            &[hn.w_i.clone()],
132        )?;
133
134        // compute the KZG challenges used as inputs in the circuit
135        let kzg_challenges = KZGChallengesGadget::get_challenges_native(&mut transcript, &U_i1);
136
137        // get KZG evals
138        let kzg_evaluations = W_i1
139            .get_openings()
140            .iter()
141            .zip(&kzg_challenges)
142            .map(|((v, _), &c)| EvalGadget::evaluate_native(v, c))
143            .collect::<Result<Vec<_>, _>>()?;
144
145        Ok(Self {
146            _avar: PhantomData,
147            arith: hn.ccs,
148            cf_arith: hn.cf_r1cs,
149            cf_pedersen_params: hn.cf_cs_pp,
150            poseidon_config: hn.poseidon_config,
151            pp_hash: hn.pp_hash,
152            i: hn.i,
153            z_0: hn.z_0,
154            z_i: hn.z_i,
155            U_i: hn.U_i,
156            W_i: hn.W_i,
157            u_i: hn.u_i,
158            w_i: hn.w_i,
159            U_i1,
160            W_i1,
161            proof: nimfs_proof,
162            randomness: rho,
163            cf_U_i: hn.cf_U_i,
164            cf_W_i: hn.cf_W_i,
165            kzg_challenges,
166            kzg_evaluations,
167        })
168    }
169}

173impl<C: Curve> DeciderEnabledNIFS<C, LCCCS<C>, CCCS<C>, Witness<C::ScalarField>, CCS<CF1<C>>>
174    for DeciderHyperNovaGadget
175{
176    type ProofDummyCfg = (usize, usize, usize, usize);
177    type Proof = NIMFSProof<C>;
178    type Randomness = CF1<C>;
179    type RandomnessDummyCfg = ();
180
181    fn fold_field_elements_gadget(
182        arith: &CCS<CF1<C>>,
183        transcript: &mut PoseidonSpongeVar<CF1<C>>,
184        U: LCCCSVar<C>,
185        _U_vec: Vec<FpVar<CF1<C>>>,
186        u: CCCSVar<C>,
187        proof: Self::Proof,
188        randomness: Self::Randomness,
189    ) -> Result<LCCCSVar<C>, SynthesisError> {
190        let cs = U.u.cs();
191        let nimfs_proof = NIMFSProofVar::<C>::new_witness(cs.clone(), || Ok(proof))?;
192        let rho = FpVar::<CF1<C>>::new_input(cs.clone(), || Ok(randomness))?;
193        let (computed_U_i1, rho_bits) = NIMFSGadget::<C>::verify(
194            cs.clone(),
195            arith,
196            transcript,
197            &[U],
198            &[u],
199            nimfs_proof,
200            Boolean::TRUE, // enabled
201        )?;
202        Boolean::le_bits_to_fp(&rho_bits)?.enforce_equal(&rho)?;
203        Ok(computed_U_i1)
204    }
205
206    fn fold_group_elements_native(
207        U_commitments: &[C],
208        u_commitments: &[C],
209        _: Option<Self::Proof>,
210        r: Self::Randomness,
211    ) -> Result<Vec<C>, Error> {
212        let U_C = U_commitments[0];
213        let u_C = u_commitments[0];
214        let C = U_C + u_C.mul(r);
215        Ok(vec![C])
216    }
217}

234    #[test]
235    fn test_lcccs_checker_gadget() -> Result<(), Error> {
236        let mut rng = test_rng();
237        let n_rows = 2_u32.pow(5) as usize;
238        let n_cols = 2_u32.pow(5) as usize;
239        let r1cs = R1CS::<Fr>::rand(&mut rng, n_rows, n_cols);
240        let ccs = CCS::from(r1cs);
241        let z: Vec<Fr> = (0..n_cols).map(|_| Fr::rand(&mut rng)).collect();
242
243        let (pedersen_params, _) = Pedersen::<Projective>::setup(&mut rng, ccs.n_witnesses())?;
244
245        let (lcccs, w) = ccs.to_lcccs::<_, Projective, Pedersen<Projective>, false>(
246            &mut rng,
247            &pedersen_params,
248            &z,
249        )?;
250
251        let cs = ConstraintSystem::<Fr>::new_ref();
252
253        // CCS's (sparse) matrices are constants in the circuit
254        let ccs_mat = CCSMatricesVar::<Fr>::new_constant(cs.clone(), ccs.clone())?;
255        let w_var = WitnessVar::new_witness(cs.clone(), || Ok(w))?;
256        let lcccs_var = LCCCSVar::new_input(cs.clone(), || Ok(lcccs))?;
257
258        ccs_mat.enforce_relation(&w_var, &lcccs_var)?;
259
260        assert!(cs.is_satisfied()?);
261        Ok(())
262    }

266        let mut rng = ark_std::test_rng();
267        let poseidon_config = poseidon_canonical_config::<Fr>();
268
269        let F_circuit = CubicFCircuit::<Fr>::new(())?;
270        let z_0 = vec![Fr::from(3_u32)];
271
272        const MU: usize = 1;
273        const NU: usize = 1;
274
275        type HN = HyperNova<
276            Projective,
277            Projective2,
278            CubicFCircuit<Fr>,
279            Pedersen<Projective>,
280            Pedersen<Projective2>,
281            MU,
282            NU,
283            false,
284        >;
285        let prep_param = PreprocessorParam::<
286            Projective,
287            Projective2,
288            CubicFCircuit<Fr>,
289            Pedersen<Projective>,
290            Pedersen<Projective2>,
291            false,
292        >::new(poseidon_config, F_circuit);
293        let hn_params = HN::preprocess(&mut rng, &prep_param)?;
294
295        // generate a Nova instance and do a step of it
296        let mut hypernova = HN::init(&hn_params, F_circuit, z_0.clone())?;
297        hypernova.prove_step(&mut rng, (), None)?;
298
299        let ivc_proof = hypernova.ivc_proof();
300        HN::verify(hn_params.1, ivc_proof)?;
301
302        // load the DeciderEthCircuit from the generated Nova instance
303        let decider_circuit = DeciderEthCircuit::<Projective, Projective2>::try_from(hypernova)?;
304
305        let cs = ConstraintSystem::<Fr>::new_ref();
306
307        // generate the constraints and check that are satisfied by the inputs
308        decider_circuit.generate_constraints(cs.clone())?;
309        assert!(cs.is_satisfied()?);
310        dbg!(cs.num_constraints());
311        Ok(())
312    }
