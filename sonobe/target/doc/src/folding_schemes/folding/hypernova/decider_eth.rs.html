21#[derive(Debug, Clone, Eq, PartialEq, CanonicalSerialize, CanonicalDeserialize)]
22pub struct Proof<C1, CS1, S>
23where
24    C1: Curve,
25    CS1: CommitmentScheme<C1, ProverChallenge = C1::ScalarField, Challenge = C1::ScalarField>,
26    S: SNARK<C1::ScalarField>,
27{
28    snark_proof: S::Proof,
29    kzg_proof: CS1::Proof,
30    // rho used at the last fold, U_{i+1}=NIMFS.V(rho, U_i, u_i), it is checked in-circuit
31    rho: C1::ScalarField,
32    // the KZG challenge is provided by the prover, but in-circuit it is checked to match
33    // the in-circuit computed computed one.
34    kzg_challenge: C1::ScalarField,
35}

37/// Onchain Decider, for ethereum use cases
38#[derive(Clone, Debug)]
39pub struct Decider<C1, C2, FC, CS1, CS2, S, FS, const MU: usize, const NU: usize> {
40    _c1: PhantomData<C1>,
41    _c2: PhantomData<C2>,
42    _fc: PhantomData<FC>,
43    _cs1: PhantomData<CS1>,
44    _cs2: PhantomData<CS2>,
45    _s: PhantomData<S>,
46    _fs: PhantomData<FS>,
47}
48
49impl<C1, C2, FC, CS1, CS2, S, FS, const MU: usize, const NU: usize> DeciderTrait<C1, C2, FC, FS>
50    for Decider<C1, C2, FC, CS1, CS2, S, FS, MU, NU>
51where
52    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
53    C2: Curve,
54    FC: FCircuit<C1::ScalarField>,
55    // CS1 is a KZG commitment, where challenge is C1::Fr elem
56    CS1: CommitmentScheme<
57        C1,
58        ProverChallenge = C1::ScalarField,
59        Challenge = C1::ScalarField,
60        Proof = KZGProof<C1>,
61    >,
62    // enforce that the CS2 is Pedersen commitment scheme, since we're at Ethereum's EVM decider
63    CS2: CommitmentScheme<C2, ProverParams = PedersenParams<C2>>,
64    S: SNARK<C1::ScalarField>,
65    FS: FoldingScheme<C1, C2, FC>,
66    // constrain FS into HyperNova, since this is a Decider specifically for HyperNova
67    HyperNova<C1, C2, FC, CS1, CS2, MU, NU, false>: From<FS>,
68    crate::folding::hypernova::ProverParams<C1, C2, CS1, CS2, false>:
69        From<<FS as FoldingScheme<C1, C2, FC>>::ProverParam>,
70    crate::folding::hypernova::VerifierParams<C1, C2, CS1, CS2, false>:
71        From<<FS as FoldingScheme<C1, C2, FC>>::VerifierParam>,
72{
73    type PreprocessorParam = ((FS::ProverParam, FS::VerifierParam), usize);
74    type ProverParam = (S::ProvingKey, CS1::ProverParams);
75    type Proof = Proof<C1, CS1, S>;
76    type VerifierParam = VerifierParam<C1, CS1::VerifierParams, S::VerifyingKey>;
77    type PublicInput = Vec<C1::ScalarField>;
78    type CommittedInstance = Vec<C1>;
79
80    fn preprocess(
81        mut rng: impl RngCore + CryptoRng,
82        ((pp, vp), state_len): Self::PreprocessorParam,
83    ) -> Result<(Self::ProverParam, Self::VerifierParam), Error> {
84        // get the FoldingScheme prover & verifier params from HyperNova
85        let hypernova_pp: <HyperNova<C1, C2, FC, CS1, CS2, MU, NU, false> as FoldingScheme<
86            C1,
87            C2,
88            FC,
89        >>::ProverParam = pp.into();
90        let hypernova_vp: <HyperNova<C1, C2, FC, CS1, CS2, MU, NU, false> as FoldingScheme<
91            C1,
92            C2,
93            FC,
94        >>::VerifierParam = vp.into();
95        let pp_hash = hypernova_vp.pp_hash()?;
96
97        let s = hypernova_vp.ccs.s;
98        let t = hypernova_vp.ccs.t;
99
100        let circuit = DeciderEthCircuit::<C1, C2>::dummy((
101            hypernova_vp.ccs,
102            hypernova_vp.cf_r1cs,
103            hypernova_pp.cf_cs_pp,
104            hypernova_pp.poseidon_config,
105            (s, t, MU, NU),
106            (),
107            state_len,
108            1, // HyperNova's LCCCS contains 1 commitment
109        ));
110
111        // get the Groth16 specific setup for the circuit
112        let (g16_pk, g16_vk) = S::circuit_specific_setup(circuit, &mut rng)
113            .map_err(|e| Error::SNARKSetupFail(e.to_string()))?;
114
115        let pp = (g16_pk, hypernova_pp.cs_pp);
116
117        let vp = Self::VerifierParam {
118            pp_hash,
119            snark_vp: g16_vk,
120            cs_vp: hypernova_vp.cs_vp,
121        };
122        Ok((pp, vp))
123    }
124
125    fn prove(
126        mut rng: impl RngCore + CryptoRng,
127        pp: Self::ProverParam,
128        folding_scheme: FS,
129    ) -> Result<Self::Proof, Error> {
130        let (snark_pk, cs_pk): (S::ProvingKey, CS1::ProverParams) = pp;
131
132        let circuit = DeciderEthCircuit::<C1, C2>::try_from(HyperNova::from(folding_scheme))?;
133
134        let rho = circuit.randomness;
135
136        // get the challenges that have been already computed when preparing the circuit inputs in
137        // the above `try_from` call
138        let kzg_challenges = circuit.kzg_challenges.clone();
139
140        // generate KZG proofs
141        let kzg_proofs = circuit
142            .W_i1
143            .get_openings()
144            .iter()
145            .zip(&kzg_challenges)
146            .map(|((v, _), &c)| {
147                CS1::prove_with_challenge(&cs_pk, c, v, &C1::ScalarField::zero(), None)
148            })
149            .collect::<Result<Vec<_>, _>>()?;
150
151        let snark_proof =
152            S::prove(&snark_pk, circuit, &mut rng).map_err(|e| Error::Other(e.to_string()))?;
153
154        Ok(Self::Proof {
155            snark_proof,
156            rho,
157            kzg_proof: (kzg_proofs.len() == 1)
158                .then(|| kzg_proofs[0].clone())
159                .ok_or(Error::NotExpectedLength(kzg_proofs.len(), 1))?,
160            kzg_challenge: (kzg_challenges.len() == 1)
161                .then(|| kzg_challenges[0])
162                .ok_or(Error::NotExpectedLength(kzg_challenges.len(), 1))?,
163        })
164    }
165
166    fn verify(
167        vp: Self::VerifierParam,
168        i: C1::ScalarField,
169        z_0: Vec<C1::ScalarField>,
170        z_i: Vec<C1::ScalarField>,
171        // we don't use the instances at the verifier level, since we check them in-circuit
172        running_commitments: &Self::CommittedInstance,
173        incoming_commitments: &Self::CommittedInstance,
174        proof: &Self::Proof,
175    ) -> Result<bool, Error> {
176        if i <= C1::ScalarField::one() {
177            return Err(Error::NotEnoughSteps);
178        }
179
180        let Self::VerifierParam {
181            pp_hash,
182            snark_vp,
183            cs_vp,
184        } = vp;
185
186        // 6.2. Fold the commitments
187        let C = DeciderHyperNovaGadget::fold_group_elements_native(
188            running_commitments,
189            incoming_commitments,
190            None,
191            proof.rho,
192        )?[0];
193
194        // Note: the NIMFS proof is checked inside the DeciderEthCircuit, which ensures that the
195        // 'proof.U_i1' is correctly computed
196        let public_input: Vec<C1::ScalarField> = [
197            &[pp_hash, i][..],
198            &z_0,
199            &z_i,
200            &C.inputize_nonnative(),
201            &[proof.kzg_challenge, proof.kzg_proof.eval, proof.rho],
202        ]
203        .concat();
204
205        let snark_v = S::verify(&snark_vp, &public_input, &proof.snark_proof)
206            .map_err(|e| Error::Other(e.to_string()))?;
207        if !snark_v {
208            return Err(Error::SNARKVerificationFail);
209        }
210
211        // 7.3. Verify the KZG proof
212        // we're at the Ethereum EVM case, so the CS1 is KZG commitments
213        CS1::verify_with_challenge(&cs_vp, proof.kzg_challenge, &C, &proof.kzg_proof)?;
214
215        Ok(true)
216    }
217}

235    #[test]
236    fn test_decider() -> Result<(), Error> {
237        const MU: usize = 1;
238        const NU: usize = 1;
239        // use HyperNova as FoldingScheme
240        type HN = HyperNova<
241            Projective,
242            Projective2,
243            CubicFCircuit<Fr>,
244            KZG<'static, Bn254>,
245            Pedersen<Projective2>,
246            MU,
247            NU,
248            false,
249        >;
250        type D = Decider<
251            Projective,
252            Projective2,
253            CubicFCircuit<Fr>,
254            KZG<'static, Bn254>,
255            Pedersen<Projective2>,
256            Groth16<Bn254>, // here we define the Snark to use in the decider
257            HN,             // here we define the FoldingScheme to use
258            MU,
259            NU,
260        >;
261
262        let mut rng = rand::rngs::OsRng;
263        let poseidon_config = poseidon_canonical_config::<Fr>();
264
265        let F_circuit = CubicFCircuit::<Fr>::new(())?;
266        let z_0 = vec![Fr::from(3_u32)];
267
268        let prep_param = PreprocessorParam::new(poseidon_config, F_circuit);
269        let hypernova_params = HN::preprocess(&mut rng, &prep_param)?;
270
271        let mut hypernova = HN::init(&hypernova_params, F_circuit, z_0.clone())?;
272        hypernova.prove_step(&mut rng, (), Some((vec![], vec![])))?;
273        hypernova.prove_step(&mut rng, (), Some((vec![], vec![])))?; // do a 2nd step
274
275        // prepare the Decider prover & verifier params
276        let (decider_pp, decider_vp) =
277            D::preprocess(&mut rng, (hypernova_params, F_circuit.state_len()))?;
278
279        // decider proof generation
280        let proof = D::prove(rng, decider_pp, hypernova.clone())?;
281
282        // decider proof verification
283        let verified = D::verify(
284            decider_vp,
285            hypernova.i,
286            hypernova.z_0,
287            hypernova.z_i,
288            &hypernova.U_i.get_commitments(),
289            &hypernova.u_i.get_commitments(),
290            &proof,
291        )?;
292        assert!(verified);
293        Ok(())
294    }
