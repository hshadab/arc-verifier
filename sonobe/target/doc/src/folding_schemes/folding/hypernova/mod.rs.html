48/// Configuration for HyperNova's CycleFold circuit
49pub struct HyperNovaCycleFoldConfig<C: Curve, const MU: usize, const NU: usize> {
50    r: CF1<C>,
51    points: Vec<C>,
52}
53
54impl<C: Curve, const MU: usize, const NU: usize> Default for HyperNovaCycleFoldConfig<C, MU, NU> {
55    fn default() -> Self {
56        Self {
57            r: CF1::<C>::zero(),
58            points: vec![C::zero(); MU + NU],
59        }
60    }
61}
62
63impl<C: Curve, const MU: usize, const NU: usize> CycleFoldConfig<C>
64    for HyperNovaCycleFoldConfig<C, MU, NU>
65{
66    const RANDOMNESS_BIT_LENGTH: usize = NOVA_N_BITS_RO;
67    const N_INPUT_POINTS: usize = MU + NU;
68    const N_UNIQUE_RANDOMNESSES: usize = 1;
69
70    fn alloc_points(&self, cs: ConstraintSystemRef<CF2<C>>) -> Result<Vec<C::Var>, SynthesisError> {
71        let points = Vec::new_witness(cs.clone(), || Ok(self.points.clone()))?;
72        for point in &points {
73            Self::mark_point_as_public(point)?;
74        }
75        Ok(points)
76    }
77
78    fn alloc_randomnesses(
79        &self,
80        cs: ConstraintSystemRef<CF2<C>>,
81    ) -> Result<Vec<Vec<Boolean<CF2<C>>>>, SynthesisError> {
82        let one = &CF1::<C>::one().into_bigint().to_bits_le()[..NOVA_N_BITS_RO];
83        let r = &self.r.into_bigint().to_bits_le()[..NOVA_N_BITS_RO];
84        let one_var = Vec::new_constant(cs.clone(), one)?;
85        let r_var = Vec::new_witness(cs.clone(), || Ok(r))?;
86        Self::mark_randomness_as_public(&r_var)?;
87        Ok([vec![one_var], vec![r_var; MU + NU - 1]].concat())
88    }
89}

91/// Witness for the LCCCS & CCCS, containing the w vector, and the r_w used as randomness in the Pedersen commitment.
92#[derive(Debug, Clone, Eq, PartialEq, CanonicalSerialize, CanonicalDeserialize)]
93pub struct Witness<F: PrimeField> {
94    pub w: Vec<F>,
95    pub r_w: F,
96}
97
98impl<F: PrimeField> Witness<F> {
99    pub fn new(w: Vec<F>) -> Self {
100        // note: at the current version, we don't use the blinding factors and we set them to 0
101        // always.
102        Self { w, r_w: F::zero() }
103    }
104}

106impl<F: PrimeField> Dummy<&CCS<F>> for Witness<F> {
107    fn dummy(ccs: &CCS<F>) -> Self {
108        Self::new(vec![F::zero(); ccs.n_witnesses()])
109    }
110}

113    type Var = WitnessVar<F>;

117    }

258    /// R1CS of the CycleFold circuit
259    pub cf_r1cs: R1CS<C2::ScalarField>,
260    pub poseidon_config: PoseidonConfig<C1::ScalarField>,
261    /// CommitmentScheme::ProverParams over C1
262    pub cs_pp: CS1::ProverParams,
263    /// CycleFold CommitmentScheme::ProverParams, over C2
264    pub cf_cs_pp: CS2::ProverParams,
265    /// F circuit, the circuit that is being folded
266    pub F: FC,
267    /// public params hash
268    pub pp_hash: C1::ScalarField,
269    pub i: C1::ScalarField,
270    /// initial state
271    pub z_0: Vec<C1::ScalarField>,
272    /// current i-th state
273    pub z_i: Vec<C1::ScalarField>,
274    /// HyperNova instances
275    pub W_i: Witness<C1::ScalarField>,
276    pub U_i: LCCCS<C1>,
277    pub w_i: Witness<C1::ScalarField>,
278    pub u_i: CCCS<C1>,

281    pub cf_W_i: CycleFoldWitness<C2>,

285impl<C1, C2, FC, CS1, CS2, const MU: usize, const NU: usize, const H: bool> MultiFolding<C1, C2, FC>
286    for HyperNova<C1, C2, FC, CS1, CS2, MU, NU, H>
287where
288    C1: Curve,
289    C2: Curve,
290    FC: FCircuit<C1::ScalarField>,
291    CS1: CommitmentScheme<C1, H>,
292    CS2: CommitmentScheme<C2, H>,
293    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
294{
295    type RunningInstance = (LCCCS<C1>, Witness<C1::ScalarField>);
296    type IncomingInstance = (CCCS<C1>, Witness<C1::ScalarField>);
297    type MultiInstance = (Vec<Self::RunningInstance>, Vec<Self::IncomingInstance>);
298
299    /// Creates a new LCCS instance for the given state, which satisfies the HyperNova.CCS. This
300    /// method can be used to generate the 'other' LCCS instances to be folded in the multi-folding
301    /// step.
302    fn new_running_instance(
303        &self,
304        mut rng: impl RngCore,
305        state: Vec<C1::ScalarField>,
306        external_inputs: FC::ExternalInputs,
307    ) -> Result<Self::RunningInstance, Error> {
308        let r1cs_z = self.new_instance_generic(state, external_inputs)?;
309        // compute committed instances, w_{i+1}, u_{i+1}, which will be used as w_i, u_i, so we
310        // assign them directly to w_i, u_i.
311        let (U_i, W_i) = self
312            .ccs
313            .to_lcccs::<_, _, CS1, H>(&mut rng, &self.cs_pp, &r1cs_z)?;
314
315        #[cfg(test)]
316        self.ccs.check_relation(&W_i, &U_i)?;
317
318        Ok((U_i, W_i))
319    }
320
321    /// Creates a new CCCS instance for the given state, which satisfies the HyperNova.CCS. This
322    /// method can be used to generate the 'other' CCCS instances to be folded in the multi-folding
323    /// step.
324    fn new_incoming_instance(
325        &self,
326        mut rng: impl RngCore,
327        state: Vec<C1::ScalarField>,
328        external_inputs: FC::ExternalInputs,
329    ) -> Result<Self::IncomingInstance, Error> {
330        let r1cs_z = self.new_instance_generic(state, external_inputs)?;
331        // compute committed instances, w_{i+1}, u_{i+1}, which will be used as w_i, u_i, so we
332        // assign them directly to w_i, u_i.
333        let (u_i, w_i) = self
334            .ccs
335            .to_cccs::<_, _, CS1, H>(&mut rng, &self.cs_pp, &r1cs_z)?;
336
337        #[cfg(test)]
338        self.ccs.check_relation(&w_i, &u_i)?;
339
340        Ok((u_i, w_i))
341    }
342}

344impl<C1, C2, FC, CS1, CS2, const MU: usize, const NU: usize, const H: bool>
345    HyperNova<C1, C2, FC, CS1, CS2, MU, NU, H>
346where
347    C1: Curve,
348    C2: Curve,
349    FC: FCircuit<C1::ScalarField>,
350    CS1: CommitmentScheme<C1, H>,
351    CS2: CommitmentScheme<C2, H>,
352    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
353{
354    /// internal helper for new_running_instance & new_incoming_instance methods, returns the R1CS
355    /// z=[u,x,w] vector to be used to create the LCCCS & CCCS fresh instances.
356    fn new_instance_generic(
357        &self,
358        state: Vec<C1::ScalarField>,
359        external_inputs: FC::ExternalInputs,
360    ) -> Result<Vec<C1::ScalarField>, Error> {
361        // prepare the initial dummy instances
362        let U_i = LCCCS::<C1>::dummy(&self.ccs);
363        let mut u_i = CCCS::<C1>::dummy(&self.ccs);
364        let (_, cf_U_i): (CycleFoldWitness<C2>, CycleFoldCommittedInstance<C2>) =
365            self.cf_r1cs.dummy_witness_instance();
366
367        let sponge = PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(
368            &self.poseidon_config,
369            self.pp_hash,
370        );
371
372        u_i.x = vec![
373            U_i.hash(
374                &sponge,
375                C1::ScalarField::zero(), // i
376                &self.z_0,
377                &state,
378            ),
379            cf_U_i.hash_cyclefold(&sponge),
380        ];
381        let us = vec![u_i.clone(); NU - 1];
382
383        // compute u_{i+1}.x
384        let U_i1 = LCCCS::dummy(&self.ccs);
385
386        let augmented_f_circuit = AugmentedFCircuit::<C1, C2, FC, MU, NU> {
387            poseidon_config: self.poseidon_config.clone(),
388            ccs: self.ccs.clone(),
389            pp_hash: Some(self.pp_hash),
390            i: Some(C1::ScalarField::zero()),
391            i_usize: Some(0),
392            z_0: Some(self.z_0.clone()),
393            z_i: Some(state.clone()),
394            external_inputs: Some(external_inputs),
395            U_i: Some(U_i.clone()),
396            Us: None,
397            u_i_C: Some(u_i.C),
398            us: Some(us),
399            U_i1_C: Some(U_i1.C),
400            F: self.F.clone(),
401            nimfs_proof: None,
402
403            // cyclefold values
404            cf_u_i_cmW: None,
405            cf_U_i: None,
406            cf_cmT: None,
407        };
408
409        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
410        augmented_f_circuit.generate_constraints(cs.clone())?;
411        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
412
413        #[cfg(test)]
414        assert!(cs.is_satisfied()?);
415
416        let (r1cs_w_i1, r1cs_x_i1) = extract_w_x::<C1::ScalarField>(&cs); // includes 1 and public inputs
417
418        let r1cs_z = [
419            vec![C1::ScalarField::one()],
420            r1cs_x_i1.clone(),
421            r1cs_w_i1.clone(),
422        ]
423        .concat();
424        Ok(r1cs_z)
425    }
426}

428impl<C1, C2, FC, CS1, CS2, const MU: usize, const NU: usize, const H: bool>
429    FoldingScheme<C1, C2, FC> for HyperNova<C1, C2, FC, CS1, CS2, MU, NU, H>
430where
431    C1: Curve,
432    C2: Curve,
433    FC: FCircuit<C1::ScalarField>,
434    CS1: CommitmentScheme<C1, H>,
435    CS2: CommitmentScheme<C2, H>,
436    C1: Curve<BaseField = C2::ScalarField, ScalarField = C2::BaseField>,
437{
438    /// Reuse Nova's PreprocessorParam.
439    type PreprocessorParam = PreprocessorParam<C1, C2, FC, CS1, CS2, H>;
440    type ProverParam = ProverParams<C1, C2, CS1, CS2, H>;
441    type VerifierParam = VerifierParams<C1, C2, CS1, CS2, H>;
442    type RunningInstance = (LCCCS<C1>, Witness<C1::ScalarField>);
443    type IncomingInstance = (CCCS<C1>, Witness<C1::ScalarField>);
444    type MultiCommittedInstanceWithWitness =
445        (Vec<Self::RunningInstance>, Vec<Self::IncomingInstance>);
446    type CFInstance = (CycleFoldCommittedInstance<C2>, CycleFoldWitness<C2>);
447    type IVCProof = IVCProof<C1, C2>;
448
449    fn pp_deserialize_with_mode<R: std::io::prelude::Read>(
450        mut reader: R,
451        compress: ark_serialize::Compress,
452        validate: ark_serialize::Validate,
453        fc_params: FC::Params,
454    ) -> Result<Self::ProverParam, Error> {
455        let poseidon_config = poseidon_canonical_config::<C1::ScalarField>();
456
457        // generate the r1cs & cf_r1cs needed for the VerifierParams. In this way we avoid needing
458        // to serialize them, saving significant space in the VerifierParams serialized size.
459
460        // main circuit R1CS:
461        let f_circuit = FC::new(fc_params)?;
462        let augmented_F_circuit = AugmentedFCircuit::<C1, C2, FC, MU, NU>::empty(
463            &poseidon_config,
464            f_circuit.clone(),
465            None,
466        )?;
467        let ccs = augmented_F_circuit.ccs;
468
469        let cs_pp = CS1::ProverParams::deserialize_with_mode(&mut reader, compress, validate)?;
470        let cf_cs_pp = CS2::ProverParams::deserialize_with_mode(&mut reader, compress, validate)?;
471
472        Ok(ProverParams {
473            poseidon_config,
474            cs_pp,
475            cf_cs_pp,
476            ccs: Some(ccs),
477        })
478    }
479
480    fn vp_deserialize_with_mode<R: std::io::prelude::Read>(
481        mut reader: R,
482        compress: ark_serialize::Compress,
483        validate: ark_serialize::Validate,
484        fc_params: FC::Params,
485    ) -> Result<Self::VerifierParam, Error> {
486        let poseidon_config = poseidon_canonical_config::<C1::ScalarField>();
487
488        // generate the r1cs & cf_r1cs needed for the VerifierParams. In this way we avoid needing
489        // to serialize them, saving significant space in the VerifierParams serialized size.
490
491        // main circuit R1CS:
492        let f_circuit = FC::new(fc_params)?;
493        let augmented_F_circuit = AugmentedFCircuit::<C1, C2, FC, MU, NU>::empty(
494            &poseidon_config,
495            f_circuit.clone(),
496            None,
497        )?;
498        let ccs = augmented_F_circuit.ccs;
499
500        // CycleFold circuit R1CS
501        let cf_circuit = CycleFoldCircuit::<_, HyperNovaCycleFoldConfig<C1, MU, NU>>::default();
502        let cf_r1cs = get_r1cs_from_cs::<C2::ScalarField>(cf_circuit)?;
503
504        let cs_vp = CS1::VerifierParams::deserialize_with_mode(&mut reader, compress, validate)?;
505        let cf_cs_vp = CS2::VerifierParams::deserialize_with_mode(&mut reader, compress, validate)?;
506
507        Ok(VerifierParams {
508            poseidon_config,
509            ccs,
510            cf_r1cs,
511            cs_vp,
512            cf_cs_vp,
513        })
514    }
515
516    fn preprocess(
517        mut rng: impl RngCore,
518        prep_param: &Self::PreprocessorParam,
519    ) -> Result<(Self::ProverParam, Self::VerifierParam), Error> {
520        if MU < 1 || NU < 1 {
521            return Err(Error::CantBeZero("mu,nu".to_string()));
522        }
523
524        let augmented_f_circuit = AugmentedFCircuit::<C1, C2, FC, MU, NU>::empty(
525            &prep_param.poseidon_config,
526            prep_param.F.clone(),
527            None,
528        )?;
529        let ccs = augmented_f_circuit.ccs.clone();
530
531        let cf_circuit = CycleFoldCircuit::<_, HyperNovaCycleFoldConfig<C1, MU, NU>>::default();
532        let cf_r1cs = get_r1cs_from_cs::<C2::ScalarField>(cf_circuit)?;
533
534        // if cs params exist, use them, if not, generate new ones
535        let (cs_pp, cs_vp) = match (&prep_param.cs_pp, &prep_param.cs_vp) {
536            (Some(cs_pp), Some(cs_vp)) => (cs_pp.clone(), cs_vp.clone()),
537            // `CS1` is for committing to HyperNova's witness vector `w`, so we
538            // set `len` to the number of witnesses in `r1cs`.
539            _ => CS1::setup(&mut rng, ccs.n_witnesses())?,
540        };
541        let (cf_cs_pp, cf_cs_vp) = match (&prep_param.cf_cs_pp, &prep_param.cf_cs_vp) {
542            (Some(cf_cs_pp), Some(cf_cs_vp)) => (cf_cs_pp.clone(), cf_cs_vp.clone()),
543            _ => CS2::setup(
544                &mut rng,
545                // `CS2` is for committing to CycleFold's witness vector `w` and
546                // error term `e`, where the length of `e` is the number of
547                // constraints, so we set `len` to the maximum of `e` and `w`'s
548                // lengths.
549                max(cf_r1cs.n_constraints(), cf_r1cs.n_witnesses()),
550            )?,
551        };
552
553        let pp = ProverParams::<C1, C2, CS1, CS2, H> {
554            poseidon_config: prep_param.poseidon_config.clone(),
555            cs_pp,
556            cf_cs_pp,
557            ccs: Some(ccs.clone()),
558        };
559        let vp = VerifierParams::<C1, C2, CS1, CS2, H> {
560            poseidon_config: prep_param.poseidon_config.clone(),
561            ccs,
562            cf_r1cs,
563            cs_vp: cs_vp.clone(),
564            cf_cs_vp: cf_cs_vp.clone(),
565        };
566        Ok((pp, vp))
567    }

569    /// Initializes the HyperNova+CycleFold's IVC for the given parameters and initial state `z_0`.
570    fn init(
571        params: &(Self::ProverParam, Self::VerifierParam),
572        F: FC,
573        z_0: Vec<C1::ScalarField>,
574    ) -> Result<Self, Error> {
575        let (pp, vp) = params;
576        if MU < 1 || NU < 1 {
577            return Err(Error::CantBeZero("mu,nu".to_string()));
578        }
579
580        // compute the public params hash
581        let pp_hash = vp.pp_hash()?;
582
583        // `sponge` is for digest computation.
584        let sponge =
585            PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(&pp.poseidon_config, pp_hash);
586
587        // prepare the HyperNova's AugmentedFCircuit and CycleFold's circuits and obtain its CCS
588        // and R1CS respectively
589        let augmented_f_circuit = AugmentedFCircuit::<C1, C2, FC, MU, NU>::empty(
590            &pp.poseidon_config,
591            F.clone(),
592            pp.ccs.clone(),
593        )?;
594        let ccs = augmented_f_circuit.ccs.clone();
595
596        let cf_circuit = CycleFoldCircuit::<_, HyperNovaCycleFoldConfig<C1, MU, NU>>::default();
597        let cf_r1cs = get_r1cs_from_cs::<C2::ScalarField>(cf_circuit)?;
598
599        // setup the dummy instances
600        let W_dummy = Witness::<C1::ScalarField>::dummy(&ccs);
601        let U_dummy = LCCCS::<C1>::dummy(&ccs);
602        let w_dummy = W_dummy.clone();
603        let mut u_dummy = CCCS::<C1>::dummy(&ccs);
604        let (cf_W_dummy, cf_U_dummy): (CycleFoldWitness<C2>, CycleFoldCommittedInstance<C2>) =
605            cf_r1cs.dummy_witness_instance();
606        u_dummy.x = vec![
607            U_dummy.hash(&sponge, C1::ScalarField::zero(), &z_0, &z_0),
608            cf_U_dummy.hash_cyclefold(&sponge),
609        ];
610
611        // W_dummy=W_0 is a 'dummy witness', all zeroes, but with the size corresponding to the
612        // R1CS that we're working with.
613        Ok(Self {
614            ccs,
615            cf_r1cs,
616            poseidon_config: pp.poseidon_config.clone(),
617            cs_pp: pp.cs_pp.clone(),
618            cf_cs_pp: pp.cf_cs_pp.clone(),
619            F,
620            pp_hash,
621            i: C1::ScalarField::zero(),
622            z_0: z_0.clone(),
623            z_i: z_0,
624            W_i: W_dummy,
625            U_i: U_dummy,
626            w_i: w_dummy,
627            u_i: u_dummy,
628            // cyclefold running instance
629            cf_W_i: cf_W_dummy,
630            cf_U_i: cf_U_dummy,
631        })
632    }

634    /// Implements IVC.P of HyperNova+CycleFold
635    fn prove_step(
636        &mut self,
637        mut rng: impl RngCore,
638        external_inputs: FC::ExternalInputs,
639        other_instances: Option<Self::MultiCommittedInstanceWithWitness>,
640    ) -> Result<(), Error> {
641        // ensure that commitments are blinding if user has specified so.
642
643        if H {
644            let blinding_commitments = if self.i == C1::ScalarField::zero() {
645                vec![self.w_i.r_w]
646            } else {
647                vec![self.w_i.r_w, self.W_i.r_w]
648            };
649            if blinding_commitments.contains(&C1::ScalarField::zero()) {
650                return Err(Error::IncorrectBlinding(
651                    H,
652                    format!("{blinding_commitments:?}"),
653                ));
654            }
655        }
656
657        let (Us, Ws, us, ws) = if MU > 1 || NU > 1 {
658            let other_instances = other_instances.ok_or(Error::MissingOtherInstances(MU, NU))?;
659
660            #[allow(clippy::type_complexity)]
661            let (lcccs, cccs): (
662                Vec<(LCCCS<C1>, Witness<C1::ScalarField>)>,
663                Vec<(CCCS<C1>, Witness<C1::ScalarField>)>,
664            ) = other_instances;
665
666            // recall, mu & nu is the number of all the LCCCS & CCCS respectively, including the
667            // running and incoming instances that are not part of the 'other_instances', hence the +1
668            // in the couple of following checks.
669            if lcccs.len() + 1 != MU {
670                return Err(Error::NotSameLength(
671                    "other_instances.lcccs.len()".to_string(),
672                    lcccs.len(),
673                    "hypernova.mu".to_string(),
674                    MU,
675                ));
676            }
677            if cccs.len() + 1 != NU {
678                return Err(Error::NotSameLength(
679                    "other_instances.cccs.len()".to_string(),
680                    cccs.len(),
681                    "hypernova.nu".to_string(),
682                    NU,
683                ));
684            }
685
686            let (Us, Ws): (Vec<LCCCS<C1>>, Vec<Witness<C1::ScalarField>>) =
687                lcccs.into_iter().unzip();
688            let (us, ws): (Vec<CCCS<C1>>, Vec<Witness<C1::ScalarField>>) = cccs.into_iter().unzip();
689            (Us, Ws, us, ws)
690        } else {
691            (vec![], vec![], vec![], vec![])
692        };
693
694        let augmented_f_circuit: AugmentedFCircuit<C1, C2, FC, MU, NU>;
695
696        if self.z_i.len() != self.F.state_len() {
697            return Err(Error::NotSameLength(
698                "z_i.len()".to_string(),
699                self.z_i.len(),
700                "F.state_len()".to_string(),
701                self.F.state_len(),
702            ));
703        }
704
705        if self.i > C1::ScalarField::from_le_bytes_mod_order(&usize::MAX.to_le_bytes()) {
706            return Err(Error::MaxStep);
707        }
708
709        let i_usize;
710
711        #[cfg(target_pointer_width = "64")]
712        {
713            let mut i_bytes: [u8; 8] = [0; 8];
714            i_bytes.copy_from_slice(&self.i.into_bigint().to_bytes_le()[..8]);
715            i_usize = usize::from_le_bytes(i_bytes);
716        }
717
718        #[cfg(target_pointer_width = "32")]
719        {
720            let mut i_bytes: [u8; 4] = [0; 4];
721            i_bytes.copy_from_slice(&self.i.into_bigint().to_bytes_le()[..4]);
722            i_usize = usize::from_le_bytes(i_bytes);
723        }
724
725        let (U_i1, mut W_i1);
726
727        if self.i == C1::ScalarField::zero() {
728            W_i1 = Witness::<C1::ScalarField>::dummy(&self.ccs);
729            W_i1.r_w = self.W_i.r_w;
730            U_i1 = LCCCS::dummy(&self.ccs);
731
732            augmented_f_circuit = AugmentedFCircuit::<C1, C2, FC, MU, NU> {
733                poseidon_config: self.poseidon_config.clone(),
734                ccs: self.ccs.clone(),
735                pp_hash: Some(self.pp_hash),
736                i: Some(C1::ScalarField::zero()),
737                i_usize: Some(0),
738                z_0: Some(self.z_0.clone()),
739                z_i: Some(self.z_i.clone()),
740                external_inputs: Some(external_inputs.clone()),
741                U_i: Some(self.U_i.clone()),
742                Us: Some(Us),
743                u_i_C: Some(self.u_i.C),
744                us: Some(us),
745                U_i1_C: Some(U_i1.C),
746                F: self.F.clone(),
747                nimfs_proof: None,
748
749                // cyclefold values
750                cf_u_i_cmW: None,
751                cf_U_i: None,
752                cf_cmT: None,
753            };
754        } else {
755            let mut transcript_p: PoseidonSponge<C1::ScalarField> =
756                PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(
757                    &self.poseidon_config,
758                    self.pp_hash,
759                );
760
761            let (all_Us, all_us, all_Ws, all_ws) = (
762                [&[self.U_i.clone()][..], &Us].concat(),
763                [&[self.u_i.clone()][..], &us].concat(),
764                [vec![self.W_i.clone()], Ws].concat(),
765                [vec![self.w_i.clone()], ws].concat(),
766            );
767
768            let (rho, nimfs_proof);
769            (nimfs_proof, U_i1, W_i1, rho) = NIMFS::<C1, PoseidonSponge<C1::ScalarField>>::prove(
770                &mut transcript_p,
771                &self.ccs,
772                &all_Us,
773                &all_us,
774                &all_Ws,
775                &all_ws,
776            )?;
777
778            // sanity check: check the folded instance relation
779            #[cfg(test)]
780            self.ccs.check_relation(&W_i1, &U_i1)?;
781
782            // CycleFold part:
783            let (cf_w_i, cf_u_i) = HyperNovaCycleFoldConfig::<C1, MU, NU> {
784                r: rho,
785                points: [
786                    all_Us.iter().map(|Us_i| Us_i.C).collect::<Vec<_>>(),
787                    all_us.iter().map(|us_i| us_i.C).collect::<Vec<_>>(),
788                ]
789                .concat(),
790            }
791            .build_circuit()
792            .generate_incoming_instance_witness::<_, CS2, H>(&self.cf_cs_pp, &mut rng)?;
793
794            let (cf_W_i1, cf_U_i1, cf_cmTs) = CycleFoldAugmentationGadget::fold_native::<_, CS2, H>(
795                &mut transcript_p,
796                &self.cf_r1cs,
797                &self.cf_cs_pp,
798                self.cf_W_i.clone(),
799                self.cf_U_i.clone(),
800                vec![cf_w_i],
801                vec![cf_u_i.clone()],
802            )?;
803
804            augmented_f_circuit = AugmentedFCircuit::<C1, C2, FC, MU, NU> {
805                poseidon_config: self.poseidon_config.clone(),
806                ccs: self.ccs.clone(),
807                pp_hash: Some(self.pp_hash),
808                i: Some(self.i),
809                i_usize: Some(i_usize),
810                z_0: Some(self.z_0.clone()),
811                z_i: Some(self.z_i.clone()),
812                external_inputs: Some(external_inputs),
813                U_i: Some(self.U_i.clone()),
814                Us: Some(Us),
815                u_i_C: Some(self.u_i.C),
816                us: Some(us),
817                U_i1_C: Some(U_i1.C),
818                F: self.F.clone(),
819                nimfs_proof: Some(nimfs_proof),
820
821                // cyclefold values
822                cf_u_i_cmW: Some(cf_u_i.cmW),
823                cf_U_i: Some(self.cf_U_i.clone()),
824                cf_cmT: Some(cf_cmTs[0]),
825            };
826
827            // assign the next round instances
828            self.cf_W_i = cf_W_i1;
829            self.cf_U_i = cf_U_i1;
830        }
831
832        let cs = ConstraintSystem::<C1::ScalarField>::new_ref();
833        let z_i1 = augmented_f_circuit
834            .compute_next_state(cs.clone())?
835            .value()?;
836        let cs = cs.into_inner().ok_or(Error::NoInnerConstraintSystem)?;
837
838        #[cfg(test)]
839        assert!(cs.is_satisfied()?);
840
841        let (r1cs_w_i1, r1cs_x_i1) = extract_w_x::<C1::ScalarField>(&cs); // includes 1 and public inputs
842
843        let r1cs_z = [
844            vec![C1::ScalarField::one()],
845            r1cs_x_i1.clone(),
846            r1cs_w_i1.clone(),
847        ]
848        .concat();
849        // compute committed instances, w_{i+1}, u_{i+1}, which will be used as w_i, u_i, so we
850        // assign them directly to w_i, u_i.
851        let (u_i, w_i) = self
852            .ccs
853            .to_cccs::<_, C1, CS1, H>(&mut rng, &self.cs_pp, &r1cs_z)?;
854        self.u_i = u_i.clone();
855        self.w_i = w_i.clone();
856
857        // set values for next iteration
858        self.i += C1::ScalarField::one();
859        // assign z_{i+1} into z_i
860        self.z_i = z_i1;
861        self.U_i = U_i1.clone();
862        self.W_i = W_i1.clone();
863
864        #[cfg(test)]
865        {
866            // check the new LCCCS instance relation
867            self.ccs.check_relation(&self.W_i, &self.U_i)?;
868            // check the new CCCS instance relation
869            self.ccs.check_relation(&self.w_i, &self.u_i)?;
870        }
871
872        Ok(())
873    }

942    /// Implements IVC.V of Hyp.clone()erNova+CycleFold. Notice that this method does not include the
943    /// commitments verification, which is done in the Decider.
944    fn verify(vp: Self::VerifierParam, ivc_proof: Self::IVCProof) -> Result<(), Error> {
945        let Self::IVCProof {
946            i: num_steps,
947            z_0,
948            z_i,
949            W_i,
950            U_i,
951            w_i,
952            u_i,
953            cf_W_i,
954            cf_U_i,
955        } = ivc_proof;
956
957        if num_steps == C1::ScalarField::zero() {
958            if z_0 != z_i {
959                return Err(Error::IVCVerificationFail);
960            }
961            return Ok(());
962        }
963        // `sponge` is for digest computation.
964        let sponge =
965            PoseidonSponge::<C1::ScalarField>::new_with_pp_hash(&vp.poseidon_config, vp.pp_hash()?);
966
967        if u_i.x.len() != 2 || U_i.x.len() != 2 {
968            return Err(Error::IVCVerificationFail);
969        }
970
971        // check that u_i's output points to the running instance
972        // u_i.X[0] == H(i, z_0, z_i, U_i)
973        let expected_u_i_x = U_i.hash(&sponge, num_steps, &z_0, &z_i);
974        if expected_u_i_x != u_i.x[0] {
975            return Err(Error::IVCVerificationFail);
976        }
977        // u_i.X[1] == H(cf_U_i)
978        let expected_cf_u_i_x = cf_U_i.hash_cyclefold(&sponge);
979        if expected_cf_u_i_x != u_i.x[1] {
980            return Err(Error::IVCVerificationFail);
981        }
982
983        // check LCCCS satisfiability
984        vp.ccs.check_relation(&W_i, &U_i)?;
985        // check CCCS satisfiability
986        vp.ccs.check_relation(&w_i, &u_i)?;
987
988        // check CycleFold's RelaxedR1CS satisfiability
989        vp.cf_r1cs.check_relation(&cf_W_i, &cf_U_i)?;
990
991        Ok(())
992    }

1031    #[allow(clippy::type_complexity)]
1032    // test_ivc allowing to choose the CommitmentSchemes
1033    pub fn test_ivc_opt<
1034        CS1: CommitmentScheme<Projective, H>,
1035        CS2: CommitmentScheme<Projective2, H>,
1036        const H: bool,
1037    >(
1038        poseidon_config: PoseidonConfig<Fr>,
1039        F_circuit: CubicFCircuit<Fr>,
1040    ) -> Result<(), Error> {
1041        let mut rng = ark_std::test_rng();
1042
1043        const MU: usize = 2;
1044        const NU: usize = 3;
1045
1046        type HN<CS1, CS2, const H: bool> =
1047            HyperNova<Projective, Projective2, CubicFCircuit<Fr>, CS1, CS2, MU, NU, H>;
1048
1049        let prep_param =
1050            PreprocessorParam::<Projective, Projective2, CubicFCircuit<Fr>, CS1, CS2, H>::new(
1051                poseidon_config.clone(),
1052                F_circuit,
1053            );
1054        let hypernova_params = HN::preprocess(&mut rng, &prep_param)?;
1055
1056        let z_0 = vec![Fr::from(3_u32)];
1057        let mut hypernova = HN::init(&hypernova_params, F_circuit, z_0.clone())?;
1058
1059        let (w_i_blinding, W_i_blinding) = if H {
1060            (Fr::rand(&mut rng), Fr::rand(&mut rng))
1061        } else {
1062            (Fr::zero(), Fr::zero())
1063        };
1064        hypernova.w_i.r_w = w_i_blinding;
1065        hypernova.W_i.r_w = W_i_blinding;
1066
1067        let num_steps: usize = 3;
1068        for _ in 0..num_steps {
1069            // prepare some new instances to fold in the multifolding step
1070            let mut lcccs = vec![];
1071            for j in 0..MU - 1 {
1072                let instance_state = vec![Fr::from(j as u32 + 85_u32)];
1073                let (U, W) = hypernova.new_running_instance(&mut rng, instance_state, ())?;
1074                lcccs.push((U, W));
1075            }
1076            let mut cccs = vec![];
1077            for j in 0..NU - 1 {
1078                let instance_state = vec![Fr::from(j as u32 + 15_u32)];
1079                let (u, w) = hypernova.new_incoming_instance(&mut rng, instance_state, ())?;
1080                cccs.push((u, w));
1081            }
1082
1083            hypernova.prove_step(&mut rng, (), Some((lcccs, cccs)))?;
1084        }
1085        assert_eq!(Fr::from(num_steps as u32), hypernova.i);
1086
1087        let ivc_proof = hypernova.ivc_proof();
1088        HN::verify(
1089            hypernova_params.1.clone(), // verifier_params
1090            ivc_proof,
1091        )?;
1092        Ok(())
1093    }
