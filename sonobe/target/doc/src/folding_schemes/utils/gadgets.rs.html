13/// `EquivalenceGadget` enforces that two in-circuit variables are equivalent,
14/// where the equivalence relation is parameterized by `M`:
15/// - For `FpVar`, it is simply an equality relation, and `M` is unused.
16/// - For `NonNativeUintVar`, we consider equivalence as a congruence relation,
17///   in terms of modular arithmetic, so `M` specifies the modulus.
18pub trait EquivalenceGadget<M> {
19    fn enforce_equivalent(&self, other: &Self) -> Result<(), SynthesisError>;
20}
21impl<M, F: PrimeField> EquivalenceGadget<M> for FpVar<F> {
22    fn enforce_equivalent(&self, other: &Self) -> Result<(), SynthesisError> {
23        self.enforce_equal(other)
24    }
25}
26impl<M, T: EquivalenceGadget<M>> EquivalenceGadget<M> for [T] {
27    fn enforce_equivalent(&self, other: &Self) -> Result<(), SynthesisError> {
28        self.iter()
29            .zip(other)
30            .try_for_each(|(a, b)| a.enforce_equivalent(b))
31    }
32}

38pub trait VectorGadget<FV> {
39    fn add(&self, other: &Self) -> Result<Vec<FV>, SynthesisError>;
40
41    fn mul_scalar(&self, other: &FV) -> Result<Vec<FV>, SynthesisError>;
42
43    fn hadamard(&self, other: &Self) -> Result<Vec<FV>, SynthesisError>;
44}
45
46impl<F: PrimeField> VectorGadget<FpVar<F>> for [FpVar<F>] {
47    fn add(&self, other: &Self) -> Result<Vec<FpVar<F>>, SynthesisError> {
48        if self.len() != other.len() {
49            return Err(SynthesisError::Unsatisfiable);
50        }
51        Ok(self.iter().zip(other.iter()).map(|(a, b)| a + b).collect())
52    }
53
54    fn mul_scalar(&self, c: &FpVar<F>) -> Result<Vec<FpVar<F>>, SynthesisError> {
55        Ok(self.iter().map(|a| a * c).collect())
56    }
57
58    fn hadamard(&self, other: &Self) -> Result<Vec<FpVar<F>>, SynthesisError> {
59        if self.len() != other.len() {
60            return Err(SynthesisError::Unsatisfiable);
61        }
62        Ok(self.iter().zip(other.iter()).map(|(a, b)| a * b).collect())
63    }
64}

67pub struct SparseMatrixVar<FV> {
68    pub n_rows: usize,
69    pub n_cols: usize,
70    // same format as the native SparseMatrix (which follows ark_relations::gr1cs::Matrix format
71    pub coeffs: Vec<Vec<(FV, usize)>>,
72}
73
74impl<F: PrimeField, CF: PrimeField, FV: AllocVar<F, CF>> AllocVar<SparseMatrix<F>, CF>
75    for SparseMatrixVar<FV>
76{
77    fn new_variable<T: Borrow<SparseMatrix<F>>>(
78        cs: impl Into<Namespace<CF>>,
79        f: impl FnOnce() -> Result<T, SynthesisError>,
80        mode: AllocationMode,
81    ) -> Result<Self, SynthesisError> {
82        f().and_then(|val| {
83            let cs = cs.into();
84
85            let mut coeffs: Vec<Vec<(FV, usize)>> = Vec::new();
86            for row in val.borrow().coeffs.iter() {
87                let mut rowVar: Vec<(FV, usize)> = Vec::new();
88                for &(value, col_i) in row.iter() {
89                    let coeffVar = FV::new_variable(cs.clone(), || Ok(value), mode)?;
90                    rowVar.push((coeffVar, col_i));
91                }
92                coeffs.push(rowVar);
93            }
94
95            Ok(Self {
96                n_rows: val.borrow().n_rows,
97                n_cols: val.borrow().n_cols,
98                coeffs,
99            })
100        })
101    }
102}

104impl<F: PrimeField> MatrixGadget<FpVar<F>> for SparseMatrixVar<FpVar<F>> {
105    fn mul_vector(&self, v: &[FpVar<F>]) -> Result<Vec<FpVar<F>>, SynthesisError> {
106        Ok(self
107            .coeffs
108            .iter()
109            .map(|row| {
110                let products = row
111                    .iter()
112                    .map(|(value, col_i)| value * &v[*col_i])
113                    .collect::<Vec<_>>();
114                if products.is_constant() {
115                    FpVar::constant(products.value().unwrap_or_default().into_iter().sum())
116                } else {
117                    products.iter().sum()
118                }
119            })
120            .collect())
121    }
122}

124/// Interprets the given vector v as the evaluations of a dense multilinear extension of n_vars,
125/// and evaluates it at the given point. This method mimics the behavior of
126/// `utils/mle.rs#dense_vec_to_dense_mle` + `DenseMultilinearExtension::evaluate` but in R1CS
127/// constraints, since dense multilinear extensions are not supported in ark_r1cs_std.
128pub fn eval_mle<F: PrimeField>(
129    // n_vars indicates the number of variables in the MLE
130    n_vars: usize,
131    // v is the vector of the evaluations of the dense multilinear extension (MLE)
132    v: Vec<FpVar<F>>,
133    // point is the point at which we want to evaluate the MLE
134    point: Vec<FpVar<F>>,
135) -> FpVar<F> {
136    // pad to 2^n_vars
137    let mut poly = v;
138    poly.resize(1 << n_vars, FpVar::zero());
139
140    for i in 1..n_vars + 1 {
141        let r = point[i - 1].clone();
142        for b in 0..(1 << (n_vars - 1)) {
143            let left = poly[b << 1].clone();
144            let right = poly[(b << 1) + 1].clone();
145            poly[b] = left.clone() + r.clone() * (right - left);
146        }
147    }
148    poly[0].clone()
149}
