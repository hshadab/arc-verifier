32/// A generic sum-check trait over a curve group
33pub trait SumCheck<F: PrimeField> {
34    type VirtualPolynomial;
35    type VPAuxInfo;
36    type MultilinearExtension;
37
38    type SumCheckProof: Clone + Debug + Default + PartialEq;
39    type SumCheckSubClaim: Clone + Debug + Default + PartialEq;
40
41    /// Extract sum from the proof
42    fn extract_sum(proof: &Self::SumCheckProof) -> F;
43
44    /// Generate proof of the sum of polynomial over {0,1}^`num_vars`
45    ///
46    /// The polynomial is represented in the form of a VirtualPolynomial.
47    fn prove(
48        poly: &Self::VirtualPolynomial,
49        transcript: &mut impl Transcript<F>,
50    ) -> Result<Self::SumCheckProof, Error>;
51
52    /// Verify the claimed sum using the proof
53    fn verify(
54        sum: F,
55        proof: &Self::SumCheckProof,
56        aux_info: &Self::VPAuxInfo,
57        transcript: &mut impl Transcript<F>,
58    ) -> Result<Self::SumCheckSubClaim, Error>;
59}

61/// Trait for sum check protocol prover side APIs.
62pub trait SumCheckProver<F: PrimeField>: Sized {
63    type VirtualPolynomial;
64    type ProverMessage;
65
66    /// Initialize the prover state to argue for the sum of the input polynomial
67    /// over {0,1}^`num_vars`.
68    fn prover_init(polynomial: &Self::VirtualPolynomial) -> Result<Self, Error>;
69
70    /// Receive message from verifier, generate prover message, and proceed to
71    /// next round.
72    ///
73    /// Main algorithm used is from section 3.2 of [XZZPS19](https://eprint.iacr.org/2019/317.pdf#subsection.3.2).
74    fn prove_round_and_update_state(
75        &mut self,
76        challenge: &Option<F>,
77    ) -> Result<Self::ProverMessage, Error>;
78}

80/// Trait for sum check protocol verifier side APIs.
81pub trait SumCheckVerifier<F: PrimeField> {
82    type VPAuxInfo;
83    type ProverMessage;
84    type Challenge;
85    type SumCheckSubClaim;
86
87    /// Initialize the verifier's state.
88    fn verifier_init(index_info: &Self::VPAuxInfo) -> Self;
89
90    /// Run verifier for the current round, given a prover message.
91    ///
92    /// Note that `verify_round_and_update_state` only samples and stores
93    /// challenges; and update the verifier's state accordingly. The actual
94    /// verifications are deferred (in batch) to `check_and_generate_subclaim`
95    /// at the last step.
96    fn verify_round_and_update_state(
97        &mut self,
98        prover_msg: &Self::ProverMessage,
99        transcript: &mut impl Transcript<F>,
100    ) -> Result<Self::Challenge, Error>;
101
102    /// This function verifies the deferred checks in the interactive version of
103    /// the protocol; and generate the subclaim. Returns an error if the
104    /// proof failed to verify.
105    ///
106    /// If the asserted sum is correct, then the multilinear polynomial
107    /// evaluated at `subclaim.point` will be `subclaim.expected_evaluation`.
108    /// Otherwise, it is highly unlikely that those two will be equal.
109    /// Larger field size guarantees smaller soundness error.
110    fn check_and_generate_subclaim(
111        &self,
112        asserted_sum: &F,
113    ) -> Result<Self::SumCheckSubClaim, Error>;
114}

116/// A SumCheckSubClaim is a claim generated by the verifier at the end of
117/// verification when it is convinced.
118#[derive(Clone, Debug, Default, PartialEq, Eq)]
119pub struct SumCheckSubClaim<F: PrimeField> {
120    /// the multi-dimensional point that this multilinear extension is evaluated
121    /// to
122    pub point: Vec<F>,
123    /// the expected evaluation
124    pub expected_evaluation: F,
125}

127#[derive(Clone, Debug, Default, Copy, PartialEq, Eq)]
128pub struct IOPSumCheck<F: PrimeField, T: Transcript<F>> {
129    #[doc(hidden)]
130    phantom: PhantomData<F>,
131    #[doc(hidden)]
132    phantom2: PhantomData<T>,
133}
134
135impl<F: PrimeField + Absorb, T: Transcript<F>> SumCheck<F> for IOPSumCheck<F, T> {
136    type SumCheckProof = IOPProof<F>;
137    type VirtualPolynomial = VirtualPolynomial<F>;
138    type VPAuxInfo = VPAuxInfo<F>;
139    type MultilinearExtension = Arc<DenseMultilinearExtension<F>>;
140    type SumCheckSubClaim = SumCheckSubClaim<F>;
141
142    fn extract_sum(proof: &Self::SumCheckProof) -> F {
143        let start = start_timer!(|| "extract sum");
144        let poly = DensePolynomial::from_coefficients_vec(proof.proofs[0].coeffs.clone());
145        let res = poly.evaluate(&F::ONE) + poly.evaluate(&F::ZERO);
146        end_timer!(start);
147        res
148    }
149
150    fn prove(
151        poly: &VirtualPolynomial<F>,
152        transcript: &mut impl Transcript<F>,
153    ) -> Result<IOPProof<F>, Error> {
154        transcript.absorb(&F::from(poly.aux_info.num_variables as u64));
155        transcript.absorb(&F::from(poly.aux_info.max_degree as u64));
156        let mut prover_state: IOPProverState<F> = IOPProverState::prover_init(poly)?;
157        let mut challenge: Option<F> = None;
158        let mut prover_msgs: Vec<IOPProverMessage<F>> =
159            Vec::with_capacity(poly.aux_info.num_variables);
160        for _ in 0..poly.aux_info.num_variables {
161            let prover_msg: IOPProverMessage<F> =
162                IOPProverState::prove_round_and_update_state(&mut prover_state, &challenge)?;
163            transcript.absorb(&prover_msg.coeffs);
164            prover_msgs.push(prover_msg);
165            challenge = Some(transcript.get_challenge());
166        }
167        if let Some(p) = challenge {
168            prover_state.challenges.push(p)
169        };
170        Ok(IOPProof {
171            point: prover_state.challenges,
172            proofs: prover_msgs,
173        })
174    }
175
176    fn verify(
177        claimed_sum: F,
178        proof: &IOPProof<F>,
179        aux_info: &VPAuxInfo<F>,
180        transcript: &mut impl Transcript<F>,
181    ) -> Result<SumCheckSubClaim<F>, Error> {
182        transcript.absorb(&F::from(aux_info.num_variables as u64));
183        transcript.absorb(&F::from(aux_info.max_degree as u64));
184        let mut verifier_state = IOPVerifierState::verifier_init(aux_info);
185        for i in 0..aux_info.num_variables {
186            let prover_msg = proof.proofs.get(i).expect("proof is incomplete");
187            transcript.absorb(&prover_msg.coeffs);
188            IOPVerifierState::verify_round_and_update_state(
189                &mut verifier_state,
190                prover_msg,
191                transcript,
192            )?;
193        }
194
195        IOPVerifierState::check_and_generate_subclaim(&verifier_state, &claimed_sum)
196    }
197}

218    #[test]
219    pub fn sumcheck_poseidon() -> Result<(), Error> {
220        let n_vars = 5;
221
222        let mut rng = test_rng();
223        let poly_mle = DenseMultilinearExtension::rand(n_vars, &mut rng);
224        let virtual_poly = VirtualPolynomial::new_from_mle(&Arc::new(poly_mle), Fr::ONE);
225
226        let _ = sumcheck_poseidon_opt(virtual_poly)?;
227
228        // test with zero poly
229        let poly_mle = DenseMultilinearExtension::from_evaluations_vec(
230            n_vars,
231            vec![Fr::zero(); 2u32.pow(n_vars as u32) as usize],
232        );
233        let virtual_poly = VirtualPolynomial::new_from_mle(&Arc::new(poly_mle), Fr::ONE);
234        let _ = sumcheck_poseidon_opt(virtual_poly)?;
235        Ok(())
236    }
237
238    fn sumcheck_poseidon_opt(virtual_poly: VirtualPolynomial<Fr>) -> Result<(), Error> {
239        let poseidon_config = poseidon_canonical_config::<Fr>();
240
241        // sum-check prove
242        let mut transcript_p: PoseidonSponge<Fr> = PoseidonSponge::<Fr>::new(&poseidon_config);
243        let sum_check =
244            IOPSumCheck::<Fr, PoseidonSponge<Fr>>::prove(&virtual_poly, &mut transcript_p)?;
245
246        // sum-check verify
247        let claimed_sum = IOPSumCheck::<Fr, PoseidonSponge<Fr>>::extract_sum(&sum_check);
248        let mut transcript_v: PoseidonSponge<Fr> = PoseidonSponge::<Fr>::new(&poseidon_config);
249        let res_verify = IOPSumCheck::<Fr, PoseidonSponge<Fr>>::verify(
250            claimed_sum,
251            &sum_check,
252            &virtual_poly.aux_info,
253            &mut transcript_v,
254        );
255
256        assert!(res_verify.is_ok());
257        Ok(())
258    }
