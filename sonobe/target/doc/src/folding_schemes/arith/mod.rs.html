10/// [`Arith`] is a trait about constraint systems (R1CS, CCS, etc.), where we
11/// define methods for getting information about the constraint system.
12pub trait Arith: Clone {
13    /// Returns the degree of the constraint system
14    fn degree(&self) -> usize;
15
16    /// Returns the number of constraints in the constraint system
17    fn n_constraints(&self) -> usize;
18
19    /// Returns the number of variables in the constraint system
20    fn n_variables(&self) -> usize;
21
22    /// Returns the number of public inputs / public IO / instances / statements
23    /// in the constraint system
24    fn n_public_inputs(&self) -> usize;
25
26    /// Returns the number of witnesses / secret inputs in the constraint system
27    fn n_witnesses(&self) -> usize;
28
29    /// Returns a tuple containing (w, x) (witness and public inputs respectively)
30    fn split_z<F: PrimeField>(&self, z: &[F]) -> (Vec<F>, Vec<F>);
31}

33/// `ArithRelation` *treats a constraint system as a relation* between a witness
34/// of type `W` and a statement / public input / public IO / instance of type
35/// `U`, and in this trait, we define the necessary operations on the relation.
36///
37/// Note that the same constraint system may support different types of `W` and
38/// `U`, and the satisfiability check may vary.
39///
40/// For example, both plain R1CS and relaxed R1CS are represented by 3 matrices,
41/// but the types of `W` and `U` are different:
42/// - The plain R1CS has `W` and `U` as vectors of field elements.
43///   
44///   `W = w` and `U = x` satisfy R1CS if `Az ∘ Bz = Cz`, where `z = [1, x, w]`.
45///
46/// - In Nova, Relaxed R1CS has `W` as [`crate::folding::nova::Witness`],
47///   and `U` as [`crate::folding::nova::CommittedInstance`].
48///  
49///   `W = (w, e, ...)` and `U = (u, x, ...)` satisfy Relaxed R1CS if
50///   `Az ∘ Bz = uCz + e`, where `z = [u, x, w]`.
51///   (commitments in `U` are not checked here)
52///
53///   Also, `W` and `U` have non-native field elements as their components when
54///   used as CycleFold witness and instance.
55///
56/// - In ProtoGalaxy, Relaxed R1CS has `W` as [`crate::folding::protogalaxy::Witness`],
57///   and `U` as [`crate::folding::protogalaxy::CommittedInstance`].
58///    
59///   `W = (w, ...)` and `U = (x, e, β, ...)` satisfy Relaxed R1CS if
60///   `e = Σ pow_i(β) v_i`, where `v = Az ∘ Bz - Cz`, `z = [1, x, w]`.
61///   (commitments in `U` are not checked here)
62///
63/// This is also the case of CCS, where `W` and `U` may be vectors of field
64/// elements, [`crate::folding::hypernova::Witness`] and [`crate::folding::hypernova::lcccs::LCCCS`],
65/// or [`crate::folding::hypernova::Witness`] and [`crate::folding::hypernova::cccs::CCCS`].
66pub trait ArithRelation<W, U>: Arith {
67    type Evaluation;
68
69    /// Returns a dummy witness and instance
70    fn dummy_witness_instance<'a>(&'a self) -> (W, U)
71    where
72        W: Dummy<&'a Self>,
73        U: Dummy<&'a Self>,
74    {
75        (W::dummy(self), U::dummy(self))
76    }
77
78    /// Evaluates the constraint system `self` at witness `w` and instance `u`.
79    /// Returns the evaluation result.
80    ///
81    /// The evaluation result is usually a vector of field elements.
82    /// For instance:
83    /// - Evaluating the plain R1CS at `W = w` and `U = x` returns
84    ///   `Az ∘ Bz - Cz`, where `z = [1, x, w]`.
85    ///
86    /// - Evaluating the relaxed R1CS in Nova at `W = (w, e, ...)` and
87    ///   `U = (u, x, ...)` returns `Az ∘ Bz - uCz`, where `z = [u, x, w]`.
88    ///
89    /// - Evaluating the relaxed R1CS in ProtoGalaxy at `W = (w, ...)` and
90    ///   `U = (x, e, β, ...)` returns `Az ∘ Bz - Cz`, where `z = [1, x, w]`.
91    ///
92    /// However, we use `Self::Evaluation` to represent the evaluation result
93    /// for future extensibility.
94    fn eval_relation(&self, w: &W, u: &U) -> Result<Self::Evaluation, Error>;
95
96    /// Checks if the evaluation result is valid. The witness `w` and instance
97    /// `u` are also parameters, because the validity check may need information
98    /// contained in `w` and/or `u`.
99    ///
100    /// For instance:
101    /// - The evaluation `v` of plain R1CS at satisfying `W` and `U` should be
102    ///   an all-zero vector.
103    ///
104    /// - The evaluation `v` of relaxed R1CS in Nova at satisfying `W` and `U`
105    ///   should be equal to the error term `e` in the witness.
106    ///
107    /// - The evaluation `v` of relaxed R1CS in ProtoGalaxy at satisfying `W`
108    ///   and `U` should satisfy `e = Σ pow_i(β) v_i`, where `e` is the error
109    ///   term in the committed instance.
110    fn check_evaluation(w: &W, u: &U, v: Self::Evaluation) -> Result<(), Error>;
111
112    /// Checks if witness `w` and instance `u` satisfy the constraint system
113    /// `self` by first computing the evaluation result and then checking the
114    /// validity of the evaluation result.
115    ///
116    /// Used only for testing.
117    fn check_relation(&self, w: &W, u: &U) -> Result<(), Error> {
118        let e = self.eval_relation(w, u)?;
119        Self::check_evaluation(w, u, e)
120    }
121}

123/// `ArithSerializer` is for serializing constraint systems.
124///
125/// Currently we only support converting parameters to bytes, but in the future
126/// we may consider implementing methods for serializing the actual data (e.g.,
127/// R1CS matrices).
128pub trait ArithSerializer {
129    /// Returns the bytes that represent the parameters, that is, the matrices sizes, the amount of
130    /// public inputs, etc, without the matrices/polynomials values.
131    fn params_to_le_bytes(&self) -> Vec<u8>;
132}

134/// `ArithSampler` allows sampling random pairs of witness and instance that
135/// satisfy the constraint system `self`.
136///
137/// This is useful for constructing a zero-knowledge layer for a folding-based
138/// IVC.
139/// An example of such a layer can be found in Appendix D of the [HyperNova]
140/// paper.
141///
142/// Note that we use a separate trait for sampling, because this operation may
143/// not be supported by all witness-instance pairs.
144/// For instance, it is difficult (if not impossible) to do this for `w` and `x`
145/// in a plain R1CS.
146///
147/// [HyperNova]: https://eprint.iacr.org/2023/573.pdf
148pub trait ArithSampler<C: Curve, W, U>: ArithRelation<W, U> {
149    /// Samples a random witness and instance that satisfy the constraint system.
150    fn sample_witness_instance<CS: CommitmentScheme<C, true>>(
151        &self,
152        params: &CS::ProverParams,
153        rng: impl RngCore,
154    ) -> Result<(W, U), Error>;
155}

157/// `ArithRelationGadget` defines the in-circuit counterparts of operations
158/// specified in `ArithRelation` on constraint systems.
159pub trait ArithRelationGadget<WVar, UVar> {
160    type Evaluation;
161
162    /// Evaluates the constraint system `self` at witness `w` and instance `u`.
163    /// Returns the evaluation result.
164    fn eval_relation(&self, w: &WVar, u: &UVar) -> Result<Self::Evaluation, SynthesisError>;
165
166    /// Generates constraints for enforcing that witness `w` and instance `u`
167    /// satisfy the constraint system `self` by first computing the evaluation
168    /// result and then checking the validity of the evaluation result.
169    fn enforce_relation(&self, w: &WVar, u: &UVar) -> Result<(), SynthesisError> {
170        let e = self.eval_relation(w, u)?;
171        Self::enforce_evaluation(w, u, e)
172    }
173
174    /// Generates constraints for enforcing that the evaluation result is valid.
175    /// The witness `w` and instance `u` are also parameters, because the
176    /// validity check may need information contained in `w` and/or `u`.
177    fn enforce_evaluation(w: &WVar, u: &UVar, e: Self::Evaluation) -> Result<(), SynthesisError>;
178}
