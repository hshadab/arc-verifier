12/// CommitmentScheme defines the vector commitment scheme trait. Where `H` indicates if to use the
13/// commitment in hiding mode or not.
14pub trait CommitmentScheme<C: Curve, const H: bool = false>: Clone + Debug {
15    type ProverParams: Clone + Debug + CanonicalSerialize + CanonicalDeserialize;
16    type VerifierParams: Clone + Debug + CanonicalSerialize + CanonicalDeserialize;
17    type Proof: Clone + Debug + CanonicalSerialize + CanonicalDeserialize;
18    type ProverChallenge: Clone + Debug;
19    type Challenge: Clone + Debug;
20
21    fn is_hiding() -> bool;
22
23    fn setup(
24        rng: impl RngCore,
25        len: usize,
26    ) -> Result<(Self::ProverParams, Self::VerifierParams), Error>;
27
28    fn commit(
29        params: &Self::ProverParams,
30        v: &[C::ScalarField],
31        blind: &C::ScalarField,
32    ) -> Result<C, Error>;
33
34    fn prove(
35        params: &Self::ProverParams,
36        transcript: &mut impl Transcript<C::ScalarField>,
37        cm: &C,
38        v: &[C::ScalarField],
39        blind: &C::ScalarField,
40        rng: Option<&mut dyn RngCore>,
41    ) -> Result<Self::Proof, Error>;
42
43    /// same as `prove` but instead of providing a Transcript to use, providing the already
44    /// computed challenge
45    fn prove_with_challenge(
46        params: &Self::ProverParams,
47        challenge: Self::ProverChallenge,
48        v: &[C::ScalarField],
49        blind: &C::ScalarField,
50        rng: Option<&mut dyn RngCore>,
51    ) -> Result<Self::Proof, Error>;
52
53    fn verify(
54        params: &Self::VerifierParams,
55        transcript: &mut impl Transcript<C::ScalarField>,
56        cm: &C,
57        proof: &Self::Proof,
58    ) -> Result<(), Error>;
59
60    /// same as `verify` but instead of providing a Transcript to use, providing the already
61    /// computed challenge
62    fn verify_with_challenge(
63        params: &Self::VerifierParams,
64        challenge: Self::Challenge,
65        cm: &C,
66        proof: &Self::Proof,
67    ) -> Result<(), Error>;
68}

133    fn test_homomorphic_property_using_Commitment_trait_opt<C: Curve, CS: CommitmentScheme<C>>(
134        poseidon_config: &PoseidonConfig<C::ScalarField>,
135        prover_params: &CS::ProverParams,
136        verifier_params: &CS::VerifierParams,
137        r: C::ScalarField,
138        v_1: &[C::ScalarField],
139        v_2: &[C::ScalarField],
140    ) -> Result<(), Error> {
141        // compute the commitment of the two vectors using the given CommitmentScheme
142        let cm_1 = CS::commit(prover_params, v_1, &C::ScalarField::zero())?;
143        let cm_2 = CS::commit(prover_params, v_2, &C::ScalarField::zero())?;
144
145        // random linear combination of the commitments and their witnesses (vectors v_i)
146        let cm_3 = cm_1 + cm_2.mul(r);
147        let v_3: Vec<C::ScalarField> = v_1.iter().zip(v_2).map(|(a, b)| *a + (r * b)).collect();
148
149        // compute the proof of the cm_3
150        let transcript_p = &mut PoseidonSponge::<C::ScalarField>::new(poseidon_config);
151        let proof = CS::prove(
152            prover_params,
153            transcript_p,
154            &cm_3,
155            &v_3,
156            &C::ScalarField::zero(),
157            None,
158        )?;
159
160        // verify the opening proof
161        let transcript_v = &mut PoseidonSponge::<C::ScalarField>::new(poseidon_config);
162        CS::verify(verifier_params, transcript_v, &cm_3, &proof)?;
163        Ok(())
164    }
