1/// Adaptation of the prover methods and structs from arkworks/poly-commit's KZG10 implementation
2/// into the CommitmentScheme trait.
3///
4/// The motivation to do so, is that we want to be able to use KZG / Pedersen for committing to
5/// vectors indistinctly, and the arkworks KZG10 implementation contains all the methods under the
6/// same trait, which requires the Pairing trait, where the prover does not need access to the
7/// Pairing but only to G1.

29/// ProverKey defines a similar struct as in ark_poly_commit::kzg10::Powers, but instead of
30/// depending on the Pairing trait it depends on the SonobeCurve trait.
31#[derive(Debug, Clone, Default, Eq, PartialEq)]
32pub struct ProverKey<'a, C: Curve> {
33    /// Group elements of the form `Î²^i G`, for different values of `i`.
34    pub powers_of_g: Cow<'a, [C::Affine]>,
35}
36
37impl<'a, C: Curve> CanonicalSerialize for ProverKey<'a, C> {
38    fn serialize_with_mode<W: std::io::prelude::Write>(
39        &self,
40        mut writer: W,
41        compress: ark_serialize::Compress,
42    ) -> Result<(), ark_serialize::SerializationError> {
43        self.powers_of_g.serialize_with_mode(&mut writer, compress)
44    }
45
46    fn serialized_size(&self, compress: ark_serialize::Compress) -> usize {
47        self.powers_of_g.serialized_size(compress)
48    }
49}
50
51impl<'a, C: Curve> CanonicalDeserialize for ProverKey<'a, C> {
52    fn deserialize_with_mode<R: std::io::prelude::Read>(
53        reader: R,
54        compress: ark_serialize::Compress,
55        validate: ark_serialize::Validate,
56    ) -> Result<Self, ark_serialize::SerializationError> {
57        let powers_of_g_vec = Vec::deserialize_with_mode(reader, compress, validate)?;
58        Ok(ProverKey {
59            powers_of_g: ark_std::borrow::Cow::Owned(powers_of_g_vec),
60        })
61    }
62}
63
64impl<'a, C: Curve> Valid for ProverKey<'a, C> {
65    fn check(&self) -> Result<(), ark_serialize::SerializationError> {
66        match self.powers_of_g.clone() {
67            Cow::Borrowed(powers) => powers.to_vec().check(),
68            Cow::Owned(powers) => powers.check(),
69        }
70    }
71}

79/// KZG implements the CommitmentScheme trait for the KZG commitment scheme.
80#[derive(Debug, Clone, Default, Eq, PartialEq)]
81pub struct KZG<'a, E: Pairing, const H: bool = false> {
82    _a: PhantomData<&'a ()>,
83    _e: PhantomData<E>,
84}
85
86impl<'a, E: Pairing<G1: Curve>, const H: bool> CommitmentScheme<E::G1, H> for KZG<'a, E, H> {
87    type ProverParams = ProverKey<'a, E::G1>;
88    type VerifierParams = VerifierKey<E>;
89    type Proof = Proof<E::G1>;
90    type ProverChallenge = E::ScalarField;
91    type Challenge = E::ScalarField;
92
93    fn is_hiding() -> bool {
94        if H {
95            return true;
96        }
97        false
98    }
99
100    /// setup returns the tuple (ProverKey, VerifierKey). For real world deployments the setup must
101    /// be computed in the most trustless way possible, usually through an MPC ceremony.
102    fn setup(
103        mut rng: impl RngCore,
104        len: usize,
105    ) -> Result<(Self::ProverParams, Self::VerifierParams), Error> {
106        let len = len.next_power_of_two();
107        let universal_params =
108            KZG10::<E, DensePolynomial<E::ScalarField>>::setup(len, false, &mut rng)
109                .expect("Setup failed");
110        let powers_of_g = universal_params.powers_of_g[..=len].to_vec();
111        let powers = ProverKey::<E::G1> {
112            powers_of_g: ark_std::borrow::Cow::Owned(powers_of_g),
113        };
114        let vk = VerifierKey {
115            g: universal_params.powers_of_g[0],
116            gamma_g: universal_params.powers_of_gamma_g[&0],
117            h: universal_params.h,
118            beta_h: universal_params.beta_h,
119            prepared_h: universal_params.prepared_h.clone(),
120            prepared_beta_h: universal_params.prepared_beta_h.clone(),
121        };
122        Ok((powers, vk))
123    }
124
125    /// commit implements the CommitmentScheme commit interface, adapting the implementation from
126    /// https://github.com/arkworks-rs/poly-commit/tree/c724fa666e935bbba8db5a1421603bab542e15ab/poly-commit/src/kzg10/mod.rs#L178
127    /// with the main difference being the removal of the blinding factors and the no-dependency to
128    /// the Pairing trait.
129    fn commit(
130        params: &Self::ProverParams,
131        v: &[E::ScalarField],
132        _blind: &E::ScalarField,
133    ) -> Result<E::G1, Error> {
134        if !_blind.is_zero() || H {
135            return Err(Error::NotSupportedYet("hiding".to_string()));
136        }
137
138        let polynomial = poly_from_vec(v.to_vec())?;
139        check_degree_is_too_large(polynomial.degree(), params.powers_of_g.len())?;
140
141        let (num_leading_zeros, plain_coeffs) =
142            skip_first_zero_coeffs_and_convert_to_bigints(&polynomial);
143        let commitment = <E::G1 as VariableBaseMSM>::msm_bigint(
144            &params.powers_of_g[num_leading_zeros..],
145            &plain_coeffs,
146        );
147        Ok(commitment)
148    }
149
150    /// prove implements the CommitmentScheme prove interface, adapting the implementation from
151    /// https://github.com/arkworks-rs/poly-commit/tree/c724fa666e935bbba8db5a1421603bab542e15ab/poly-commit/src/kzg10/mod.rs#L307
152    /// with the main difference being the removal of the blinding factors and the no-dependency to
153    /// the Pairing trait.
154    fn prove(
155        params: &Self::ProverParams,
156        transcript: &mut impl Transcript<E::ScalarField>,
157        cm: &E::G1,
158        v: &[E::ScalarField],
159        _blind: &E::ScalarField,
160        _rng: Option<&mut dyn RngCore>,
161    ) -> Result<Self::Proof, Error> {
162        transcript.absorb_nonnative(cm);
163        let challenge = transcript.get_challenge();
164        Self::prove_with_challenge(params, challenge, v, _blind, _rng)
165    }
166
167    fn prove_with_challenge(
168        params: &Self::ProverParams,
169        challenge: Self::ProverChallenge,
170        v: &[E::ScalarField],
171        _blind: &E::ScalarField,
172        _rng: Option<&mut dyn RngCore>,
173    ) -> Result<Self::Proof, Error> {
174        if !_blind.is_zero() || H {
175            return Err(Error::NotSupportedYet("hiding".to_string()));
176        }
177
178        let polynomial = poly_from_vec(v.to_vec())?;
179        check_degree_is_too_large(polynomial.degree(), params.powers_of_g.len())?;
180
181        // Compute q(x) = (p(x) - p(z)) / (x-z). Observe that this quotient does not change with z
182        // because p(z) is the remainder term. We can therefore omit p(z) when computing the
183        // quotient.
184        let divisor = DensePolynomial::<E::ScalarField>::from_coefficients_vec(vec![
185            -challenge,
186            E::ScalarField::one(),
187        ]);
188        let (witness_poly, remainder_poly) = DenseOrSparsePolynomial::from(&polynomial)
189            .divide_with_q_and_r(&DenseOrSparsePolynomial::from(&divisor))
190            // the panic inside `divide_with_q_and_r` should never be reached, since the divisor
191            // polynomial is constructed right before and is set to not be zero. And the `.unwrap`
192            // should not give an error.
193            .unwrap();
194
195        let eval = if remainder_poly.is_zero() {
196            E::ScalarField::zero()
197        } else {
198            remainder_poly[0]
199        };
200
201        check_degree_is_too_large(witness_poly.degree(), params.powers_of_g.len())?;
202        let (num_leading_zeros, witness_coeffs) =
203            skip_first_zero_coeffs_and_convert_to_bigints(&witness_poly);
204        let proof = <E::G1 as VariableBaseMSM>::msm_bigint(
205            &params.powers_of_g[num_leading_zeros..],
206            &witness_coeffs,
207        );
208
209        Ok(Proof { eval, proof })
210    }
211
212    fn verify(
213        params: &Self::VerifierParams,
214        transcript: &mut impl Transcript<E::ScalarField>,
215        cm: &E::G1,
216        proof: &Self::Proof,
217    ) -> Result<(), Error> {
218        transcript.absorb_nonnative(cm);
219        let challenge = transcript.get_challenge();
220        Self::verify_with_challenge(params, challenge, cm, proof)
221    }
222
223    fn verify_with_challenge(
224        params: &Self::VerifierParams,
225        challenge: Self::Challenge,
226        cm: &E::G1,
227        proof: &Self::Proof,
228    ) -> Result<(), Error> {
229        if H {
230            return Err(Error::NotSupportedYet("hiding".to_string()));
231        }
232
233        // verify the KZG proof using arkworks method
234        let v = KZG10::<E, DensePolynomial<E::ScalarField>>::check(
235            params, // vk
236            &KZG10Commitment(cm.into_affine()),
237            challenge,
238            proof.eval,
239            &KZG10Proof::<E> {
240                w: proof.proof.into_affine(),
241                random_v: None,
242            },
243        )?;
244        if !v {
245            return Err(Error::CommitmentVerificationFail);
246        }
247        Ok(())
248    }
249}
250
251fn check_degree_is_too_large(
252    degree: usize,
253    num_powers: usize,
254) -> Result<(), ark_poly_commit::error::Error> {
255    let num_coefficients = degree + 1;
256    if num_coefficients > num_powers {
257        Err(ark_poly_commit::error::Error::TooManyCoefficients {
258            num_coefficients,
259            num_powers,
260        })
261    } else {
262        Ok(())
263    }
264}
265
266fn skip_first_zero_coeffs_and_convert_to_bigints<F: PrimeField, P: DenseUVPolynomial<F>>(
267    p: &P,
268) -> (usize, Vec<F::BigInt>) {
269    let mut num_leading_zeros = 0;
270    while num_leading_zeros < p.coeffs().len() && p.coeffs()[num_leading_zeros].is_zero() {
271        num_leading_zeros += 1;
272    }
273    let coeffs = convert_to_bigints(&p.coeffs()[num_leading_zeros..]);
274    (num_leading_zeros, coeffs)
275}
276
277fn convert_to_bigints<F: PrimeField>(p: &[F]) -> Vec<F::BigInt> {
278    ark_std::cfg_iter!(p)
279        .map(|s| s.into_bigint())
280        .collect::<Vec<_>>()
281}
