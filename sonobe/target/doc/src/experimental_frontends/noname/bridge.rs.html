1// From https://github.com/dmpierre/ark-noname/tree/feat/sonobe-integration
2use std::collections::HashMap;
3
4use ark_ff::PrimeField;
5use ark_r1cs_std::fields::fp::FpVar;
6use ark_relations::gr1cs::{
7    ConstraintSynthesizer, ConstraintSystemRef, LinearCombination, SynthesisError, Variable,
8};
9use noname::backends::{
10    r1cs::{GeneratedWitness, LinearCombination as NoNameLinearCombination, R1CS},
11    BackendField,
12};
13use noname::witness::CompiledCircuit;
14use num_bigint::BigUint;
15
16pub struct NonameSonobeCircuit<'a, 'b, 'c, F: PrimeField, BF: BackendField> {
17    pub compiled_circuit: CompiledCircuit<R1CS<BF>>,
18    pub witness: GeneratedWitness<BF>,
19    pub assigned_z_i: &'a Vec<FpVar<F>>,
20    pub assigned_external_inputs: &'b Vec<FpVar<F>>,
21    pub assigned_z_i1: &'c Vec<FpVar<F>>,
22}
23
24impl<'a, 'b, 'c, F: PrimeField, BF: BackendField> ConstraintSynthesizer<F>
25    for NonameSonobeCircuit<'a, 'b, 'c, F, BF>
26{
27    fn generate_constraints(self, cs: ConstraintSystemRef<F>) -> Result<(), SynthesisError> {
28        let public_io_length = self.assigned_z_i.len() * 2;
29        let external_inputs_len = self.assigned_external_inputs.len();
30
31        // we need to map noname r1cs indexes with sonobe
32        let mut idx_to_var = HashMap::new();
33
34        // for both the  z_i, z_i1 vectors, we assume that they have been assigned in the order
35        // with which it will appear in the witness
36        let mut z_i_pointer = 0;
37        let mut z_i1_pointer = 0;
38        let mut external_inputs_pointer = 0;
39
40        // arkworks assigns by default the 1 constant
41        // assumes witness is: [1, public_outputs, public_inputs, private_inputs, aux]
42        let witness_size = self.witness.witness.len();
43        for idx in 1..witness_size {
44            if idx <= public_io_length {
45                if idx <= self.assigned_z_i.len() {
46                    // in noname public outputs come first
47                    // we are in the case of public outputs (z_i1 vector)
48                    // those have already been assigned at specific indexes by sonobe
49                    let var = match &self.assigned_z_i1[z_i1_pointer] {
50                        FpVar::Var(allocated_fp) => allocated_fp.variable,
51                        _ => return Err(SynthesisError::Unsatisfiable),
52                    };
53                    idx_to_var.insert(idx, var);
54                    z_i1_pointer += 1;
55                } else {
56                    // we are in the case of public inputs (z_i values)
57                    // those have already been assigned at specific indexes by sonobe
58                    let var = match &self.assigned_z_i[z_i_pointer] {
59                        FpVar::Var(allocated_fp) => allocated_fp.variable,
60                        _ => return Err(SynthesisError::Unsatisfiable),
61                    };
62                    idx_to_var.insert(idx, var);
63                    z_i_pointer += 1;
64                }
65            } else if idx <= public_io_length + external_inputs_len {
66                // we are in the case of external inputs
67                // those have already been assigned at specific indexes
68                let var = match &self.assigned_external_inputs[external_inputs_pointer] {
69                    FpVar::Var(allocated_fp) => allocated_fp.variable,
70                    _ => return Err(SynthesisError::Unsatisfiable),
71                };
72                idx_to_var.insert(idx, var);
73                external_inputs_pointer += 1;
74            } else {
75                // we are in the case of auxiliary private inputs
76                // we need to assign those
77                let value: BigUint = Into::into(self.witness.witness[idx]);
78                let field_element = F::from(value);
79                let var = cs.new_witness_variable(|| Ok(field_element))?;
80                idx_to_var.insert(idx, var);
81            }
82        }
83
84        if (z_i_pointer != self.assigned_z_i.len())
85            || (external_inputs_pointer != self.assigned_external_inputs.len())
86        {
87            return Err(SynthesisError::AssignmentMissing);
88        }
89        let make_index = |index: usize| match index == 0 {
90            true => Ok(Variable::One),
91            false => {
92                let var = idx_to_var
93                    .get(&index)
94                    .ok_or(SynthesisError::AssignmentMissing)?;
95                Ok(var.to_owned())
96            }
97        };
98
99        let make_lc = |lc_data: NoNameLinearCombination<BF>| {
100            let mut lc = LinearCombination::<F>::zero();
101            for (cellvar, coeff) in lc_data.terms.into_iter() {
102                let idx = make_index(cellvar.index)?;
103                let coeff = F::from(Into::<BigUint>::into(coeff));
104
105                lc += (coeff, idx)
106            }
107
108            // add constant
109            let constant = F::from(Into::<BigUint>::into(lc_data.constant));
110            lc += (constant, make_index(0)?);
111            Ok(lc)
112        };
113
114        for constraint in self.compiled_circuit.circuit.backend.constraints {
115            let lc_a = make_lc(constraint.a)?;
116            let lc_b = make_lc(constraint.b)?;
117            let lc_c = make_lc(constraint.c)?;
118            cs.enforce_r1cs_constraint(|| lc_a, || lc_b, || lc_c)?;
119        }
120
121        Ok(())
122    }
123}
