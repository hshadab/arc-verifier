19// `L` indicates the length of the ExternalInputs vector of field elements.
20#[derive(Debug, Clone)]
21pub struct NonameFCircuit<F: PrimeField, BF: BackendField, const SL: usize, const EIL: usize> {
22    pub circuit: CompiledCircuit<R1CSNoname<BF>>,
23    _f: PhantomData<F>,
24}
25
26impl<F: PrimeField, BF: BackendField, const SL: usize, const EIL: usize> FCircuit<F>
27    for NonameFCircuit<F, BF, SL, EIL>
28{
29    type Params = String;
30    type ExternalInputs = VecF<F, EIL>;
31    type ExternalInputsVar = VecFpVar<F, EIL>;
32
33    fn new(code: Self::Params) -> Result<Self, Error> {
34        let compiled_circuit = compile_source_code::<BF>(&code).map_err(|_| {
35            Error::Other("Encountered an error while compiling a noname circuit".to_owned())
36        })?;
37        Ok(NonameFCircuit {
38            circuit: compiled_circuit,
39            _f: PhantomData,
40        })
41    }
42
43    fn state_len(&self) -> usize {
44        SL
45    }
46
47    fn generate_step_constraints(
48        &self,
49        cs: ConstraintSystemRef<F>,
50        _i: usize,
51        z_i: Vec<FpVar<F>>,
52        external_inputs: Self::ExternalInputsVar,
53    ) -> Result<Vec<FpVar<F>>, SynthesisError> {
54        let wtns_external_inputs =
55            NonameInputs::from_fpvars((&external_inputs.0, "external_inputs".to_string()));
56        let wtns_ivc_inputs = NonameInputs::from_fpvars((&z_i, "ivc_inputs".to_string()));
57        let noname_witness = self
58            .circuit
59            .generate_witness(wtns_ivc_inputs.0, wtns_external_inputs.0)
60            .map_err(|_| SynthesisError::Unsatisfiable)?;
61        let z_i1_end_index = z_i.len() + 1;
62        let assigned_z_i1: Vec<FpVar<F>> = (1..z_i1_end_index)
63            .map(|idx| -> Result<FpVar<F>, SynthesisError> {
64                // the assigned zi1 is of the same size than the initial zi and is located in the
65                // output of the witness vector
66                // we prefer to assign z_i1 here since (1) we have to return it, (2) we can't return
67                // anything with the `generate_constraints` method used below
68                let value: BigUint = Into::into(noname_witness.witness[idx]);
69                let field_element = F::from(value);
70                FpVar::<F>::new_witness(cs.clone(), || Ok(field_element))
71            })
72            .collect::<Result<Vec<FpVar<F>>, SynthesisError>>()?;
73
74        let noname_circuit = NonameSonobeCircuit {
75            compiled_circuit: self.circuit.clone(),
76            witness: noname_witness,
77            assigned_z_i: &z_i,
78            assigned_external_inputs: &external_inputs.0,
79            assigned_z_i1: &assigned_z_i1,
80        };
81        noname_circuit.generate_constraints(cs.clone())?;
82
83        Ok(assigned_z_i1)
84    }
85}

100    /// Native implementation of `NONAME_CIRCUIT_EXTERNAL_INPUTS`
101    fn external_inputs_step_native<F: PrimeField>(z_i: Vec<F>, external_inputs: Vec<F>) -> Vec<F> {
102        let xx = external_inputs[0] + z_i[0];
103        let yy = external_inputs[1] * z_i[1];
104        assert_eq!(yy, xx);
105        vec![xx, yy]
106    }
107
108    const NONAME_CIRCUIT_EXTERNAL_INPUTS: &str =
109        "fn main(pub ivc_inputs: [Field; 2], external_inputs: [Field; 2]) -> [Field; 2] {
110    let xx = external_inputs[0] + ivc_inputs[0];
111    let yy = external_inputs[1] * ivc_inputs[1];
112    assert_eq(yy, xx);
113    return [xx, yy];
114}";
115
116    const NONAME_CIRCUIT_NO_EXTERNAL_INPUTS: &str =
117        "fn main(pub ivc_inputs: [Field; 2]) -> [Field; 2] {
118    let out = ivc_inputs[0] * ivc_inputs[1];
119    return [out, ivc_inputs[1]];
120}";
121
122    #[test]
123    fn test_step_native() -> Result<(), Error> {
124        let cs = ConstraintSystem::<Fr>::new_ref();
125        let params = NONAME_CIRCUIT_EXTERNAL_INPUTS.to_owned();
126        // state length = 2, external inputs length = 2
127        let circuit = NonameFCircuit::<Fr, R1csBn254Field, 2, 2>::new(params)?;
128        let inputs_public = vec![Fr::from(2), Fr::from(5)];
129        let inputs_private = vec![Fr::from(8), Fr::from(2)];
130
131        let ivc_inputs_var =
132            Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(inputs_public.clone()))?;
133        let external_inputs_var =
134            Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(inputs_private.clone()))?;
135
136        let z_i1 = circuit.generate_step_constraints(
137            cs.clone(),
138            0,
139            ivc_inputs_var,
140            VecFpVar(external_inputs_var),
141        )?;
142        let z_i1_native = external_inputs_step_native(inputs_public, inputs_private);
143
144        assert_eq!(z_i1[0].value()?, z_i1_native[0]);
145        assert_eq!(z_i1[1].value()?, z_i1_native[1]);
146        Ok(())
147    }

149    #[test]
150    fn test_step_constraints() -> Result<(), Error> {
151        let cs = ConstraintSystem::<Fr>::new_ref();
152        let params = NONAME_CIRCUIT_EXTERNAL_INPUTS.to_owned();
153        // state length = 2, external inputs length = 2
154        let circuit = NonameFCircuit::<Fr, R1csBn254Field, 2, 2>::new(params)?;
155        let inputs_public = vec![Fr::from(2), Fr::from(5)];
156        let inputs_private = vec![Fr::from(8), Fr::from(2)];
157
158        let ivc_inputs_var = Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(inputs_public))?;
159        let external_inputs_var = Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(inputs_private))?;
160
161        let z_i1 = circuit.generate_step_constraints(
162            cs.clone(),
163            0,
164            ivc_inputs_var,
165            VecFpVar(external_inputs_var),
166        )?;
167        assert!(cs.is_satisfied()?);
168        assert_eq!(z_i1[0].value()?, Fr::from(10_u8));
169        assert_eq!(z_i1[1].value()?, Fr::from(10_u8));
170        Ok(())
171    }
