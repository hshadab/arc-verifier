16pub struct NonameInputs(pub JsonInputs);
17
18impl<F: PrimeField> From<(&Vec<F>, String)> for NonameInputs {
19    fn from(value: (&Vec<F>, String)) -> Self {
20        let (values, key) = value;
21        let mut inputs = HashMap::new();
22        if values.is_empty() {
23            NonameInputs(JsonInputs(inputs))
24        } else {
25            let field_elements: Vec<String> = values
26                .iter()
27                .map(|value| {
28                    if value.is_zero() {
29                        "0".to_string()
30                    } else {
31                        value.to_string()
32                    }
33                })
34                .collect();
35            inputs.insert(key, json!(field_elements));
36            NonameInputs(JsonInputs(inputs))
37        }
38    }
39}
40
41impl NonameInputs {
42    pub fn from_fpvars<F: PrimeField>(value: (&Vec<FpVar<F>>, String)) -> Self {
43        let (values, key) = value;
44        let mut inputs = HashMap::new();
45        if !values.is_empty() {
46            let field_elements: Vec<String> = values
47                .iter()
48                .map(|var| var.value().unwrap_or_default().to_string())
49                .collect::<Vec<String>>();
50            inputs.insert(key, json!(field_elements));
51        }
52        NonameInputs(JsonInputs(inputs))
53    }
54}

58pub fn compile_source_code<BF: BackendField>(
59    code: &str,
60) -> Result<CompiledCircuit<R1CS<BF>>, Error> {
61    let mut sources = Sources::new();
62
63    // parse the transitive dependency
64    let mut checker = TypeChecker::<R1CS<BF>>::new();
65    let _ = typecheck_next_file(
66        &mut checker,
67        None,
68        &mut sources,
69        "main.no".to_string(),
70        code.to_string(),
71        0,
72    )
73    .unwrap();
74    let r1cs = R1CS::<BF>::new();
75    // compile
76    CircuitWriter::generate_circuit(checker, r1cs).map_err(|_| {
77        Error::Other("Encountered an error while compiling a noname circuit".to_owned())
78    })
