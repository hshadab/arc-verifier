36impl<F: PrimeField, const SL: usize, const EIL: usize> FCircuit<F> for NoirFCircuit<F, SL, EIL> {
37    type Params = PathOrBin;
38    type ExternalInputs = VecF<F, EIL>;
39    type ExternalInputsVar = VecFpVar<F, EIL>;
40
41    fn new(source: Self::Params) -> Result<Self, Error> {
42        let input_string = match source {
43            PathOrBin::Path(path) => {
44                let file_path = path.with_extension("json");
45                std::fs::read(&file_path).map_err(|_| Error::Other(format!("{} is not a valid path\nRun either `nargo compile` to generate missing build artifacts or `nargo prove` to construct a proof", file_path.display())))?
46            }
47            PathOrBin::Bin(bin) => bin,
48        };
49        let program: ProgramArtifactGeneric<F> = serde_json::from_slice(&input_string)
50            .map_err(|err| Error::JSONSerdeError(err.to_string()))?;
51        let circuit: Circuit<GenericFieldElement<F>> = program.bytecode.functions[0].clone();
52        let ivc_input_length = circuit.public_parameters.0.len();
53        let ivc_return_length = circuit.return_values.0.len();
54
55        if ivc_input_length != ivc_return_length {
56            return Err(Error::NotSameLength(
57                "IVC input: ".to_string(),
58                ivc_input_length,
59                "IVC output: ".to_string(),
60                ivc_return_length,
61            ));
62        }
63
64        Ok(NoirFCircuit { circuit })
65    }
66
67    fn state_len(&self) -> usize {
68        SL
69    }
70
71    fn generate_step_constraints(
72        &self,
73        cs: ConstraintSystemRef<F>,
74        _i: usize,
75        z_i: Vec<FpVar<F>>,
76        external_inputs: Self::ExternalInputsVar, // inputs that are not part of the state
77    ) -> Result<Vec<FpVar<F>>, SynthesisError> {
78        let mut acvm = ACVM::new(
79            &StubbedBlackBoxSolver,
80            &self.circuit.opcodes,
81            WitnessMap::new(),
82            &[],
83            &[],
84        );
85
86        let mut already_assigned_witness_values = HashMap::new();
87
88        self.circuit.public_parameters.0.iter().for_each(|witness| {
89            let idx: usize = witness.as_usize();
90            let witness = AcvmWitness(witness.witness_index());
91            already_assigned_witness_values.insert(witness, &z_i[idx]);
92
93            let val = z_i[idx].value().unwrap_or_default();
94
95            let f = GenericFieldElement::<F>::from_repr(val);
96            acvm.overwrite_witness(witness, f);
97        });
98
99        // write witness values for external_inputs
100        self.circuit.private_parameters.iter().for_each(|witness| {
101            let idx = witness.as_usize() - z_i.len();
102            let witness = AcvmWitness(witness.witness_index());
103            already_assigned_witness_values.insert(witness, &external_inputs.0[idx]);
104
105            let val = external_inputs.0[idx].value().unwrap_or_default();
106
107            let f = GenericFieldElement::<F>::from_repr(val);
108            acvm.overwrite_witness(witness, f);
109        });
110
111        // computes the witness
112        let _ = acvm.solve();
113        let witness_map = acvm.finalize();
114
115        // get the z_{i+1} output state
116        let assigned_z_i1 = self
117            .circuit
118            .return_values
119            .0
120            .iter()
121            .map(|witness| {
122                let noir_field_element = witness_map
123                    .get(witness)
124                    .ok_or(SynthesisError::AssignmentMissing)?;
125                FpVar::<F>::new_witness(cs.clone(), || Ok(noir_field_element.into_repr()))
126            })
127            .collect::<Result<Vec<FpVar<F>>, SynthesisError>>()?;
128
129        // initialize circuit and set already assigned values
130        let mut acir_circuit = AcirCircuitSonobe::from((&self.circuit, witness_map));
131        acir_circuit.already_assigned_witnesses = already_assigned_witness_values;
132
133        acir_circuit.generate_constraints(cs.clone())?;
134
135        Ok(assigned_z_i1)
136    }
137}

152    /// Native implementation of `src/noir/test_folder/test_circuit`
153    fn external_inputs_step_native<F: PrimeField>(z_i: Vec<F>, external_inputs: Vec<F>) -> Vec<F> {
154        let xx = external_inputs[0] * z_i[0];
155        let yy = external_inputs[1] * z_i[1];
156        vec![xx, yy]
157    }
158
159    #[test]
160    fn test_step_native() -> Result<(), Error> {
161        let inputs = vec![Fr::from(2), Fr::from(5)];
162        let res = external_inputs_step_native(inputs.clone(), inputs);
163        assert_eq!(res, vec![Fr::from(4), Fr::from(25)]);
164        Ok(())
165    }

167    #[test]
168    fn test_step_constraints() -> Result<(), Error> {
169        let cs = ConstraintSystem::<Fr>::new_ref();
170        let cur_path = env::current_dir()?;
171        // external inputs length: 2, state length: 2
172        let noirfcircuit = NoirFCircuit::<Fr, 2, 2>::new(
173            cur_path
174                .join("src/noir/test_folder/test_circuit/target/test_circuit.json")
175                .into(),
176        )?;
177        let inputs = vec![Fr::from(2), Fr::from(5)];
178        let z_i = Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(inputs.clone()))?;
179        let external_inputs = Vec::<FpVar<Fr>>::new_witness(cs.clone(), || Ok(inputs))?;
180        let output = noirfcircuit.generate_step_constraints(
181            cs.clone(),
182            0,
183            z_i,
184            VecFpVar(external_inputs),
185        )?;
186        assert_eq!(output[0].value()?, Fr::from(4));
187        assert_eq!(output[1].value()?, Fr::from(25));
188        Ok(())
189    }
