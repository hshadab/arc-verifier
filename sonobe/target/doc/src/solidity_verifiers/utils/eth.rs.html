1//! This module provides a trait and implementations for converting Rust types
2//! to EVM calldata.
3use ark_ec::{
4    pairing::Pairing,
5    short_weierstrass::{Affine, Projective, SWCurveConfig},
6    AffineRepr, CurveGroup,
7};
8use ark_ff::{BigInteger, Fp, Fp2, Fp2Config, FpConfig, PrimeField};
9use ark_groth16::Proof;
10
11pub trait ToEth {
12    fn to_eth(&self) -> Vec<u8>;
13}
14
15impl<T: ToEth> ToEth for [T] {
16    fn to_eth(&self) -> Vec<u8> {
17        self.iter().flat_map(ToEth::to_eth).collect()
18    }
19}
20
21impl ToEth for u8 {
22    fn to_eth(&self) -> Vec<u8> {
23        vec![*self]
24    }
25}
26
27impl<P: FpConfig<N>, const N: usize> ToEth for Fp<P, N> {
28    fn to_eth(&self) -> Vec<u8> {
29        self.into_bigint().to_bytes_be()
30    }
31}
32
33impl<P: Fp2Config<Fp: ToEth>> ToEth for Fp2<P> {
34    fn to_eth(&self) -> Vec<u8> {
35        [self.c1.to_eth(), self.c0.to_eth()].concat()
36    }
37}
38
39impl<P: SWCurveConfig<BaseField: ToEth>> ToEth for Affine<P> {
40    fn to_eth(&self) -> Vec<u8> {
41        // the encoding of the additive identity is [0, 0] on the EVM
42        let (x, y) = self.xy().unwrap_or_default();
43
44        [x.to_eth(), y.to_eth()].concat()
45    }
46}
47
48impl<P: SWCurveConfig<BaseField: ToEth>> ToEth for Projective<P> {
49    fn to_eth(&self) -> Vec<u8> {
50        self.into_affine().to_eth()
51    }
52}
53
54impl<E: Pairing<G1Affine: ToEth, G2Affine: ToEth>> ToEth for Proof<E> {
55    fn to_eth(&self) -> Vec<u8> {
56        [self.a.to_eth(), self.b.to_eth(), self.c.to_eth()].concat()
57    }
58}
