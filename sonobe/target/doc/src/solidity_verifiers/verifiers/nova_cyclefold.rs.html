42impl From<NovaCycleFoldVerifierKey> for NovaCycleFoldDecider {
43    fn from(value: NovaCycleFoldVerifierKey) -> Self {
44        let groth16_verifier = Groth16Verifier::from(value.g16_vk);
45        let public_inputs_len = groth16_verifier.gamma_abc_len;
46        let bits_per_limb = NonNativeUintVar::<Fq>::bits_per_limb();
47        Self {
48            pp_hash: value.pp_hash,
49            groth16_verifier,
50            kzg10_verifier: KZG10Verifier::from(value.kzg_vk),
51            z_len: value.z_len,
52            public_inputs_len,
53            num_limbs: (250_f32 / (bits_per_limb as f32)).ceil() as usize,
54            bits_per_limb,
55        }
56    }
57}

92// implements From assuming that the 'batchCheck' method from the KZG10 template will not be used
93// in the NovaCycleFoldDecider verifier contract
94impl
95    From<(
96        DeciderVerifierParam<G1Projective, ArkKZG10VerifierKey<Bn254>, ArkG16VerifierKey<Bn254>>,
97        usize,
98    )> for NovaCycleFoldVerifierKey
99{
100    fn from(
101        value: (
102            DeciderVerifierParam<
103                G1Projective,
104                ArkKZG10VerifierKey<Bn254>,
105                ArkG16VerifierKey<Bn254>,
106            >,
107            usize,
108        ),
109    ) -> Self {
110        let decider_vp = value.0;
111        let g16_vk = Groth16VerifierKey::from(decider_vp.snark_vp);
112        // pass `Vec::new()` since batchCheck will not be used
113        let kzg_vk = KZG10VerifierKey::from((decider_vp.cs_vp, Vec::new()));
114        Self {
115            pp_hash: decider_vp.pp_hash,
116            g16_vk,
117            kzg_vk,
118            z_len: value.1,
119        }
120    }
121}

213    /// This is the circuit that we want to fold, it implements the FCircuit trait. The parameter z_i
214    /// denotes the current state which contains 5 elements, and z_{i+1} denotes the next state which
215    /// we get by applying the step.
216    /// In this example we set z_i and z_{i+1} to have five elements, and at each step we do different
217    /// operations on each of them.

219    pub struct MultiInputsFCircuit<F: PrimeField> {
220        _f: PhantomData<F>,
221    }
222    impl<F: PrimeField> FCircuit<F> for MultiInputsFCircuit<F> {
223        type Params = ();
224        type ExternalInputs = ();
225        type ExternalInputsVar = ();
226
227        fn new(_params: Self::Params) -> Result<Self, Error> {
228            Ok(Self { _f: PhantomData })
229        }
230        fn state_len(&self) -> usize {
231            5
232        }
233        /// generates the constraints for the step of F for the given z_i
234        fn generate_step_constraints(
235            &self,
236            cs: ConstraintSystemRef<F>,
237            _i: usize,
238            z_i: Vec<FpVar<F>>,
239            _external_inputs: Self::ExternalInputsVar,
240        ) -> Result<Vec<FpVar<F>>, SynthesisError> {
241            let four = FpVar::<F>::new_constant(cs.clone(), F::from(4u32))?;
242            let forty = FpVar::<F>::new_constant(cs.clone(), F::from(40u32))?;
243            let onehundred = FpVar::<F>::new_constant(cs.clone(), F::from(100u32))?;
244            let a = z_i[0].clone() + four.clone();
245            let b = z_i[1].clone() + forty.clone();
246            let c = z_i[2].clone() * four;
247            let d = z_i[3].clone() * forty;
248            let e = z_i[4].clone() + onehundred;
249
250            Ok(vec![a, b, c, d, e])
251        }
252    }

294    fn init_params<FC: FCircuit<Fr, Params = ()>>(
295    ) -> ((FS_PP<FC>, FS_VP<FC>), (DECIDER_PP<FC>, DECIDER_VP<FC>)) {
296        let mut rng = ark_std::rand::rngs::OsRng;
297        let poseidon_config = poseidon_canonical_config::<Fr>();
298
299        let f_circuit = FC::new(()).unwrap();
300        let prep_param =
301            PreprocessorParam::<G1, G2, FC, KZG<'static, Bn254>, Pedersen<G2>, false>::new(
302                poseidon_config,
303                f_circuit.clone(),
304            );
305        let nova_params = NOVA::preprocess(&mut rng, &prep_param).unwrap();
306        let decider_params =
307            DECIDER::<FC>::preprocess(&mut rng, (nova_params.clone(), f_circuit.state_len()))
308                .unwrap();
309
310        (nova_params, decider_params)
311    }

313    fn interact_with_contract<'a, FC: FCircuit<Fr>>(
314        nova_cyclefold_verifier_bytecode: &[u8],
315        nova: &NOVA<FC>,
316        proof: &Proof<G1Projective, KZG<Bn254>, Groth16<Bn254>>,
317        mode: NovaVerificationMode,
318    ) {
319        let mut evm = Evm::default();
320        let verifier_address = evm.create(nova_cyclefold_verifier_bytecode.to_vec());
321
322        let calldata: Vec<u8> = prepare_calldata_for_nova_cyclefold_verifier(
323            mode,
324            nova.i,
325            nova.z_0.clone(),
326            nova.z_i.clone(),
327            &nova.U_i,
328            &nova.u_i,
329            proof,
330        )
331        .unwrap();
332
333        let (_, output) = evm.call(verifier_address, calldata.clone());
334        assert_eq!(*output.last().unwrap(), 1);
335
336        // change i to make calldata invalid, placed between bytes 4 - 35
337        let mut invalid_calldata = calldata.clone();
338        invalid_calldata[35] += 1;
339        let (_, output) = evm.call(verifier_address, invalid_calldata.clone());
340        assert_eq!(*output.last().unwrap(), 0);
341
342        // change z_0 to make the EVM check fail, placed between bytes 35 - 67
343        let mut invalid_calldata = calldata.clone();
344        invalid_calldata[67] += 1;
345        let (_, output) = evm.call(verifier_address, invalid_calldata.clone());
346        assert_eq!(*output.last().unwrap(), 0);
347
348        // change z_i to make the EVM check fail, placed between bytes 68 - 100
349        let mut invalid_calldata = calldata.clone();
350        invalid_calldata[99] += 1;
351        let (_, output) = evm.call(verifier_address, invalid_calldata.clone());
352        assert_eq!(*output.last().unwrap(), 0);
353    }

355    /// This function allows to define which FCircuit to use for the test, and how many prove_step
356    /// rounds to perform.
357    /// Actions performed by this test:
358    /// - runs the NovaCycleFold folding scheme for the given FCircuit and n_steps times
359    /// - generates a DeciderEth proof, and executes it through the EVM
360    /// - modifies the calldata and checks that it does not pass the EVM check
361    /// - modifies the z_0 and checks that it does not pass the EVM check
362    #[allow(clippy::type_complexity)]
363    fn nova_cyclefold_solidity_verifier_opt<FC: FCircuit<Fr, Params = ()>>(
364        fs_params: (FS_PP<FC>, FS_VP<FC>),
365        decider_params: (DECIDER_PP<FC>, DECIDER_VP<FC>),
366        z_0: Vec<Fr>,
367        n_steps: usize,
368    ) {
369        let (decider_pp, decider_vp) = decider_params;
370
371        let f_circuit = FC::new(()).unwrap();
372
373        let nova_cyclefold_vk =
374            NovaCycleFoldVerifierKey::from((decider_vp.clone(), f_circuit.state_len()));
375
376        let mut rng = ark_std::rand::rngs::OsRng;
377
378        let mut nova = NOVA::<FC>::init(&fs_params, f_circuit, z_0).unwrap();
379        for _ in 0..n_steps {
380            nova.prove_step(&mut rng, FC::ExternalInputs::default(), None)
381                .unwrap();
382        }
383
384        let start = Instant::now();
385        let proof = DECIDER::<FC>::prove(rng, decider_pp, nova.clone()).unwrap();
386        println!("generated Decider proof: {:?}", start.elapsed());
387
388        let verified = DECIDER::<FC>::verify(
389            decider_vp,
390            nova.i,
391            nova.z_0.clone(),
392            nova.z_i.clone(),
393            &nova.U_i.get_commitments(),
394            &nova.u_i.get_commitments(),
395            &proof,
396        )
397        .unwrap();
398        assert!(verified);
399
400        let decider_solidity_code = get_decider_template_for_cyclefold_decider(nova_cyclefold_vk);
401
402        let nova_cyclefold_verifier_bytecode =
403            compile_solidity(decider_solidity_code, "NovaDecider");
404
405        for mode in [Explicit, Opaque, OpaqueWithInputs] {
406            interact_with_contract(&nova_cyclefold_verifier_bytecode, &nova, &proof, mode);
407        }
408    }
