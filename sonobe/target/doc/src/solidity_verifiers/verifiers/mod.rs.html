21pub trait ProtocolVerifierKey: CanonicalDeserialize + CanonicalSerialize {
22    const PROTOCOL_NAME: &'static str;
23
24    fn serialize_name<W: Write>(&self, writer: &mut W) -> Result<(), SerializationError> {
25        Self::PROTOCOL_NAME
26            .to_string()
27            .serialize_uncompressed(writer)
28    }
29
30    fn serialize_protocol_verifier_key<W: Write>(
31        &self,
32        writer: &mut W,
33    ) -> Result<(), SerializationError> {
34        self.serialize_name(writer)?;
35        self.serialize_compressed(writer)
36    }
37    fn deserialize_protocol_verifier_key<R: Read + Copy>(
38        mut reader: R,
39    ) -> Result<Self, SerializationError> {
40        let name: String = String::deserialize_uncompressed(&mut reader)?;
41        let data = Self::deserialize_compressed(&mut reader)?;
42
43        if name != Self::PROTOCOL_NAME {
44            return Err(SerializationError::InvalidData);
45        }
46
47        Ok(data)
48    }
49
50    fn render_as_template(self, pragma: Option<String>) -> Vec<u8>;
51}
