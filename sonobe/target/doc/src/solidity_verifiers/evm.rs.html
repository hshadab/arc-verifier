16pub fn save_solidity(name: impl AsRef<str>, solidity: &str) {
17    let curdir = PathBuf::from(".");
18    let curdir_abs_path = fs::canonicalize(curdir).expect("Failed to get current directory");
19    let curdir_abs_path = curdir_abs_path
20        .to_str()
21        .expect("Failed to convert path to string");
22    let dir_generated = format!("{curdir_abs_path}/generated");
23    create_dir_all(dir_generated.clone()).unwrap();
24    File::create(format!("{}/{}", dir_generated, name.as_ref()))
25        .unwrap()
26        .write_all(solidity.as_bytes())
27        .unwrap();
28}

30/// Compile solidity with `--via-ir` flag, then return creation bytecode.
31///
32/// # Panics
33/// Panics if executable `solc` can not be found, or compilation fails.
34pub fn compile_solidity(solidity: impl AsRef<[u8]>, contract_name: &str) -> Vec<u8> {
35    let mut process = match Command::new("solc")
36        .stdin(Stdio::piped())
37        .stdout(Stdio::piped())
38        .stderr(Stdio::piped())
39        .arg("--bin")
40        .arg("--optimize")
41        .arg("-")
42        .spawn()
43    {
44        Ok(process) => process,
45        Err(err) if err.kind() == io::ErrorKind::NotFound => {
46            panic!("Command 'solc' not found");
47        }
48        Err(err) => {
49            panic!("Failed to spawn process with command 'solc':\n{err}");
50        }
51    };
52    process
53        .stdin
54        .take()
55        .unwrap()
56        .write_all(solidity.as_ref())
57        .unwrap();
58    let output = process.wait_with_output().unwrap();
59    let stdout = str::from_utf8(&output.stdout).unwrap();
60    if let Some(binary) = find_binary(stdout, contract_name) {
61        binary
62    } else {
63        panic!(
64            "Compilation fails:\n{}",
65            str::from_utf8(&output.stderr).unwrap()
66        )
67    }
68}

70/// Find binary from `stdout` with given `contract_name`.
71/// `contract_name` is provided since `solc` may compile multiple contracts or libraries.
72/// hence, we need to find the correct binary.
73fn find_binary(stdout: &str, contract_name: &str) -> Option<Vec<u8>> {
74    let intro_str = format!("======= <stdin>:{contract_name} =======\nBinary:\n");
75    let start = stdout.find(&intro_str)?;
76    let end = stdout[start + intro_str.len()..]
77        .find('\n')
78        .map(|pos| pos + start + intro_str.len())
79        .unwrap_or(stdout.len());
80    let binary_section = stdout[start + intro_str.len()..end].trim();
81    Some(hex::decode(binary_section).unwrap())
82}

86pub struct Evm<'a> {

88}

90impl<'a> Default for Evm<'a> {

92        Self {
93            evm: EvmBuilder::default().with_db(InMemoryDB::default()).build(),
94        }
95    }

98impl<'a> Evm<'a> {
99    /// Apply create transaction with given `bytecode` as creation bytecode.
100    /// Return created `address`.
101    ///
102    /// # Panics
103    /// Panics if execution reverts or halts unexpectedly.
104    pub fn create(&mut self, bytecode: Vec<u8>) -> Address {
105        let (_, output) = self.transact_success_or_panic(TxEnv {
106            gas_limit: u64::MAX,
107            transact_to: TransactTo::Create,
108            data: bytecode.into(),
109            ..Default::default()
110        });
111        match output {
112            Output::Create(_, Some(address)) => address,
113            _ => unreachable!(),
114        }
115    }
116
117    /// Apply call transaction to given `address` with `calldata`.
118    /// Returns `gas_used` and `return_data`.
119    ///
120    /// # Panics
121    /// Panics if execution reverts or halts unexpectedly.
122    pub fn call(&mut self, address: Address, calldata: Vec<u8>) -> (u64, Vec<u8>) {
123        let (gas_used, output) = self.transact_success_or_panic(TxEnv {
124            gas_limit: u64::MAX,
125            transact_to: TransactTo::Call(address),
126            data: calldata.into(),
127            ..Default::default()
128        });
129        match output {
130            Output::Call(output) => (gas_used, output.into()),
131            _ => unreachable!(),
132        }
133    }
134
135    fn transact_success_or_panic(&mut self, tx: TxEnv) -> (u64, Output) {
136        *self.evm.tx_mut() = tx;
137        let result = self.evm.transact_commit().unwrap();
138        match result {
139            ExecutionResult::Success {
140                gas_used,
141                output,
142                logs,
143                ..
144            } => {
145                if !logs.is_empty() {
146                    println!("--- logs from {} ---", logs[0].address);
147                    for (log_idx, log) in logs.iter().enumerate() {
148                        println!("log#{log_idx}");
149                        for (topic_idx, topic) in log.topics().iter().enumerate() {
150                            println!("  topic{topic_idx}: {topic:?}");
151                        }
152                    }
153                    println!("--- end ---");
154                }
155                (gas_used, output)
156            }
157            ExecutionResult::Revert { gas_used, output } => (gas_used, Output::Call(output)),
158            ExecutionResult::Halt { reason, gas_used } => panic!(
159                "Transaction halts unexpectedly with gas_used {gas_used} and reason {reason:?}"
160            ),
161        }
162    }
163}
