27// Would be nice to link this to the `Template` or `ProtocolVerifierKey` traits.
28// Sadly, this requires Boxing with `dyn` or similar which would complicate the code more than is actually required.
29impl Protocol {
30    pub(crate) fn render(
31        &self,
32        data: &[u8],
33        pragma: Option<String>,
34    ) -> Result<Vec<u8>, SerializationError> {
35        match self {
36            Self::Groth16 => Ok(Groth16VerifierKey::deserialize_protocol_verifier_key(data)?
37                .render_as_template(pragma)),
38
39            Self::Kzg => Ok(KZG10VerifierKey::deserialize_protocol_verifier_key(data)?
40                .render_as_template(pragma)),
41            Self::NovaCycleFold => Ok(NovaCycleFoldVerifierKey::deserialize_protocol_verifier_key(
42                data,
43            )?
44            .render_as_template(pragma)),
45        }
46    }
47}

90#[command(author = "0xPARC & PSE", version, about = ABOUT, long_about = Some(LONG_ABOUT))]
91#[command(propagate_version = true)]
92/// A tool to create Solidity Contracts which act as verifiers for the major Folding Schemes implemented
93/// within the `sonobe` repo.
94pub(crate) struct Cli {
95    #[command(flatten)]
96    pub verbosity: clap_verbosity_flag::Verbosity,
97
98    /// Selects the protocol for which we want to generate the Solidity Verifier contract.
99    #[arg(short = 'p', long, value_enum, rename_all = "lower")]
100    pub protocol: Protocol,
101
102    #[arg(short = 'o', long, default_value=get_default_out_path().into_os_string())]
103    /// Sets the output path for all the artifacts generated by the command.
104    pub out: PathBuf,
105
106    #[arg(short = 'k', long)]
107    /// Sets the input path for the file containing the verifier key required by the protocol chosen such that the verification contract can be generated.
108    pub protocol_vk: PathBuf,
109
110    /// Selects the Solidity compiler version to be set in the Solidity Verifier contract artifact.
111    #[arg(long, default_value=None)]
112    pub pragma: Option<String>,
113}
