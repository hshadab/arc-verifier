pub fn compute_q(&self, z: &[F]) -> Result<VirtualPolynomial<F>, Error>

Computes q(x) = \sum^q c_i * \prod_{j \in S_i} ( \sum_{y \in {0,1}^s’} M_j(x, y) * z(y) ) polynomial over x

Source

pub fn compute_Q( &self, z: &[F], beta: &[F], ) -> Result<VirtualPolynomial<F>, Error>

Computes Q(x) = eq(beta, x) * q(x) = eq(beta, x) * \sum^q c_i * \prod_{j \in S_i} ( \sum_{y \in {0,1}^s’} M_j(x, y) * z(y) ) polynomial over x
